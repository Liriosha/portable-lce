52std::uint32_t SkinManager::getPlayerSkinId(int iPad,
53 GAME_SETTINGS** gameSettingsA,
54 DLCManager& dlcManager) {
55 DLCPack* Pack = nullptr;
56 DLCSkinFile* skinFile = nullptr;
57 std::uint32_t dwSkin = gameSettingsA[iPad]->dwSelectedSkin;
58 char chars[256];
59
60 if (GET_IS_DLC_SKIN_FROM_BITMASK(dwSkin)) {
61 snprintf(chars, 256, "dlcskin%08d.png",
62 GET_DLC_SKIN_ID_FROM_BITMASK(dwSkin));
63
64 Pack = dlcManager.getPackContainingSkin(chars);
65
66 if (Pack) {
67 skinFile = Pack->getSkinFile(chars);
68
69 bool bSkinIsFree =
70 skinFile->getParameterAsBool(DLCManager::e_DLCParamType_Free);
71 bool bLicensed = Pack->hasPurchasedFile(DLCManager::e_DLCType_Skin,
72 skinFile->getPath());
73
74 if (bSkinIsFree || bLicensed) {
75 return dwSkin;
76 } else {
77 return 0;
78 }
79 }
80 }
81
82 return dwSkin;
83}

155void SkinManager::validateFavoriteSkins(int iPad, GAME_SETTINGS** gameSettingsA,
156 DLCManager& dlcManager) {
157 unsigned int uiCount = getPlayerFavoriteSkinsCount(iPad, gameSettingsA);
158
159 unsigned int uiValidSkin = 0;
160 char chars[256];
161
162 for (unsigned int i = 0; i < uiCount; i++) {
163 snprintf(chars, 256, "dlcskin%08d.png",
164 getPlayerFavoriteSkin(iPad, i, gameSettingsA));
165
166 DLCPack* pDLCPack = dlcManager.getPackContainingSkin(chars);
167
168 if (pDLCPack != nullptr) {
169 DLCSkinFile* pSkinFile = pDLCPack->getSkinFile(chars);
170
171 if (pDLCPack->hasPurchasedFile(DLCManager::e_DLCType_Skin, "") ||
172 (pSkinFile && pSkinFile->isFree())) {
173 gameSettingsA[iPad]->uiFavoriteSkinA[uiValidSkin++] =
174 gameSettingsA[iPad]->uiFavoriteSkinA[i];
175 }
176 }
177 }
178
179 for (unsigned int i = uiValidSkin; i < MAX_FAVORITE_SKINS; i++) {
180 gameSettingsA[iPad]->uiFavoriteSkinA[i] = 0xFFFFFFFF;
181 }
182}

197void SkinManager::addMemoryTextureFile(const std::string& wName,
198 std::uint8_t* pbData,
199 unsigned int byteCount) {
200 std::lock_guard<std::mutex> lock(csMemFilesLock);
201 PMEMDATA pData = nullptr;
202 auto it = m_MEM_Files.find(wName);
203 if (it != m_MEM_Files.end()) {
204#if !defined(_CONTENT_PACKAGE)
205 printf("Incrementing the memory texture file count for %s\n",
206 wName.c_str());
207#endif
208 pData = (*it).second;
209
210 if (pData->byteCount == 0 && byteCount != 0) {
211 if (pData->pbData != nullptr) delete[] pData->pbData;
212
213 pData->pbData = pbData;
214 pData->byteCount = byteCount;
215 }
216
217 ++pData->ucRefCount;
218 return;
219 }
220
221 pData = new MEMDATA();
222 pData->pbData = pbData;
223 pData->byteCount = byteCount;
224 pData->ucRefCount = 1;
225
226 m_MEM_Files[wName] = pData;
227}

229void SkinManager::removeMemoryTextureFile(const std::string& wName) {
230 std::lock_guard<std::mutex> lock(csMemFilesLock);
231
232 auto it = m_MEM_Files.find(wName);
233 if (it != m_MEM_Files.end()) {
234#if !defined(_CONTENT_PACKAGE)
235 printf("Decrementing the memory texture file count for %s\n",
236 wName.c_str());
237#endif
238 PMEMDATA pData = (*it).second;
239 --pData->ucRefCount;
240 if (pData->ucRefCount <= 0) {
241#if !defined(_CONTENT_PACKAGE)
242 printf("Erasing the memory texture file data for %s\n",
243 wName.c_str());
244#endif
245 delete pData;
246 m_MEM_Files.erase(wName);
247 }
248 }

413std::uint32_t SkinManager::getSkinIdFromPath(const std::string& skin) {
414 bool dlcSkin = false;
415 unsigned int skinId = 0;
416
417 if (skin.size() >= 14) {
418 dlcSkin = skin.substr(0, 3).compare("dlc") == 0;
419
420 std::string skinValue = skin.substr(7, skin.size());
421 skinValue = skinValue.substr(0, skinValue.find_first_of('.'));
422
423 std::stringstream ss;
424 if (dlcSkin)
425 ss << std::dec << skinValue.c_str();
426 else
427 ss << std::hex << skinValue.c_str();
428 ss >> skinId;
429
430 skinId = MAKE_SKIN_BITMASK(dlcSkin, skinId);
431 }
432 return skinId;
433}

435std::string SkinManager::getSkinPathFromId(std::uint32_t skinId) {
436 char chars[256];
437 if (GET_IS_DLC_SKIN_FROM_BITMASK(skinId)) {
438 snprintf(chars, 256, "dlcskin%08d.png",
439 GET_DLC_SKIN_ID_FROM_BITMASK(skinId));
440 } else {
441 std::uint32_t ugcSkinIndex = GET_UGC_SKIN_ID_FROM_BITMASK(skinId);
442 std::uint32_t defaultSkinIndex =
443 GET_DEFAULT_SKIN_ID_FROM_BITMASK(skinId);
444 if (ugcSkinIndex == 0) {
445 snprintf(chars, 256, "defskin%08X.png", defaultSkinIndex);
446 } else {
447 snprintf(chars, 256, "ugcskin%08X.png", ugcSkinIndex);
448 }
449 }
450 return chars;
