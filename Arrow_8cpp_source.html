70Arrow::Arrow(Level* level, std::shared_ptr<LivingEntity> mob,
71 std::shared_ptr<LivingEntity> target, float power,
72 float uncertainty)
73 : Entity(level) {
74 _init();
75
76 viewScale = 10;
77 owner = mob;
78 if (mob->instanceof(eTYPE_PLAYER)) pickup = PICKUP_ALLOWED;
79
80 y = mob->y + mob->getHeadHeight() - 0.1f;
81
82 double xd = target->x - mob->x;
83 double yd = (target->y + target->getHeadHeight() - 0.7f) - y;
84 double zd = target->z - mob->z;
85 double sd = sqrt(xd * xd + zd * zd);
86 if (sd < 0.0000001) return;
87
88 float yRot = (float)(atan2(zd, xd) * 180 / std::numbers::pi) - 90;
89 float xRot = (float)-(atan2(yd, sd) * 180 / std::numbers::pi);
90
91 double xdn = xd / sd;
92 double zdn = zd / sd;
93 moveTo(mob->x + xdn, y, mob->z + zdn, yRot, xRot);
94 heightOffset = 0;
95
96 float yo = (float)sd * 0.2f;
97 shoot(xd, yd + yo, zd, power, uncertainty);
98}

132 yd = -sinf(xRot / 180 * std::numbers::pi);

139void Arrow::shoot(double xd, double yd, double zd, float pow,
140 float uncertainty) {
141 float dist = (float)sqrt(xd * xd + yd * yd + zd * zd);
142
143 xd /= dist;
144 yd /= dist;
145 zd /= dist;
146
147 xd += (random->nextGaussian() * (random->nextBoolean() ? -1 : 1)) *
148 0.0075f * uncertainty;
149 yd += (random->nextGaussian() * (random->nextBoolean() ? -1 : 1)) *
150 0.0075f * uncertainty;
151 zd += (random->nextGaussian() * (random->nextBoolean() ? -1 : 1)) *
152 0.0075f * uncertainty;
153
154 xd *= pow;
155 yd *= pow;
156 zd *= pow;
157
158 this->xd = xd;
159 this->yd = yd;
160 this->zd = zd;
161
162 double sd = sqrt(xd * xd + zd * zd);
163
164 yRotO = yRot = (float)(atan2(xd, zd) * 180 / std::numbers::pi);
165 xRotO = xRot = (float)(atan2(yd, sd) * 180 / std::numbers::pi);
166 life = 0;
167}
