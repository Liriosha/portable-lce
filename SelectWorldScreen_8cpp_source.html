99void SelectWorldScreen::buttonClicked(Button* button) {
100 Log::info("SelectWorldScreen::buttonClicked START\n");
101 if (!button->active) return;
102 if (button->id == BUTTON_DELETE_ID) {
103 std::string worldName = getWorldName(selectedWorld);
104 if (worldName != "") {
105 isDeleting = true;
106
107 Language* language = Language::getInstance();
108 std::string title =
109 language->getElement("selectWorld.deleteQuestion");
110 std::string warning =
111 "'" + worldName + "' " +
112 language->getElement("selectWorld.deleteWarning");
113 std::string yes = language->getElement("selectWorld.deleteButton");
114 std::string no = language->getElement("gui.cancel");
115
116 ConfirmScreen* confirmScreen =
117 new ConfirmScreen(this, title, warning, yes, no, selectedWorld);
118 minecraft->setScreen(confirmScreen);
119 }
120 } else if (button->id == BUTTON_SELECT_ID) {
121 worldSelected(selectedWorld);
122 } else if (button->id == BUTTON_CREATE_ID) {
123 minecraft->setScreen(new CreateWorldScreen(this));
124 } else if (button->id == BUTTON_RENAME_ID) {
125 minecraft->setScreen(
126 new RenameWorldScreen(this, getWorldId(selectedWorld)));
127 } else if (button->id == BUTTON_CANCEL_ID) {
128 Log::info(
129 "SelectWorldScreen::buttonClicked 'Cancel' "
130 "minecraft->setScreen(lastScreen)\n");
131 minecraft->setScreen(lastScreen);
132 } else {
133 worldSelectionList->buttonClicked(button);
134 }
135}

251void SelectWorldScreen::WorldSelectionList::renderItem(int i, int x, int y,
252 int h, Tesselator* t) {
253 LevelSummary* levelSummary = parent->levelList->at(i);
254
255 std::string name = levelSummary->getLevelName();
256 if (name.length() == 0) {
257 name = parent->worldLang + " " + toWString<int>(i + 1);
258 }
259
260 std::string id = levelSummary->getLevelId();
261
262 // levelSummary->getLastPlayed() is milliseconds since the FILETIME
263 // epoch (1601-01-01 UTC). Convert to chrono::system_clock (1970
264 // epoch) by subtracting the constant offset, then break down with
265 // gmtime_r for display.
266 constexpr int64_t kFileTimeEpochToUnixEpochMs = 11644473600000LL;
267 const int64_t lastPlayedUnixMs =
268 levelSummary->getLastPlayed() - kFileTimeEpochToUnixEpochMs;
269 const auto tp = std::chrono::system_clock::time_point{
270 std::chrono::milliseconds{lastPlayedUnixMs}};
271 auto dp = std::chrono::floor<std::chrono::days>(tp);
272 std::chrono::year_month_day ymd{dp};
273 std::chrono::hh_mm_ss hms{
274 std::chrono::floor<std::chrono::minutes>(tp - dp)};
275
276 // 4J Stu - Currently shows years as 4 digits, where java only showed 2
277 // 4J - TODO Localise this
278 id += std::format(" ({}/{}/{} {}:{:02d}", (unsigned)ymd.day(),
279 (unsigned)ymd.month(), (int)ymd.year(),
280 (int)hms.hours().count(), (int)hms.minutes().count());
281
282 int64_t size = levelSummary->getSizeOnDisk();
283 id = id + ", " + toWString<float>(size / 1024 * 100 / 1024 / 100.0f) +
284 " MB)";
285 std::string info;
286
287 if (levelSummary->isRequiresConversion()) {
288 info = parent->conversionLang + " " + info;
289 }
290
291 parent->drawString(parent->font, name, x + 2, y + 1, 0xffffff);
292 parent->drawString(parent->font, id, x + 2, y + 12, 0x808080);
293 parent->drawString(parent->font, info, x + 2, y + 12 + 10, 0x808080);
294}
