64void BeaconScreen::tick() {
65 if (buttonsNotDrawn && beacon->getLevels() >= 0) {
66 buttonsNotDrawn = false;
67
68 int xo = (width - imageWidth) / 2;
69 int yo = (height - imageHeight) / 2;
70
71 for (int tier = 0; tier <= 2; ++tier) {
72 int effectCount = BeaconTileEntity::BEACON_EFFECTS_EFFECTS;
73 int actualCount = 0;
74 for (int e = 0; e < effectCount; ++e) {
75 if (BeaconTileEntity::BEACON_EFFECTS[tier][e] != nullptr) {
76 actualCount++;
77 } else {
78 break;
79 }
80 }
81
82 int totalWidth = actualCount * 22 + (actualCount - 1) * 2;
83 int startX = xo + 53 + (actualCount * 24 - totalWidth) / 2;
84
85 for (int e = 0; e < actualCount; ++e) {
86 MobEffect* effect = BeaconTileEntity::BEACON_EFFECTS[tier][e];
87 if (effect == nullptr) break;
88
89 int buttonId = (tier << 8) | effect->id;
90 BeaconPowerButton* button = new BeaconPowerButton(
91 this, buttonId, startX + e * 24, yo + 22 + tier * 25,
92 effect->id, tier);
93 buttons.push_back(button);
94
95 if (tier >= beacon->getLevels()) {
96 button->active = false;
97 } else if (effect->id == beacon->getPrimaryPower()) {
98 button->setSelected(true);
99 }
100 }
101 }
102
103 int tier = 3;
104 int effectCount = BeaconTileEntity::BEACON_EFFECTS_EFFECTS;
105 int actualCount = 0;
106 for (int e = 0; e < effectCount; ++e) {
107 if (BeaconTileEntity::BEACON_EFFECTS[tier][e] != nullptr) {
108 actualCount++;
109 } else {
110 break;
111 }
112 }
113
114 int totalWidth = (actualCount + 1) * 22 + actualCount * 2;
115 int startX = xo + 143 + ((actualCount + 1) * 24 - totalWidth) / 2;
116
117 for (int e = 0; e < actualCount; ++e) {
118 MobEffect* effect = BeaconTileEntity::BEACON_EFFECTS[tier][e];
119 if (effect == nullptr) break;
120
121 int buttonId = (tier << 8) | effect->id;
122 BeaconPowerButton* button = new BeaconPowerButton(
123 this, buttonId, startX + e * 24, yo + 47, effect->id, tier);
124 buttons.push_back(button);
125
126 if (tier >= beacon->getLevels()) {
127 button->active = false;
128 } else if (effect->id == beacon->getSecondaryPower()) {
129 button->setSelected(true);
130 }
131 }
132
133 if (beacon->getPrimaryPower() > 0) {
134 int buttonId = (tier << 8) | beacon->getPrimaryPower();
135 BeaconPowerButton* button =
136 new BeaconPowerButton(this, buttonId, startX + actualCount * 24,
137 yo + 47, beacon->getPrimaryPower(), tier);
138 buttons.push_back(button);
139
140 if (tier >= beacon->getLevels()) {
141 button->active = false;
142 } else if (beacon->getPrimaryPower() ==
143 beacon->getSecondaryPower()) {
144 button->setSelected(true);
145 }
146 }
147 }
148
149 beaconConfirmButton->active =
150 (beacon->getItem(0) != nullptr && beacon->getPrimaryPower() > 0);
151}

209void BeaconScreen::buttonClicked(Button* button) {
210 if (button->id == -2) {
211 minecraft->player->closeContainer();
212 } else if (button->id == -1) {
213 // 4jcraft: copied from IUIScene_BeaconMenu
214 ByteArrayOutputStream baos;
215 DataOutputStream dos(&baos);
216 dos.writeInt(beacon->getPrimaryPower());
217 dos.writeInt(beacon->getSecondaryPower());
218
219 minecraft->player->connection->send(
220 std::make_shared<CustomPayloadPacket>(
221 CustomPayloadPacket::SET_BEACON_PACKET, baos.toByteArray()));
222 minecraft->player->closeContainer();
223 } else if (dynamic_cast<BeaconPowerButton*>(button)) {
224 int effectId = button->id & 255;
225 int tier = button->id >> 8;
226
227 if (tier < 3) {
228 beacon->setPrimaryPower(effectId);
229 } else {
230 beacon->setSecondaryPower(effectId);
231 }
232
233 for (Button* btn : buttons) {
234 delete btn;
235 }
236 buttons.clear();
237 init();
238 tick();
239 }
