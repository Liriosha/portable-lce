37bool ReadPortableBinaryFile(File& file, std::uint8_t*& data,
38 unsigned int& size) {
39 const int64_t fileLength = file.length();
40 if (fileLength < 0 ||
41 fileLength >
42 static_cast<int64_t>(std::numeric_limits<unsigned int>::max())) {
43 data = nullptr;
44 size = 0;
45 return false;
46 }
47
48 const std::size_t capacity = static_cast<std::size_t>(fileLength);
49 std::uint8_t* buffer = new std::uint8_t[capacity == 0 ? 1 : capacity];
50 auto readResult =
51 PlatformFilesystem.readFile(file.getPath(), buffer, capacity);
52 if (readResult.status != IPlatformFilesystem::ReadStatus::Ok ||
53 readResult.fileSize > std::numeric_limits<unsigned int>::max()) {
54 delete[] buffer;
55 data = nullptr;
56 size = 0;
57 return false;
58 }
59
60 data = buffer;
61 size = static_cast<unsigned int>(readResult.fileSize);
62 return true;
63}

207

223 if (!image->loadMipmapPng(l, pbData, dataBytes)) {

295int DLCTexturePack::onPackMounted(int iPad, std::uint32_t dwErr,
296 std::uint32_t dwLicenceMask) {
297 DLCTexturePack* texturePack = this;
298 texturePack->m_bLoadingData = false;
299 if (dwErr != 0) {
300 // corrupt DLC
301 Log::info("Failed to mount DLC for pad %d: %u\n", iPad, dwErr);
302 } else {
303 Log::info("Mounted DLC for texture pack, attempting to load data\n");
304 texturePack->m_dlcDataPack =
305 new DLCPack(texturePack->m_dlcInfoPack->getName(), dwLicenceMask);
306 texturePack->setHasAudio(false);
307 unsigned int dwFilesProcessed = 0;
308 // Load the DLC textures
309 std::string dataFilePath =
310 texturePack->m_dlcInfoPack->getFullDataPath();
311 if (!dataFilePath.empty()) {
312 if (!gameServices().dlcReadDataFile(
313 dwFilesProcessed,
314 getFilePath(texturePack->m_dlcInfoPack->GetPackID(),
315 dataFilePath),
316 texturePack->m_dlcDataPack)) {
317 delete texturePack->m_dlcDataPack;
318 texturePack->m_dlcDataPack = nullptr;
319 }
320
321 // Load the UI data
322 if (texturePack->m_dlcDataPack != nullptr) {
323 File archivePath(
324 getFilePath(texturePack->m_dlcInfoPack->GetPackID(),
325 std::string("media.arc")));
326 if (archivePath.exists())
327 texturePack->m_archiveFile = new ArchiveFile(archivePath);
328
333 DLCPack* pack = texturePack->m_dlcInfoPack->GetParentPack();
334 LevelGenerationOptions* levelGen =
335 gameServices().getLevelGenerationOptions();
336 if (levelGen != nullptr && !levelGen->hasLoadedData()) {
337 int gameRulesCount = pack->getDLCItemsCount(
338 DLCManager::e_DLCType_GameRulesHeader);
339 for (int i = 0; i < gameRulesCount; ++i) {
340 DLCGameRulesHeader* dlcFile =
341 (DLCGameRulesHeader*)pack->getFile(
342 DLCManager::e_DLCType_GameRulesHeader, i);
343
344 if (!dlcFile->getGrfPath().empty()) {
345 File grf(getFilePath(
346 texturePack->m_dlcInfoPack->GetPackID(),
347 dlcFile->getGrfPath()));
348 if (grf.exists()) {
349 std::uint8_t* pbData = nullptr;
350 unsigned int fileSize = 0;
351 if (ReadPortableBinaryFile(grf, pbData,
352 fileSize)) {
353 // 4J-PB - is it possible that we can get
354 // here after a read fail and it's not an
355 // error?
356 dlcFile->setGrfData(
357 pbData, fileSize,
358 texturePack->m_stringTable);
359
360 delete[] pbData;
361
362 gameServices().setLevelGenerationOptions(
363 dlcFile->lgo);
364 } else {
365 gameServices().fatalLoadError();
366 }
367 }
368 }
369 }
370 if (levelGen->requiresBaseSave() &&
371 !levelGen->getBaseSavePath().empty()) {
372 File grf(
373 getFilePath(texturePack->m_dlcInfoPack->GetPackID(),
374 levelGen->getBaseSavePath()));
375 if (grf.exists()) {
376 std::uint8_t* pbData = nullptr;
377 unsigned int fileSize = 0;
378 if (ReadPortableBinaryFile(grf, pbData, fileSize)) {
379 // 4J-PB - is it possible that we can get here
380 // after a read fail and it's not an error?
381 levelGen->setBaseSaveData(pbData, fileSize);
382 } else {
383 gameServices().fatalLoadError();
384 }
385 }
386 }
387 }
388
389 // any audio data?
390 // DLCPack *pack = texturePack->m_dlcInfoPack->GetParentPack();
391 if (pack->getDLCItemsCount(DLCManager::e_DLCType_Audio) > 0) {
392 DLCAudioFile* dlcFile = (DLCAudioFile*)pack->getFile(
393 DLCManager::e_DLCType_Audio, 0);
394 texturePack->setHasAudio(true);
395 // init the streaming sound ids for this texture pack
396 int iOverworldStart, iNetherStart, iEndStart;
397 int iOverworldC, iNetherC, iEndC;
398
399 iOverworldStart = 0;
400 iOverworldC = dlcFile->GetCountofType(
401 DLCAudioFile::e_AudioType_Overworld);
402 iNetherStart = iOverworldC;
403 iNetherC = dlcFile->GetCountofType(
404 DLCAudioFile::e_AudioType_Nether);
405 iEndStart = iOverworldC + iNetherC;
406 iEndC =
407 dlcFile->GetCountofType(DLCAudioFile::e_AudioType_End);
408
409 static_cast<SoundEngine*>(
410 Minecraft::GetInstance()->soundEngine)
411 ->SetStreamingSounds(
412 iOverworldStart, iOverworldStart + iOverworldC,
413 iNetherStart, iNetherStart + iNetherC, iEndStart,
414 iEndStart + iEndC,
415 iEndStart + iEndC); // push the CD start to after
416 }
417 }
418 texturePack->loadColourTable();
419 }
420
421 // 4J-PB - we need to leave the texture pack mounted if it contained
422 // streaming audio
423 if (texturePack->hasAudio() == false) {
424 }
425 }
426
427 texturePack->m_bHasLoadedData = true;
428 if (gameServices().getLevelGenerationOptions())
429 gameServices().getLevelGenerationOptions()->setLoadedData();
430 gameServices().setAction(PlatformInput.GetPrimaryPad(),
431 eAppAction_ReloadTexturePack);
432
433 return 0;
434}

467 std::string path = "";
468 if (m_dlcDataPack != nullptr &&
469 m_dlcDataPack->doesPackContainFile(DLCManager::e_DLCType_UIData,
470 "TexturePack.xzp")) {
471 DLCUIDataFile* dataFile = (DLCUIDataFile*)m_dlcDataPack->getFile(
472 DLCManager::e_DLCType_UIData, "TexturePack.xzp");
473
474 std::uint32_t dwSize = 0;
475 std::uint8_t* pbData = dataFile->getData(dwSize);
476
477 constexpr int LOCATOR_SIZE =
478 256; // Use this to allocate space to hold a ResourceLocator string
479 char szResourceLocator[LOCATOR_SIZE];
480 snprintf(szResourceLocator, LOCATOR_SIZE,

483 path = szResourceLocator;
