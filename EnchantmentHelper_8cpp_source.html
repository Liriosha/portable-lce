89 if (item->id != Item::enchantedBook_Id) {

156int EnchantmentHelper::getDamageProtection(
157 std::vector<std::shared_ptr<ItemInstance>> armor, DamageSource* source) {
158 getDamageProtectionIteration.sum = 0;
159 getDamageProtectionIteration.source = source;
160
161 runIterationOnInventory(getDamageProtectionIteration, armor);
162
163 if (getDamageProtectionIteration.sum > 25) {
164 getDamageProtectionIteration.sum = 25;
165 }
166 // enchantment protection is on the scale of 0 to 25, where 20 or more
167 // will nullify nearly all damage (there will be damage spill)
168 return ((getDamageProtectionIteration.sum + 1) >> 1) +
169 random.nextInt((getDamageProtectionIteration.sum >> 1) + 1);
170}

275int EnchantmentHelper::getEnchantmentCost(
276 Random* random, int slot, int bookcases,
277 std::shared_ptr<ItemInstance> itemInstance) {
278 Item* item = itemInstance->getItem();
279 int itemValue = item->getEnchantmentValue();
280
281 if (itemValue <= 0) {
282 // not enchantable
283 return 0;
284 }
285
286 // 4J Stu - Updated function to 1.3 version for TU7
287 if (bookcases > 15) {
288 bookcases = 15;
289 }
290
291 int selected = random->nextInt(8) + 1 + (bookcases >> 1) +
292 random->nextInt(bookcases + 1);
293 if (slot == 0) {
294 return std::max((selected / 3), 1);
295 }
296 if (slot == 1) {
297 return std::max(selected, bookcases * 2);
298 }
299 return selected;

302std::shared_ptr<ItemInstance> EnchantmentHelper::enchantItem(
303 Random* random, std::shared_ptr<ItemInstance> itemInstance,
304 int enchantmentCost) {
305 std::vector<EnchantmentInstance*>* newEnchantment =
306 EnchantmentHelper::selectEnchantment(random, itemInstance,
307 enchantmentCost);
308 bool isBook = itemInstance->id == Item::book_Id;
309
310 if (isBook) itemInstance->id = Item::enchantedBook_Id;
311
312 if (newEnchantment != nullptr) {
313 for (auto it = newEnchantment->begin(); it != newEnchantment->end();
314 ++it) {
315 EnchantmentInstance* e = *it;
316 if (isBook) {
317 Item::enchantedBook->addEnchantment(itemInstance, e);
318 } else {
319 itemInstance->enchant(e->enchantment, e->level);
320 }
321 delete e;
322 }
323 delete newEnchantment;
324 }
325 return itemInstance;

335std::vector<EnchantmentInstance*>* EnchantmentHelper::selectEnchantment(
336 Random* random, std::shared_ptr<ItemInstance> itemInstance,
337 int enchantmentCost) {
338 // withdraw bonus from item
339 Item* item = itemInstance->getItem();
340 int itemBonus = item->getEnchantmentValue();
341
342 if (itemBonus <= 0) {
343 return nullptr;
344 }
345 // 4J Stu - Update function to 1.3 version for TU7
346 itemBonus /= 2;
347 itemBonus = 1 + random->nextInt((itemBonus >> 1) + 1) +
348 random->nextInt((itemBonus >> 1) + 1);
349
350 int enchantmentValue = itemBonus + enchantmentCost;
351
352 // the final enchantment cost will have another random span of +- 15%
353 float deviation = (random->nextFloat() + random->nextFloat() - 1.0f) * .15f;
354 int realValue = (int)((float)enchantmentValue * (1.0f + deviation) + .5f);
355 if (realValue < 1) {
356 realValue = 1;
357 }
358
359 std::vector<EnchantmentInstance*>* results = nullptr;
360
361 std::unordered_map<int, EnchantmentInstance*>* availableEnchantments =
362 getAvailableEnchantmentResults(realValue, itemInstance);
363 if (availableEnchantments != nullptr && !availableEnchantments->empty()) {
364 std::vector<WeighedRandomItem*> values;
365 for (auto it = availableEnchantments->begin();
366 it != availableEnchantments->end(); ++it) {
367 values.push_back(it->second);
368 }
369 EnchantmentInstance* instance =
370 (EnchantmentInstance*)WeighedRandom::getRandomItem(random, &values);
371 values.clear();
372
373 if (instance != nullptr) {
374 results = new std::vector<EnchantmentInstance*>();
375 results->push_back(
376 instance->copy()); // 4J Stu - Inserting a copy so we can clear
377 // memory from the availableEnchantments
378 // collection
379
380 int bonusChance = realValue;
381 while (random->nextInt(50) <= bonusChance) {
382 // remove incompatible enchantments from previous result
383 // final Iterator<Integer> mapIter =
384 // availableEnchantments.keySet().iterator(); while
385 // (mapIter.hasNext())
386 for (auto it = availableEnchantments->begin();
387 it != availableEnchantments->end();) {
388 int nextEnchantment = it->first; // mapIter.next();
389 bool valid = true;
390 // for (EnchantmentInstance *current : results)
391 for (auto resIt = results->begin(); resIt != results->end();
392 ++resIt) {
393 EnchantmentInstance* current = *resIt;
394 if (!current->enchantment->isCompatibleWith(
395 Enchantment::enchantments[nextEnchantment])) {
396 valid = false;
397 break;
398 }
399 }
400 if (!valid) {
401 // mapIter.remove();
402 delete it->second;
403 it = availableEnchantments->erase(it);
404 } else {
405 ++it;
406 }
407 }
408
409 if (!availableEnchantments->empty()) {
410 for (auto it = availableEnchantments->begin();
411 it != availableEnchantments->end(); ++it) {
412 values.push_back(it->second);
413 }
414 EnchantmentInstance* nextInstance =
415 (EnchantmentInstance*)WeighedRandom::getRandomItem(
416 random, &values);
417 values.clear();
418 results->push_back(
419 nextInstance
420 ->copy()); // 4J Stu - Inserting a copy so we can
421 // clear memory from the
422 // availableEnchantments collection
423 }
424
425 bonusChance >>= 1;
426 }
427 }
428 }
429 if (availableEnchantments != nullptr) {
430 for (auto it = availableEnchantments->begin();
431 it != availableEnchantments->end(); ++it) {
432 delete it->second;
433 }
434 delete availableEnchantments;
435 }
436
437 return results;
438}

440std::unordered_map<int, EnchantmentInstance*>*
441EnchantmentHelper::getAvailableEnchantmentResults(
442 int value, std::shared_ptr<ItemInstance> itemInstance) {
443 Item* item = itemInstance->getItem();
444 std::unordered_map<int, EnchantmentInstance*>* results = nullptr;
445
446 bool isBook = itemInstance->id == Item::book_Id;
447
448 // for (Enchantment e : Enchantment.enchantments)
449 for (unsigned int i = 0; i < Enchantment::enchantments.size(); ++i) {
450 Enchantment* e = Enchantment::enchantments[i];
451 if (e == nullptr) {
452 continue;
453 }
454
455 // Only picks "normal" enchantments, no specialcases
456 if (!e->category->canEnchant(item) && !isBook) {
457 continue;
458 }
459
460 for (int level = e->getMinLevel(); level <= e->getMaxLevel(); level++) {
461 if (value >= e->getMinCost(level) &&
462 value <= e->getMaxCost(level)) {
463 if (results == nullptr) {
464 results =
465 new std::unordered_map<int, EnchantmentInstance*>();
466 }
467 auto it = results->find(e->id);
468 if (it != results->end()) {
469 delete it->second;
470 }
471 (*results)[e->id] = new EnchantmentInstance(e, level);
472 }
473 }
474 }
475
476 return results;
