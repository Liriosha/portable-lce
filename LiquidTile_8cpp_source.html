49int LiquidTile::getColor(LevelSource* level, int x, int y, int z, int d) {
50 if (material == Material::water) {
51 int totalRed = 0;
52 int totalGreen = 0;
53 int totalBlue = 0;
54
55 for (int oz = -1; oz <= 1; oz++) {
56 for (int ox = -1; ox <= 1; ox++) {
57 int waterColor =
58 level->getBiome(x + ox, z + oz)->getWaterColor();
59
60 totalRed += (waterColor & 0xff0000) >> 16;
61 totalGreen += (waterColor & 0xff00) >> 8;
62 totalBlue += (waterColor & 0xff);
63 }
64 }
65
66 return (((totalRed / 9) & 0xFF) << 16) |
67 (((totalGreen / 9) & 0xFF) << 8) | (((totalBlue / 9) & 0xFF));
68 }
69 return 0xffffff;
70}

136Vec3 LiquidTile::getFlow(LevelSource* level, int x, int y, int z) {
137 Vec3 flow(0, 0, 0);
138 int mid = getRenderedDepth(level, x, y, z);
139 for (int d = 0; d < 4; d++) {
140 int xt = x;
141 int yt = y;
142 int zt = z;
143
144 if (d == 0) xt--;
145 if (d == 1) zt--;
146 if (d == 2) xt++;
147 if (d == 3) zt++;
148
149 int t = getRenderedDepth(level, xt, yt, zt);
150 if (t < 0) {
151 if (!level->getMaterial(xt, yt, zt)->blocksMotion()) {
152 t = getRenderedDepth(level, xt, yt - 1, zt);
153 if (t >= 0) {
154 int dir = t - (mid - 8);
155 flow = flow.add((xt - x) * dir, (yt - y) * dir,
156 (zt - z) * dir);
157 }
158 }
159 } else {
160 if (t >= 0) {
161 int dir = t - mid;
162 flow = flow.add((xt - x) * dir, (yt - y) * dir, (zt - z) * dir);
163 }
164 }
165 }
166 if (level->getData(x, y, z) >= 8) {
167 bool ok = false;
168 if (ok || isSolidFace(level, x, y, z - 1, 2)) ok = true;
169 if (ok || isSolidFace(level, x, y, z + 1, 3)) ok = true;
170 if (ok || isSolidFace(level, x - 1, y, z, 4)) ok = true;
171 if (ok || isSolidFace(level, x + 1, y, z, 5)) ok = true;
172 if (ok || isSolidFace(level, x, y + 1, z - 1, 2)) ok = true;
173 if (ok || isSolidFace(level, x, y + 1, z + 1, 3)) ok = true;
174 if (ok || isSolidFace(level, x - 1, y + 1, z, 4)) ok = true;
175 if (ok || isSolidFace(level, x + 1, y + 1, z, 5)) ok = true;
176 if (ok) flow = flow.normalize().add(0, -6, 0);
177 }
178 flow = flow.normalize();
179
180 return flow;
181}

340void LiquidTile::updateLiquid(Level* level, int x, int y, int z) {
341 if (level->getTile(x, y, z) != id) return;
342 if (material == Material::lava) {
343 bool water = false;
344 if (water || level->getMaterial(x, y, z - 1) == Material::water)
345 water = true;
346 if (water || level->getMaterial(x, y, z + 1) == Material::water)
347 water = true;
348 if (water || level->getMaterial(x - 1, y, z) == Material::water)
349 water = true;
350 if (water || level->getMaterial(x + 1, y, z) == Material::water)
351 water = true;
352 if (water || level->getMaterial(x, y + 1, z) == Material::water)
353 water = true;
354 if (water) {
355 int data = level->getData(x, y, z);
356 if (data == 0) {
357 level->setTileAndUpdate(x, y, z, Tile::obsidian_Id);
358 } else if (data <= 4) {
359 level->setTileAndUpdate(x, y, z, Tile::cobblestone_Id);
360 }
361 fizz(level, x, y, z);
362 }
