34// Creates a FileInputStream by opening a connection to an actual file, the file
35// named by the File object file in the file system. A new FileDescriptor object
36// is created to represent this file connection. First, if there is a security
37// manager, its checkRead method is called with the path represented by the file
38// argument as its argument.
39//
40// If the named file does not exist, is a directory rather than a regular file,
41// or for some other reason cannot be opened for reading then a
42// FileNotFoundException is thrown.
43//
44// Parameters:
45// file - the file to be opened for reading.
46// Throws:
47// FileNotFoundException - if the file does not exist, is a directory rather
48// than a regular file, or for some other reason cannot be opened for reading.
49// SecurityException - if a security manager exists and its checkRead method
50// denies read access to the file.
51FileInputStream::FileInputStream(const File& file) : m_fileHandle(nullptr) {
52 const std::string nativePath =
53 std::filesystem::path(file.getPath()).string();
54 m_fileHandle = std::fopen(nativePath.c_str(), "rb");
55
56 if (m_fileHandle == nullptr) {
57 assert(0);
58 }
59}

67// Reads a byte of data from this input stream. This method blocks if no input
68// is yet available. Returns: the next byte of data, or -1 if the end of the
69// file is reached.
70int FileInputStream::read() {
71 if (m_fileHandle == nullptr) {
72 return -1;
73 }
74
75 std::uint8_t byteRead = static_cast<std::uint8_t>(0);
76 const size_t numberOfBytesRead = std::fread(&byteRead, 1, 1, m_fileHandle);
77
78 if (std::ferror(m_fileHandle) != 0) {
79 assert(0);
80 } else if (numberOfBytesRead == 0) {
81 // File pointer is past the end of the file
82 return -1;
83 }
84
85 return static_cast<int>(byteRead);
86}

111// Reads up to len bytes of data from this input stream into an array of bytes.
112// If len is not zero, the method blocks until some input is available;
113// otherwise, no bytes are read and 0 is returned. Parameters: b - the buffer
114// into which the data is read. off - the start offset in the destination array
115// b len - the maximum number of bytes read. Returns: the total number of bytes
116// read into the buffer, or -1 if there is no more data because the end of the
117// file has been reached.
118int FileInputStream::read(std::vector<uint8_t>& b, unsigned int offset,

124 return -1;

131 assert(0);

160// Skips n bytes of input from this input stream. Fewer bytes might be skipped
161// if the end of the input stream is reached. The actual number k of bytes to be
162// skipped is equal to the smaller of n and count-pos. The value k is added into
163// pos and k is returned. Overrides: skip in class InputStream Parameters: n -
164// the number of bytes to be skipped. Returns: the actual number of bytes
165// skipped.
166int64_t FileInputStream::skip(int64_t n) {
167 if (m_fileHandle == nullptr || n <= 0) {
168 return 0;
169 }
170
171 const int64_t start = FileTell(m_fileHandle);
172 if (start < 0) {
173 return 0;
174 }
175
176 if (!FileSeek(m_fileHandle, 0, SEEK_END)) {
177 return 0;
178 }
179
180 const int64_t end = FileTell(m_fileHandle);
181 if (end < 0) {
182 return 0;
183 }
184
185 const int64_t offset = std::min(n, std::max<int64_t>(0, end - start));
186 const int64_t target = start + offset;
187 if (!FileSeek(m_fileHandle, target, SEEK_SET)) {
188 return 0;
189 }
190
191 return offset;
192}
