217void UIScene_AbstractContainerMenu::customDraw(
218 IggyCustomDrawCallbackRegion* region) {
219 Minecraft* pMinecraft = Minecraft::GetInstance();
220 if (pMinecraft->localplayers[m_iPad] == nullptr ||
221 pMinecraft->localgameModes[m_iPad] == nullptr)
222 return;
223
224 int slotId = parseSlotId(region->name);
225 std::shared_ptr<ItemInstance> item = nullptr;
226 if (std::char_traits<char16_t>::compare(region->name, u"pointerIcon", 11) ==
227 0) {
228 m_cacheSlotRenders = false;
229 item = pMinecraft->localplayers[m_iPad]->inventory->getCarried();
230 } else {
231 if (slotId == -1) {
232 app.DebugPrintf("This is not the control we are looking for\n");
233 } else {
234 m_cacheSlotRenders = true;
235 Slot* slot = m_menu->getSlot(slotId);
236 item = slot->getItem();
237 }
238 }
239
240 if (item != nullptr)
241 customDrawSlotControl(
242 region, m_iPad, item,
243 m_menu->isValidIngredient(item, slotId) ? 1.0f : 0.5f,
244 item->isFoil(), true);
245}

267void UIScene_AbstractContainerMenu::setSectionFocus(ESceneSection eSection,
268 int iPad) {
269 UIControl* newFocus = getSection(eSection);
270 if (newFocus) newFocus->setFocus(true);
271
272 if (m_focusSection != eSectionNone) {
273 UIControl* currentFocus = getSection(m_focusSection);
274 // 4J-TomK only set current focus to false if it differs from last
275 // (previously this continuously fired iggy functions when they were
276 // identical!
277 if (currentFocus != newFocus)
278 if (currentFocus) currentFocus->setFocus(false);
279 }
280
281 m_focusSection = eSection;
282}
