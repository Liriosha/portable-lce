77void BeaconTileEntity::applyEffects() {
78 if (isActive && levels > 0 && !level->isClientSide && primaryPower > 0) {
79 double range = (levels * 10) + 10;
80 int baseAmp = 0;
81 if (levels >= 4 && primaryPower == secondaryPower) {
82 baseAmp = 1;
83 }
84
85 AABB bb = AABB(x, y, z, x + 1, y + 1, z + 1).grow(range, range, range);
86 bb.y1 = level->getMaxBuildHeight();
87 std::vector<std::shared_ptr<Entity> >* players =
88 level->getEntitiesOfClass(typeid(Player), &bb);
89 for (auto it = players->begin(); it != players->end(); ++it) {
90 std::shared_ptr<Player> player =
91 std::dynamic_pointer_cast<Player>(*it);
92 player->addEffect(new MobEffectInstance(
93 primaryPower, SharedConstants::TICKS_PER_SECOND * 9, baseAmp,
94 true));
95 }
96
97 if (levels >= 4 && primaryPower != secondaryPower &&
98 secondaryPower > 0) {
99 for (auto it = players->begin(); it != players->end(); ++it) {
100 std::shared_ptr<Player> player =
101 std::dynamic_pointer_cast<Player>(*it);
102 player->addEffect(new MobEffectInstance(
103 secondaryPower, SharedConstants::TICKS_PER_SECOND * 9, 0,
104 true));
105 }
106 }
107 delete players;
108 }
109}

111void BeaconTileEntity::updateShape() {
112 if (!level->canSeeSky(x, y + 1, z)) {
113 isActive = false;
114 levels = 0;
115 } else {
116 isActive = true;
117
118 levels = 0;
119 for (int step = 1; step <= 4; step++) {
120 int ly = y - step;
121 if (ly < 0) {
122 break;
123 }
124
125 bool isOk = true;
126 for (int lx = x - step; lx <= x + step && isOk; lx++) {
127 for (int lz = z - step; lz <= z + step; lz++) {
128 int tile = level->getTile(lx, ly, lz);
129 if (tile != Tile::emeraldBlock_Id &&
130 tile != Tile::goldBlock_Id &&
131 tile != Tile::diamondBlock_Id &&
132 tile != Tile::ironBlock_Id) {
133 isOk = false;
134 break;
135 }
136 }
137 }
138 if (isOk) {
139 levels = step;
140 } else {
141 break;
142 }
143 }
144 if (levels == 0) {
145 isActive = false;
146 }
147 }
148}

150float BeaconTileEntity::getAndUpdateClientSideScale() {
151 if (!isActive) {
152 return 0;
153 }
154
155 int renderDelta = (int)(level->getGameTime() - clientSideRenderTick);
156 clientSideRenderTick = level->getGameTime();
157 if (renderDelta > 1) {
158 clientSideRenderScale -= ((float)renderDelta / (float)SCALE_TIME);
159
160 if (clientSideRenderScale < 0) {
161 clientSideRenderScale = 0;
162 }
163 }
164 clientSideRenderScale += (1.0f / (float)SCALE_TIME);
165 if (clientSideRenderScale > 1) {
166 clientSideRenderScale = 1;
167 }
168 return clientSideRenderScale;
169}
