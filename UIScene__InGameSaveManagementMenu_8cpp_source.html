4int UIScene_InGameSaveManagementMenu::loadSaveDataThumbnailReturned(
5 std::uint8_t* pbThumbnail, unsigned int dwThumbnailBytes) {
6 app.DebugPrintf("Received data for save thumbnail\n");
7
8 if (pbThumbnail && dwThumbnailBytes) {
9 m_saveDetails[m_iRequestingThumbnailId].pbThumbnailData =
10 new std::uint8_t[dwThumbnailBytes];
11 memcpy(m_saveDetails[m_iRequestingThumbnailId].pbThumbnailData,
12 pbThumbnail, dwThumbnailBytes);
13 m_saveDetails[m_iRequestingThumbnailId].dwThumbnailSize =
14 dwThumbnailBytes;
15 } else {
16 m_saveDetails[m_iRequestingThumbnailId].pbThumbnailData = nullptr;
17 m_saveDetails[m_iRequestingThumbnailId].dwThumbnailSize = 0;
18 app.DebugPrintf("Save thumbnail data is nullptr, or has size 0\n");
19 }
20 m_bSaveThumbnailReady = true;
21
22 return 0;
23}

138void UIScene_InGameSaveManagementMenu::tick() {
139 UIScene::tick();
140
141 if (m_bExitScene) // navigate forward or back
142 {
143 if (!m_bRetrievingSaveThumbnails) {
144 // need to wait for any callback retrieving thumbnail to complete
145 navigateBack();
146 }
147 }
148 // Stop loading thumbnails if we navigate forwards
149 if (hasFocus(m_iPad)) {
150 if (m_bUpdateSaveSize) {
151 m_spaceIndicatorSaves.selectSave(m_iSaveListIndex);
152 m_bUpdateSaveSize = false;
153 }
154
155 // Display the saves if we have them
156 if (!m_bSavesDisplayed) {
157 m_pSaveDetails = PlatformStorage.ReturnSavesInfo();
158 if (m_pSaveDetails != nullptr) {
159 m_spaceIndicatorSaves.reset();
160
161 m_bSavesDisplayed = true;
162
163 if (m_saveDetails != nullptr) {
164 for (unsigned int i = 0; i < m_pSaveDetails->iSaveC; ++i) {
165 if (m_saveDetails[i].pbThumbnailData != nullptr) {
166 delete m_saveDetails[i].pbThumbnailData;
167 }
168 }
169 delete m_saveDetails;
170 }
171 m_saveDetails = new SaveListDetails[m_pSaveDetails->iSaveC];
172
173 m_iSaveDetailsCount = m_pSaveDetails->iSaveC;
174 for (unsigned int i = 0; i < m_pSaveDetails->iSaveC; ++i) {
175 m_buttonListSaves.addItem(
176 m_pSaveDetails->SaveInfoA[i].UTF8SaveTitle, "");
177
178 m_saveDetails[i].saveId = i;
179 memcpy(m_saveDetails[i].UTF8SaveName,
180 m_pSaveDetails->SaveInfoA[i].UTF8SaveTitle, 128);
181 memcpy(m_saveDetails[i].UTF8SaveFilename,
182 m_pSaveDetails->SaveInfoA[i].UTF8SaveFilename,
183 MAX_SAVEFILENAME_LENGTH);
184 }
185 m_controlSavesTimer.setVisible(false);
186
187 // set focus on the first button
188 }
189 }
190
191 if (!m_bExitScene && m_bSavesDisplayed &&
192 !m_bRetrievingSaveThumbnails && !m_bAllLoaded) {
193 if (m_iRequestingThumbnailId < (m_buttonListSaves.getItemCount())) {
194 m_bRetrievingSaveThumbnails = true;
195 app.DebugPrintf("Requesting the first thumbnail\n");
196 // set the save to load
197 PSAVE_DETAILS pSaveDetails = PlatformStorage.ReturnSavesInfo();
198 IPlatformStorage::ESaveGameState eLoadStatus =
199 PlatformStorage.LoadSaveDataThumbnail(
200 &pSaveDetails->SaveInfoA[(int)m_iRequestingThumbnailId],
201 [this](std::uint8_t* data, unsigned int bytes) {
202 return loadSaveDataThumbnailReturned(data, bytes);
203 });
204
205 if (eLoadStatus !=
206 IPlatformStorage::ESaveGame_GetSaveThumbnail) {
207 // something went wrong
208 m_bRetrievingSaveThumbnails = false;
209 m_bAllLoaded = true;
210 }
211 }
212 } else if (m_bSavesDisplayed && m_bSaveThumbnailReady) {
213 m_bSaveThumbnailReady = false;
214
215 // check we're not waiting to exit the scene
216 if (!m_bExitScene) {
217 // convert to utf16
218 std::uint16_t u16Message[MAX_SAVEFILENAME_LENGTH];
219#if defined(_WINDOWS64)
220 int result = ::MultiByteToWideChar(
221 CP_UTF8, // convert from UTF-8
222 MB_ERR_INVALID_CHARS, // error on invalid chars
223 m_saveDetails[m_iRequestingThumbnailId]
224 .UTF8SaveFilename, // source UTF-8 string
225 MAX_SAVEFILENAME_LENGTH, // total length of source UTF-8
226 // string,
227 // in char's (= bytes), including end-of-string \0
228 (char*)u16Message, // destination buffer
229 MAX_SAVEFILENAME_LENGTH // size of destination buffer, in
230 // char's
231 );
232#else
233 uint32_t srcmax, dstmax;
234 uint32_t srclen, dstlen;
235 srcmax = MAX_SAVEFILENAME_LENGTH;
236 dstmax = MAX_SAVEFILENAME_LENGTH;
237
238 SceCesUcsContext context;
239 sceCesUcsContextInit(&context);
240
241 sceCesUtf8StrToUtf16Str(
242 &context,
243 (uint8_t*)m_saveDetails[m_iRequestingThumbnailId]
244 .UTF8SaveFilename,
245 srcmax, &srclen, u16Message, dstmax, &dstlen);
246#endif
247 if (m_saveDetails[m_iRequestingThumbnailId].pbThumbnailData) {
248 registerSubstitutionTexture(
249 (char*)u16Message,
250 m_saveDetails[m_iRequestingThumbnailId].pbThumbnailData,
251 m_saveDetails[m_iRequestingThumbnailId]
252 .dwThumbnailSize);
253 }
254 m_buttonListSaves.setTextureName(m_iRequestingThumbnailId,
255 (char*)u16Message);
256
257 ++m_iRequestingThumbnailId;
258 if (m_iRequestingThumbnailId <
259 (m_buttonListSaves.getItemCount())) {
260 app.DebugPrintf("Requesting another thumbnail\n");
261 // set the save to load
262 PSAVE_DETAILS pSaveDetails =
263 PlatformStorage.ReturnSavesInfo();
264 IPlatformStorage::ESaveGameState eLoadStatus =
265 PlatformStorage.LoadSaveDataThumbnail(
266 &pSaveDetails
267 ->SaveInfoA[(int)m_iRequestingThumbnailId],
268 [this](std::uint8_t* data, unsigned int bytes) {
269 return loadSaveDataThumbnailReturned(data,
270 bytes);
271 });
272 if (eLoadStatus !=
273 IPlatformStorage::ESaveGame_GetSaveThumbnail) {
274 // something went wrong
275 m_bRetrievingSaveThumbnails = false;
276 m_bAllLoaded = true;
277 }
278 } else {
279 m_bRetrievingSaveThumbnails = false;
280 m_bAllLoaded = true;
281 }
282 } else {
283 // stop retrieving thumbnails, and exit
284 m_bRetrievingSaveThumbnails = false;
285 }
286 }
287 }
288
289 switch (m_iState) {
290 case e_SavesIdle:
291 break;
292 case e_SavesRepopulateAfterDelete:
293 m_bIgnoreInput = false;
294 m_iRequestingThumbnailId = 0;
295 m_bAllLoaded = false;
296 m_bRetrievingSaveThumbnails = false;
297 m_bSavesDisplayed = false;
298 m_iSaveInfoC = 0;
299 m_buttonListSaves.clearList();
300 // PlatformStorage.ClearSavesInfo();
301 // GetSaveInfo();
302 m_iState = e_SavesIdle;
303 break;
304 }
305}

328void UIScene_InGameSaveManagementMenu::handleInput(int iPad, int key,
329 bool repeat, bool pressed,
330 bool released,
331 bool& handled) {
332 if (m_bIgnoreInput) return;
333
334 // if we're retrieving save info, ignore key presses

336

344 }
345 break;
346 case ACTION_MENU_OK:
347 case ACTION_MENU_UP:
348 case ACTION_MENU_DOWN:
349 case ACTION_MENU_PAGEUP:
350 case ACTION_MENU_PAGEDOWN:
351 sendInputToMovie(key, repeat, pressed, released);
352 handled = true;
353 break;
354 }

398int UIScene_InGameSaveManagementMenu::DeleteSaveDialogReturned(
399 void* pParam, int iPad, IPlatformStorage::EMessageResult result) {
400 UIScene_InGameSaveManagementMenu* pClass =
401 (UIScene_InGameSaveManagementMenu*)pParam;
402 // results switched for this dialog
403
404 if (result == IPlatformStorage::EMessage_ResultDecline) {
405 if (app.DebugSettingsOn() && app.GetLoadSavesFromFolderEnabled()) {
406 pClass->m_bIgnoreInput = false;
407 } else {
408 PlatformStorage.DeleteSaveData(
409 &pClass->m_pSaveDetails->SaveInfoA[pClass->m_iSaveListIndex],
410 [pClass](const bool bRes) {
411 return pClass->deleteSaveDataReturned(bRes);
412 });
413 pClass->m_controlSavesTimer.setVisible(true);
414 }
415 } else {
416 pClass->m_bIgnoreInput = false;
417 }
418
419 return 0;
420}
