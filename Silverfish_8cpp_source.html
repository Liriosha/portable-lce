103void Silverfish::serverAiStep() {
104 Monster::serverAiStep();
105
106 if (level->isClientSide) {
107 return;
108 }
109
110 if (lookForFriends > 0) {
111 lookForFriends--;
112 if (lookForFriends == 0) {
113 // see if there are any friendly monster eggs nearby
114 int baseX = Mth::floor(x);
115 int baseY = Mth::floor(y);
116 int baseZ = Mth::floor(z);
117 bool doBreak = false;
118
119 for (int yOff = 0; !doBreak && yOff <= 5 && yOff >= -5;
120 yOff = (yOff <= 0) ? 1 - yOff : 0 - yOff) {
121 for (int xOff = 0; !doBreak && xOff <= 10 && xOff >= -10;
122 xOff = (xOff <= 0) ? 1 - xOff : 0 - xOff) {
123 for (int zOff = 0; !doBreak && zOff <= 10 && zOff >= -10;
124 zOff = (zOff <= 0) ? 1 - zOff : 0 - zOff) {
125 int tile = level->getTile(baseX + xOff, baseY + yOff,
126 baseZ + zOff);
127 if (tile == Tile::monsterStoneEgg_Id) {
128 if (!level->getGameRules()->getBoolean(
129 GameRules::RULE_MOBGRIEFING)) {
130 int data = level->getData(
131 baseX + xOff, baseY + yOff, baseZ + zOff);
132
133 Tile* restoreTile = Tile::stone;
134 if (data == StoneMonsterTile::HOST_COBBLE) {
135 restoreTile = Tile::cobblestone;
136 }
137 if (data == StoneMonsterTile::HOST_STONEBRICK) {
138 restoreTile = Tile::stoneBrick;
139 }
140
141 level->setTileAndData(
142 baseX + xOff, baseY + yOff, baseZ + zOff,
143 restoreTile->id, 0, Tile::UPDATE_ALL);
144 } else {
145 level->destroyTile(baseX + xOff, baseY + yOff,
146 baseZ + zOff, false);
147 }
148 Tile::monsterStoneEgg->destroy(level, baseX + xOff,
149 baseY + yOff,
150 baseZ + zOff, 0);
151
152 if (random->nextBoolean()) {
153 doBreak = true;
154 break;
155 }
156 }
157 }
158 }
159 }
160 }
161 }
162
163 if (attackTarget == nullptr && !isPathFinding()) {
164 // if the silverfish isn't doing anything special, it will merge
165 // with any rock tile it is nearby
166 int tileX = Mth::floor(x), tileY = Mth::floor(y + .5f),
167 tileZ = Mth::floor(z);
168 int facing = random->nextInt(6);
169
170 int tile = level->getTile(tileX + Facing::STEP_X[facing],
171 tileY + Facing::STEP_Y[facing],
172 tileZ + Facing::STEP_Z[facing]);
173 if (StoneMonsterTile::isCompatibleHostBlock(tile)) {
174 level->setTileAndData(
175 tileX + Facing::STEP_X[facing], tileY + Facing::STEP_Y[facing],
176 tileZ + Facing::STEP_Z[facing], Tile::monsterStoneEgg_Id,
177 StoneMonsterTile::getDataForHostBlock(tile), Tile::UPDATE_ALL);
178 spawnAnim();
179 remove();
180 } else {
181 findRandomStrollLocation();
182 }
183
184 } else if (attackTarget != nullptr && !isPathFinding()) {
185 attackTarget = nullptr;
186 }
187}
