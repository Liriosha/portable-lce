124std::string AbstractTexturePack::getAnimationString(
125 const std::string& textureName, const std::string& path) {
126 std::string animationDefinitionFile = textureName + ".txt";
127
128 bool requiresFallback = !hasFile("\\" + textureName + ".png", false);
129
130 std::string result = "";
131
132 InputStream* fileStream =
133 getResource("\\" + path + animationDefinitionFile, requiresFallback);
134
135 if (fileStream) {
136 // Minecraft::getInstance()->getLogger().info("Found animation info for:
137 // " + animationDefinitionFile);
138#if !defined(_CONTENT_PACKAGE)
139 Log::info("Found animation info for: %s\n",
140 animationDefinitionFile.c_str());
141#endif
142 InputStreamReader isr(fileStream);
143 BufferedReader br(&isr);
144
145 std::string line = br.readLine();
146 while (!line.empty()) {
147 line = trimString(line);
148 if (line.length() > 0) {
149 result.append(",");
150 result.append(line);
151 }
152 line = br.readLine();
153 }
154 delete fileStream;
155 }
156
157 return result;
158}

179void AbstractTexturePack::loadDefaultColourTable() {
180 // Load the file
181 File coloursFile(
182 AbstractTexturePack::getPath(true).append("res/colours.col"));
183
184 if (coloursFile.exists()) {
185 uint32_t dataLength = coloursFile.length();
186 std::vector<uint8_t> data(dataLength);
187
188 FileInputStream fis(coloursFile);
189 fis.read(data, 0, dataLength);
190 fis.close();
191 if (m_colourTable != nullptr) delete m_colourTable;
192 m_colourTable = new ColourTable(data.data(), dataLength);
193
194 } else {
195 Log::info("Failed to load the default colours table\n");
196 gameServices().fatalLoadError();
197 }
198}
