80bool HalfSlabTile::shouldRenderFace(LevelSource* level, int x, int y, int z,
81 int face) {
82 if (fullSize) return Tile::shouldRenderFace(level, x, y, z, face);
83
84 if (face != Facing::UP && face != Facing::DOWN &&
85 !Tile::shouldRenderFace(level, x, y, z, face)) {
86 return false;
87 }
88
89 int ox = x, oy = y, oz = z;
90 ox += Facing::STEP_X[Facing::OPPOSITE_FACING[face]];
91 oy += Facing::STEP_Y[Facing::OPPOSITE_FACING[face]];
92 oz += Facing::STEP_Z[Facing::OPPOSITE_FACING[face]];
93
94 bool isUpper = (level->getData(ox, oy, oz) & TOP_SLOT_BIT) != 0;
95 if (isUpper) {
96 if (face == Facing::DOWN) return true;
97 if (face == Facing::UP && Tile::shouldRenderFace(level, x, y, z, face))
98 return true;
99 return !(isHalfSlab(level->getTile(x, y, z)) &&
100 (level->getData(x, y, z) & TOP_SLOT_BIT) != 0);
101 } else {
102 if (face == Facing::UP) return true;
103 if (face == Facing::DOWN &&
104 Tile::shouldRenderFace(level, x, y, z, face))
105 return true;
106 return !(isHalfSlab(level->getTile(x, y, z)) &&
107 (level->getData(x, y, z) & TOP_SLOT_BIT) == 0);
108 }
