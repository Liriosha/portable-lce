48Path* PathFinder::findPath(Entity* e, double xt, double yt, double zt,
49 float maxDist) {
50 openSet.clear();
51 nodes.clear();
52
53 bool resetAvoidWater = avoidWater;
54 int startY = Mth::floor(e->bb.y0 + 0.5f);
55 if (canFloat && e->isInWater()) {
56 startY = (int)(e->bb.y0);
57 int tileId = level->getTile((int)Mth::floor(e->x), startY,
58 (int)Mth::floor(e->z));
59 while (tileId == Tile::water_Id || tileId == Tile::calmWater_Id) {
60 ++startY;
61 tileId = level->getTile((int)Mth::floor(e->x), startY,
62 (int)Mth::floor(e->z));
63 }
64 resetAvoidWater = avoidWater;
65 avoidWater = false;
66 } else
67 startY = Mth::floor(e->bb.y0 + 0.5f);
68
69 Node* from = getNode((int)floor(e->bb.x0), startY, (int)floor(e->bb.z0));
70 Node* to = getNode((int)floor(xt - e->bbWidth / 2), (int)floor(yt),
71 (int)floor(zt - e->bbWidth / 2));
72
73 Node* size =
74 new Node((int)floor(e->bbWidth + 1), (int)floor(e->bbHeight + 1),
75 (int)floor(e->bbWidth + 1));
76 Path* path = findPath(e, from, to, size, maxDist);
77 delete size;
78
79 avoidWater = resetAvoidWater;
80 return path;
81}

83// function A*(start,goal)
84Path* PathFinder::findPath(Entity* e, Node* from, Node* to, Node* size,
85 float maxDist) {
86 from->g = 0;
87 from->h = from->distanceToSqr(to);
88 from->f = from->h;
89
90 openSet.clear();
91 openSet.insert(from);
92
93 Node* closest = from;
94
95 while (!openSet.isEmpty()) {
96 Node* x = openSet.pop();
97
98 if (x->equals(to)) {
99 return reconstruct_path(from, to);
100 }
101
102 if (x->distanceToSqr(to) < closest->distanceToSqr(to)) {
103 closest = x;
104 }
105 x->closed = true;
106
107 int neighborCount = getNeighbors(e, x, size, to, maxDist);
108 for (int i = 0; i < neighborCount; i++) {
109 Node* y = (*neighbors)[i];
110
111 float tentative_g_score = x->g + x->distanceToSqr(y);
112 if (!y->inOpenSet() || tentative_g_score < y->g) {
113 y->cameFrom = x;
114 y->g = tentative_g_score;
115 y->h = y->distanceToSqr(to);
116 if (y->inOpenSet()) {
117 openSet.changeCost(y, y->g + y->h);
118 } else {
119 y->f = y->g + y->h;
120 openSet.insert(y);
121 }
122 }
123 }
124 }
125
126 if (closest == from) return nullptr;
127 return reconstruct_path(from, closest);
128}

130int PathFinder::getNeighbors(Entity* entity, Node* pos, Node* size,
131 Node* target, float maxDist) {
132 int p = 0;
133
134 int jumpSize = 0;
135 if (isFree(entity, pos->x, pos->y + 1, pos->z, size) == TYPE_OPEN)
136 jumpSize = 1;
137
138 Node* n = getNode(entity, pos->x, pos->y, pos->z + 1, size, jumpSize);
139 Node* w = getNode(entity, pos->x - 1, pos->y, pos->z, size, jumpSize);
140 Node* e = getNode(entity, pos->x + 1, pos->y, pos->z, size, jumpSize);
141 Node* s = getNode(entity, pos->x, pos->y, pos->z - 1, size, jumpSize);
142
143 if (n != nullptr && !n->closed && n->distanceTo(target) < maxDist)
144 (*neighbors)[p++] = n;
145 if (w != nullptr && !w->closed && w->distanceTo(target) < maxDist)
146 (*neighbors)[p++] = w;
147 if (e != nullptr && !e->closed && e->distanceTo(target) < maxDist)
148 (*neighbors)[p++] = e;
149 if (s != nullptr && !s->closed && s->distanceTo(target) < maxDist)
150 (*neighbors)[p++] = s;
151
152 return p;
153}

155Node* PathFinder::getNode(Entity* entity, int x, int y, int z, Node* size,
156 int jumpSize) {
157 Node* best = nullptr;
158 int pathType = isFree(entity, x, y, z, size);
159 if (pathType == TYPE_WALKABLE) return getNode(x, y, z);
160 if (pathType == TYPE_OPEN) best = getNode(x, y, z);
161 if (best == nullptr && jumpSize > 0 && pathType != TYPE_FENCE &&
162 pathType != TYPE_TRAP &&
163 isFree(entity, x, y + jumpSize, z, size) == TYPE_OPEN) {
164 best = getNode(x, y + jumpSize, z);
165 y += jumpSize;
166 }
167
168 if (best != nullptr) {
169 int drop = 0;
170 int cost = 0;
171 while (y > 0) {
172 cost = isFree(entity, x, y - 1, z, size);
173 if (avoidWater && cost == TYPE_WATER) return nullptr;
174 if (cost != TYPE_OPEN) break;
175 // fell too far?
176 if (++drop >= 4)
177 return nullptr; // 4J - rolling this back to pre-java 1.6.4
178 // version as we're suspicious of the
179 // performance implications of this
180 // if (drop++ >=
181 // entity->getMaxFallDistance()) return
182 // nullptr;
183 y--;
184
185 if (y > 0) best = getNode(x, y, z);
186 }
187 // fell into lava?
188 if (cost == TYPE_LAVA) return nullptr;
189 }
190
191 return best;
192}

212int PathFinder::isFree(Entity* entity, int x, int y, int z, Node* size,
213 bool avoidWater, bool canOpenDoors, bool canPassDoors) {
214 bool walkable = false;
215 for (int xx = x; xx < x + size->x; xx++)
216 for (int yy = y; yy < y + size->y; yy++)
217 for (int zz = z; zz < z + size->z; zz++) {
218 int tileId = entity->level->getTile(xx, yy, zz);
219 if (tileId <= 0) continue;
220 if (tileId == Tile::trapdoor_Id)
221 walkable = true;
222 else if (tileId == Tile::water_Id ||
223 tileId == Tile::calmWater_Id) {
224 if (avoidWater)
225 return TYPE_WATER;
226 else
227 walkable = true;
228 } else if (!canPassDoors && tileId == Tile::door_wood_Id) {
229 return TYPE_BLOCKED;
230 }
231
232 Tile* tile = Tile::tiles[tileId];
233
234 // 4J Stu - Use new getTileRenderShape passing in the tileId we
235 // have already got
236 if (entity->level->getTileRenderShape(tileId) ==
237 Tile::SHAPE_RAIL) {
238 int xt = Mth::floor(entity->x);
239 int yt = Mth::floor(entity->y);
240 int zt = Mth::floor(entity->z);
241 if (entity->level->getTileRenderShape(xt, yt, zt) ==
242 Tile::SHAPE_RAIL ||
243 entity->level->getTileRenderShape(xt, yt - 1, zt) ==
244 Tile::SHAPE_RAIL) {
245 continue;
246 } else {
247 return TYPE_FENCE;
248 }
249 }
250
251 if (tile->isPathfindable(entity->level, xx, yy, zz)) continue;
252 if (canOpenDoors && tileId == Tile::door_wood_Id) continue;
253
254 int renderShape = tile->getRenderShape();
255 if (renderShape == Tile::SHAPE_FENCE ||
256 tileId == Tile::fenceGate_Id ||
257 renderShape == Tile::SHAPE_WALL)
258 return TYPE_FENCE;
259 if (tileId == Tile::trapdoor_Id) return TYPE_TRAP;
260 Material* m = tile->material;
261 if (m == Material::lava) {
262 if (entity->isInLava()) continue;
263 return TYPE_LAVA;
264 }
265 return TYPE_BLOCKED;
266 }
267
268 return walkable ? TYPE_WALKABLE : TYPE_OPEN;
269}

272Path* PathFinder::reconstruct_path(Node* from, Node* to) {
273 int count = 1;
274 Node* n = to;
275 while (n->cameFrom != nullptr) {
276 count++;
277 n = n->cameFrom;
278 }
279
280 std::vector<Node*> nodes = std::vector<Node*>(count);
281 n = to;
282 nodes[--count] = n;
283 while (n->cameFrom != nullptr) {
284 n = n->cameFrom;
285 nodes[--count] = n;
286 }
287 Path* ret = new Path(nodes);
288 return ret;
289}
