161bool VillagerGolem::doHurtTarget(std::shared_ptr<Entity> target) {
162 attackAnimationTick = 10;
163 level->broadcastEntityEvent(shared_from_this(),
164 EntityEvent::START_ATTACKING);
165 bool hurt =
166 target->hurt(DamageSource::mobAttack(
167 std::dynamic_pointer_cast<Mob>(shared_from_this())),
168 7 + random->nextInt(15));
169 if (hurt) target->yd += 0.4f;
170 playSound(eSoundType_MOB_IRONGOLEM_THROW, 1, 1);
171 return hurt;
172}

238bool VillagerGolem::hurt(DamageSource* source, float dmg) {
239 // 4J: Protect owned golem from untrusted players
240 if (isPlayerCreated()) {
241 std::shared_ptr<Entity> entity = source->getDirectEntity();
242 if (entity != nullptr && entity->instanceof(eTYPE_PLAYER)) {
243 std::shared_ptr<Player> player =
244 std::dynamic_pointer_cast<Player>(entity);
245 if (!player->isAllowedToAttackPlayers()) return false;
246 }
247 }
248
249 return Golem::hurt(source, dmg);
