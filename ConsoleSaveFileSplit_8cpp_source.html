63// Compress from data to dataCompressed. Uses a special compression method that
64// is designed just to efficiently store runs of zeros, with little overhead on
65// other stuff. Compresed format is a 4 byte uncompressed size, followed by data
66// as follows:
67//
68// Byte value
69// Meaning
70//
71// 1 - 255
72// Normal data 0 followed by 1 - 255
73// Run of 1 - 255 0s 0 followed by 0, followed by 256 to 65791 (as 2 bytes)
74// Run of 256 to 65791 zeros
75
76void ConsoleSaveFileSplit::RegionFileReference::Compress() {
77 unsigned char* dataIn = data;
78 unsigned char* dataInLast = data + fileEntry->data.length;
79
80 // std::int64_t startTime = System::currentTimeMillis();
81
82 // One pass through to work out storage space required for compressed data
83 unsigned int outputSize = 4; // 4 bytes required to store the uncompressed
84 // size for faster decompression
85 unsigned int runLength = 0;
86 while (dataIn != dataInLast) {
87 unsigned char thisByte = *dataIn++;
88 if ((thisByte != 0) || (runLength == (65535 + 256))) {
89 // We've got a non-zero value, or we've hit our maximum run length.
90 // If there was a preceeding run of zeros, encode that nwo
91 if (runLength != 0) {
92 if (runLength < 256) {
93 // Runs of 1 to 255 encoded as 0 followed by one byte of run
94 // length
95 outputSize += 2;
96 } else {
97 // Runs of 256 to 65791 encoded as two 0s followed by two
98 // bytes of run length - 256
99 outputSize += 4;
100 }
101 // Run is now processed
102 runLength = 0;
103 }
104 // Now handle the current byte
105 if (thisByte == 0) {
106 runLength++;
107 } else {
108 // Non-zero, just copy over to output
109 outputSize++;
110 }
111 } else {
112 // It's a zero - keep counting size of the run
113 runLength++;
114 }
115 }
116 // Handle any outstanding run
117 if (runLength != 0) {
118 if (runLength < 256) {
119 // Runs of 1 to 255 encoded as 0 followed by one byte of run length
120 outputSize += 2;
121 } else {
122 // Runs of 256 to 65791 encoded as two 0s followed by two bytes of
123 // run length - 256
124 outputSize += 4;
125 }
126 // Run is now processed
127 runLength = 0;
128 }
129
130 // Now actually allocate & write the compress data. First 4 bytes store the
131 // uncompressed size
132 dataCompressed = (unsigned char*)malloc(outputSize);
133 *((unsigned int*)dataCompressed) = fileEntry->data.length;
134 unsigned char* dataOut = dataCompressed + 4;
135 dataIn = data;
136
137 // Now same process as before, but actually writing
138 while (dataIn != dataInLast) {
139 unsigned char thisByte = *dataIn++;
140 if ((thisByte != 0) || (runLength == (65535 + 256))) {
141 // We've got a non-zero value, or we've hit our maximum run length.
142 // If there was a preceeding run of zeros, encode that nwo
143 if (runLength != 0) {
144 if (runLength < 256) {
145 // Runs of 1 to 255 encoded as 0 followed by one byte of run
146 // length
147 *dataOut++ = 0;
148 *dataOut++ = runLength;
149 } else {
150 // Runs of 256 to 65791 encoded as two 0s followed by two
151 // bytes of run length - 256
152 *dataOut++ = 0;
153 *dataOut++ = 0;
154 unsigned int largeRunLength = runLength - 256;
155 *dataOut++ = (largeRunLength >> 8) & 0xff;
156 *dataOut++ = (largeRunLength) & 0xff;
157 }
158 // Run is now processed
159 runLength = 0;
160 }
161 // Now handle the current byte
162 if (thisByte == 0) {
163 runLength++;
164 } else {
165 // Non-zero, just copy over to output
166 *dataOut++ = thisByte;
167 }
168 } else {
169 // It's a zero - keep counting size of the run
170 runLength++;
171 }
172 }
173 // Handle any outstanding run
174 if (runLength != 0) {
175 if (runLength < 256) {
176 // Runs of 1 to 255 encoded as 0 followed by one byte of run length
177 *dataOut++ = 0;
178 *dataOut++ = runLength;
179 } else {
180 // Runs of 256 to 65791 encoded as two 0s followed by two bytes of
181 // run length - 256
182 *dataOut++ = 0;
183 *dataOut++ = 0;
184 unsigned int largeRunLength = runLength - 256;
185 *dataOut++ = (largeRunLength >> 8) & 0xff;
186 *dataOut++ = (largeRunLength) & 0xff;
187 }
188 // Run is now processed
189 runLength = 0;
190 }
191 assert((dataOut - dataCompressed) == outputSize);
192 dataCompressedSize = outputSize;
193 // std::int64_t endTime = System::currentTimeMillis();
194 // Log::info("Compressing region file 0x%.8x from %d to %d bytes -
195 //%dms\n", fileEntry->data.regionIndex, fileEntry->data.length,
196 // dataCompressedSize, endTime - startTime);
197}

199// Decompress from dataCompressed -> data. See comment in Compress method for
200// format
201void ConsoleSaveFileSplit::RegionFileReference::Decompress() {
202 // std::int64_t startTime = System::currentTimeMillis();
203 fileEntry->data.length = *((unsigned int*)dataCompressed);
204
205 // If this is unusually large, then test how big it would be when expanded
206 // before trying to allocate. Matching the expanded size is (currently) our
207 // means of knowing that this file is ok
208 if (fileEntry->data.length > 1 * 1024 * 1024) {
209 unsigned int uncompressedSize = 0;
210 unsigned char* dataIn = dataCompressed + 4;
211 unsigned char* dataInLast = dataCompressed + dataCompressedSize;
212
213 while (dataIn != dataInLast) {
214 unsigned char thisByte = *dataIn++;
215 if (thisByte == 0) {
216 thisByte = *dataIn++;
217 if (thisByte == 0) {
218 unsigned int runLength = (*dataIn++) << 8;
219 runLength |= (*dataIn++);
220 runLength += 256;
221 uncompressedSize += runLength;
222 } else {
223 unsigned int runLength = thisByte;
224 uncompressedSize += runLength;
225 }
226 } else {
227 uncompressedSize++;
228 }
229 }
230
231 if (fileEntry->data.length != uncompressedSize) {
232 // Treat as if it was an empty region file
233 fileEntry->data.length = 0;
234 assert(0);
235 return;
236 }
237 }
238
239 data = (unsigned char*)malloc(fileEntry->data.length);
240 unsigned char* dataIn = dataCompressed + 4;
241 unsigned char* dataInLast = dataCompressed + dataCompressedSize;
242 unsigned char* dataOut = data;
243
244 while (dataIn != dataInLast) {
245 unsigned char thisByte = *dataIn++;
246 if (thisByte == 0) {
247 thisByte = *dataIn++;
248 if (thisByte == 0) {
249 unsigned int runLength = (*dataIn++) << 8;
250 runLength |= (*dataIn++);
251 runLength += 256;
252 for (unsigned int i = 0; i < runLength; i++) {
253 *dataOut++ = 0;
254 }
255 } else {
256 unsigned int runLength = thisByte;
257 for (unsigned int i = 0; i < runLength; i++) {
258 *dataOut++ = 0;
259 }
260 }
261 } else {
262 *dataOut++ = thisByte;
263 }
264 }
265 // If we failed to correctly decompress, then treat as if it was an empty
266 // region file
267 if ((dataOut - data) != fileEntry->data.length) {
268 free(data);
269 fileEntry->data.length = 0;
270 data = nullptr;
271 assert(0);
272 }
273 // std::int64_t endTime = System::currentTimeMillis();
274 // Log::info("Decompressing region file from 0x%.8x %d to %d bytes -
275 //%dms\n", fileEntry->data.regionIndex, dataCompressedSize,
276 // fileEntry->data.length, endTime - startTime);//
277}

417void ConsoleSaveFileSplit::_init(const std::string& fileName, void* pvSaveData,
418 unsigned int fileSize, ESavePlatform plat) {
419 m_lastTickTime = 0;
420
421 m_fileName = fileName;
422
423 // Get details of region files. From this point on we are responsible for
424 // the memory that the storage manager initially allocated for them
425 unsigned int regionCount = PlatformStorage.GetSubfileCount();
426 for (unsigned int i = 0; i < regionCount; i++) {
427 unsigned int regionIndex;
428 unsigned char* regionDataCompressed;
429 unsigned int regionSizeCompressed;
430
431 PlatformStorage.GetSubfileDetails(i, (int*)&regionIndex,
432 (void**)&regionDataCompressed,
433 &regionSizeCompressed);
434
435 RegionFileReference* regionFileRef = new RegionFileReference(
436 i, regionIndex, regionSizeCompressed, regionDataCompressed);
437 if (regionSizeCompressed > 0) {
438 regionFileRef->Decompress();
439 } else {
440 regionFileRef->fileEntry->data.length = 0;
441 }
442 regionFileRef->ReleaseCompressed();
443 regionFiles[regionIndex] = regionFileRef;
444 }
445
446 if (fileSize > 0) {
447 if (pvSaveData != nullptr) {
448 memcpy(pvSaveMem, pvSaveData, fileSize);
449 } else {
450 unsigned int storageLength;
451 PlatformStorage.GetSaveData(pvSaveMem, &storageLength);
452 Log::info("Filesize - %d, Adjusted size - %d\n", fileSize,
453 storageLength);
454 fileSize = storageLength;
455 }
456
457 int compressed = *(int*)pvSaveMem;
458 if (compressed == 0) {
459 unsigned int decompSize = *((int*)pvSaveMem + 1);
460
461 // An invalid save, so clear the memory and start from scratch
462 if (decompSize == 0) {
463 // 4J Stu - Saves created between 2/12/2011 and 7/12/2011 will
464 // have this problem
465 Log::info("Invalid save data format\n");
466 memset(pvSaveMem, 0, fileSize);
467 // Clear the first 8 bytes that reference the header
468 header.WriteHeader(pvSaveMem);
469 } else {
470 unsigned char* buf = new unsigned char[decompSize];
471
472 if (Compression::getCompression()->Decompress(
473 buf, &decompSize, (unsigned char*)pvSaveMem + 8,
474 fileSize - 8) == 0) {
475 assert(decompSize <= MAX_SAVE_SIZE);
476 memcpy(pvSaveMem, buf, decompSize);
477 } else {
478 // Corrupt save, although most of the terrain should
479 // actually be ok
480 Log::info("Failed to decompress save data!\n");
481#if !defined(_CONTENT_PACKAGE)
482 assert(0);
483#endif
484 memset(pvSaveMem, 0, fileSize);
485 // Clear the first 8 bytes that reference the header
486 header.WriteHeader(pvSaveMem);
487 }
488
489 delete[] buf;
490 }
491 }
492
493 header.ReadHeader(pvSaveMem, plat);
494
495 } else {
496 // Clear the first 8 bytes that reference the header
497 header.WriteHeader(pvSaveMem);
498 }
499}

515FileEntry* ConsoleSaveFileSplit::createFile(const ConsoleSavePath& fileName) {
516 LockSaveAccess();
517
518 // Determine if the file is a region file that should be split off into its

529 ReleaseSaveAccess();
530 return file;
531 }
532 }
533
534 FileEntry* file = header.AddFile(fileName.getName());
535 ReleaseSaveAccess();
536
537 return file;

540void ConsoleSaveFileSplit::deleteFile(FileEntry* file) {
541 if (file == nullptr) return;
542
543 assert(file->isRegionFile() == false);
544
545 LockSaveAccess();
546
547 unsigned int numberOfBytesRead = 0;
548 unsigned int numberOfBytesWritten = 0;
549
550 const int bufferSize = 4096;
551 int amountToRead = bufferSize;
552 std::uint8_t buffer[bufferSize];
553 unsigned int bufferDataSize = 0;
554
555 char* readStartOffset =
556 (char*)pvSaveMem + file->data.startOffset + file->getFileSize();
557
558 char* writeStartOffset = (char*)pvSaveMem + file->data.startOffset;
559
560 char* endOfDataOffset = (char*)pvSaveMem + header.GetStartOfNextData();
561
562 while (true) {
563 // Fill buffer from file
564 if (readStartOffset + bufferSize > endOfDataOffset) {
565 amountToRead = (int)(endOfDataOffset - readStartOffset);
566 } else {
567 amountToRead = bufferSize;
568 }
569
570 if (amountToRead == 0) break;
571
572 memcpy(buffer, readStartOffset, amountToRead);
573 numberOfBytesRead = amountToRead;
574
575 bufferDataSize = amountToRead;
576 readStartOffset += numberOfBytesRead;
577
578 // Write buffer to file
579 memcpy((void*)writeStartOffset, buffer, bufferDataSize);
580 numberOfBytesWritten = bufferDataSize;
581
582 writeStartOffset += numberOfBytesWritten;
583 }
584
585 header.RemoveFile(file);
586
587 finalizeWrite();
588
589 ReleaseSaveAccess();
590}

619 file->getFileSize();
620 if (bytesToGrowBy <= 0) return;
621

624 // if( bytesToGrowBy < 1024 )

629

632 file->data.length += bytesToGrowBy;
633

638bool ConsoleSaveFileSplit::writeFile(FileEntry* file, const void* lpBuffer,
639 unsigned int nNumberOfBytesToWrite,
640 unsigned int* lpNumberOfBytesWritten) {
641 assert(pvSaveMem != nullptr);
642 if (pvSaveMem == nullptr) {
643 return false;
644 }
645
646 LockSaveAccess();
647
648 if (file->isRegionFile()) {
649 unsigned int sizeRequired =
650 file->currentFilePointer + nNumberOfBytesToWrite;
651 RegionFileReference* fileRef = regionFiles[file->data.regionIndex];
652 if (sizeRequired > file->getFileSize()) {
653 fileRef->data =
654 (unsigned char*)realloc(fileRef->data, sizeRequired);
655 file->data.length = sizeRequired;
656 }
657
658 memcpy(fileRef->data + file->currentFilePointer, lpBuffer,
659 nNumberOfBytesToWrite);
660
661 // Log::info(">>>>>>>>>>>>>> writing a region file's
662 // data 0x%.8x, 0x%x offset %d of %d bytes (writing %d
663 // bytes)\n",file->data.regionIndex,fileRef->data,file->currentFilePointer,
664 // file->getFileSize(), nNumberOfBytesToWrite);
665
666 file->currentFilePointer += nNumberOfBytesToWrite;
667 file->updateLastModifiedTime();
668 fileRef->dirty = true;
669 } else {
670 PrepareForWrite(file, nNumberOfBytesToWrite);
671
672 char* writeStartOffset = (char*)pvSaveMem + file->currentFilePointer;
673 // printf("Write: pvSaveMem = %0xd, currentFilePointer = %d,
674 // writeStartOffset = %0xd\n", pvSaveMem, file->currentFilePointer,
675 // writeStartOffset);
676
677 memcpy((void*)writeStartOffset, lpBuffer, nNumberOfBytesToWrite);
678 *lpNumberOfBytesWritten = nNumberOfBytesToWrite;
679
680 if (file->data.length < 0) file->data.length = 0;
681
682 file->currentFilePointer += *lpNumberOfBytesWritten;
683
684 // printf("Wrote %d bytes to %s, new file pointer is %I64d\n",
685 // *lpNumberOfBytesWritten, file->data.filename,
686 // file->currentFilePointer);
687
688 file->updateLastModifiedTime();
689 }
690
691 ReleaseSaveAccess();
692
693 return true;
694}

696bool ConsoleSaveFileSplit::zeroFile(FileEntry* file,
697 unsigned int nNumberOfBytesToWrite,
698 unsigned int* lpNumberOfBytesWritten) {
699 assert(pvSaveMem != nullptr);
700 if (pvSaveMem == nullptr) {
701 return false;
702 }
703
704 // 4jcraft added: memset(nullptr + 0, 0, 0); was called
705 // no bytes need to be written, hence there you go
706 if (nNumberOfBytesToWrite == 0) {
707 if (lpNumberOfBytesWritten) {
708 *lpNumberOfBytesWritten = 0;
709 }
710 return 1;
711 }
712
713 LockSaveAccess();
714
715 if (file->isRegionFile()) {
716 unsigned int sizeRequired =
717 file->currentFilePointer + nNumberOfBytesToWrite;
718 RegionFileReference* fileRef = regionFiles[file->data.regionIndex];
719 if (sizeRequired > file->getFileSize()) {
720 fileRef->data =
721 (unsigned char*)realloc(fileRef->data, sizeRequired);
722 file->data.length = sizeRequired;
723 }
724
725 memset(fileRef->data + file->currentFilePointer, 0,
726 nNumberOfBytesToWrite);
727
728 // Log::info(">>>>>>>>>>>>>> writing a region file's
729 // data 0x%.8x, 0x%x offset %d of %d bytes (writing %d
730 // bytes)\n",file->data.regionIndex,fileRef->data,file->currentFilePointer,
731 // file->getFileSize(), nNumberOfBytesToWrite);
732
733 file->currentFilePointer += nNumberOfBytesToWrite;
734 file->updateLastModifiedTime();
735 fileRef->dirty = true;
736 } else {
737 PrepareForWrite(file, nNumberOfBytesToWrite);
738
739 char* writeStartOffset = (char*)pvSaveMem + file->currentFilePointer;
740 // printf("Write: pvSaveMem = %0xd, currentFilePointer = %d,
741 // writeStartOffset = %0xd\n", pvSaveMem, file->currentFilePointer,
742 // writeStartOffset);
743
744 memset((void*)writeStartOffset, 0, nNumberOfBytesToWrite);

823// In this method, attempt to write any dirty region files, subject to
824// maintaining a maximum write output rate. Writing is prioritised by time since
825// the region was last written.
826void ConsoleSaveFileSplit::tick() {
827 std::int64_t currentTime = System::currentTimeMillis();
828
829 // Don't do anything if the save system is up to something...
830 if (PlatformStorage.GetSaveState() != IPlatformStorage::ESaveGame_Idle) {
831 return;
832 }
833
834 // ...or we shouldn't be saving...
835 if (PlatformStorage.GetSaveDisabled()) {
836 return;
837 }
838
839 // ... or we haven't passed the required time since last assessing what to
840 // do
841 if ((currentTime - m_lastTickTime) < WRITE_TICK_RATE_MS) {
842 return;
843 }
844
845 LockSaveAccess();
846
847 m_lastTickTime = currentTime;
848
849 // Get total amount of data written over the time period we are interested
850 // in averaging over. Remove any older data.
851 unsigned int bytesWritten = 0;
852 for (auto it = writeHistory.begin(); it != writeHistory.end();) {
853 if ((currentTime - it->writeTime) >
854 (WRITE_BANDWIDTH_MEASUREMENT_PERIOD_SECONDS * 1000)) {
855 it = writeHistory.erase(it);
856 } else {
857 bytesWritten += it->writeSize;
858 it++;
859 }
860 }
861
862 // Compile a vector of dirty regions.
863 std::vector<DirtyRegionFile> dirtyRegions;
864 for (auto it = regionFiles.begin(); it != regionFiles.end(); it++) {
865 DirtyRegionFile dirtyRegion;
866
867 if (it->second->dirty) {
868 dirtyRegion.fileRef = it->second->fileEntry->getRegionFileIndex();
869 dirtyRegion.lastWritten = it->second->lastWritten;
870 dirtyRegions.push_back(dirtyRegion);
871 }
872 }
873
874 // Sort into ascending order, by lastWritten time. First elements will
875 // therefore be the ones least recently saved
876 std::sort(dirtyRegions.begin(), dirtyRegions.end());
877
878 bool writeRequired = false;
879 unsigned int bytesInTimePeriod = bytesWritten;
880 unsigned int bytesAddedThisTick = 0;
881 for (int i = 0; i < dirtyRegions.size(); i++) {
882 RegionFileReference* regionRef = regionFiles[dirtyRegions[i].fileRef];
883 unsigned int compressedSize = regionRef->GetCompressedSize();
884 bytesInTimePeriod += compressedSize;
885 bytesAddedThisTick += compressedSize;
886
887 // Always consider at least one item for writing, even if it breaks the
888 // rule on the maximum number of bytes we would like to send per tick
889 if ((i > 0) && (bytesAddedThisTick > WRITE_MAX_WRITE_PER_TICK)) {
890 break;
891 }
892
893 // Could we add this without breaking our bytes per second cap?
894 if ((bytesInTimePeriod / WRITE_BANDWIDTH_MEASUREMENT_PERIOD_SECONDS) >
895 WRITE_BANDWIDTH_BYTESPERSECOND) {
896 break;
897 }
898
899 // Can add for writing
900 WriteHistory writeEvent;
901 writeEvent.writeSize = compressedSize;
902 writeEvent.writeTime = System::currentTimeMillis();
903 writeHistory.push_back(writeEvent);
904
905 regionRef->Compress();
906 // Log::info("Tick: Writing region 0x%.8x, compressed
907 // as %d bytes\n",regionRef->fileEntry->getRegionFileIndex(),
908 // regionRef->dataCompressedSize);
909 PlatformStorage.UpdateSubfile(regionRef->index,
910 regionRef->dataCompressed,
911 regionRef->dataCompressedSize);
912 regionRef->dirty = false;
913 regionRef->lastWritten = System::currentTimeMillis();
914
915 writeRequired = true;
916 }
917#if !defined(_CONTENT_PACKAGE)
918 {
919 unsigned int totalDirty = 0;
920 unsigned int totalDirtyBytes = 0;
921 int64_t oldestDirty = currentTime;
922 for (auto it = regionFiles.begin(); it != regionFiles.end(); it++) {
923 if (it->second->dirty) {
924 if (it->second->lastWritten < oldestDirty) {
925 oldestDirty = it->second->lastWritten;
926 }
927 totalDirty++;
928 totalDirtyBytes += it->second->fileEntry->getFileSize();
929 }
930 }
931 }
932#endif
933
934 if (writeRequired) {
935 PlatformStorage.SaveSubfiles(
936 [this](bool bRes) { return SaveRegionFilesCallback(this, bRes); });
937 }
938
939 ReleaseSaveAccess();
940}

948void ConsoleSaveFileSplit::MoveDataBeyond(FileEntry* file,
949 unsigned int nNumberOfBytesToWrite) {
950 unsigned int numberOfBytesRead = 0;
951 unsigned int numberOfBytesWritten = 0;
952
953 const unsigned int bufferSize = 4096;
954 unsigned int amountToRead = bufferSize;
955 // assert( nNumberOfBytesToWrite <= bufferSize );
956 static std::uint8_t buffer1[bufferSize];
957 static std::uint8_t buffer2[bufferSize];
958 unsigned int buffer1Size = 0;
959 unsigned int buffer2Size = 0;
960
961 assert(header.GetFileSize() + nNumberOfBytesToWrite <= MAX_SAVE_SIZE);
962
963 // This is the start of where we want the space to be, and the start of the
964 // data that we need to move
965 char* spaceStartOffset =
966 (char*)pvSaveMem + file->data.startOffset + file->getFileSize();
967
968 // This is the end of where we want the space to be
969 char* spaceEndOffset = spaceStartOffset + nNumberOfBytesToWrite;
970
971 // This is the current end of the data that we want to move
972 char* beginEndOfDataOffset = (char*)pvSaveMem + header.GetStartOfNextData();
973
974 // This is where the end of the data is going to be
975 char* finishEndOfDataOffset = beginEndOfDataOffset + nNumberOfBytesToWrite;
976
977 // This is where we are going to read from (with the amount we want to read
978 // subtracted before we read)
979 char* readStartOffset = beginEndOfDataOffset;
980
981 // This is where we can safely write to (with the amount we want write
982 // subtracted before we write)
983 char* writeStartOffset = finishEndOfDataOffset;
984
985 // printf("\n******* MOVEDATABEYOND *******\n");
986 // printf("Space start: %d, space end: %d\n", spaceStartOffset - (char
987 // *)pvSaveMem, spaceEndOffset - (char *)pvSaveMem); printf("Current end of
988 // data: %d, new end of data: %d\n", beginEndOfDataOffset - (char
989 // *)pvSaveMem, finishEndOfDataOffset - (char *)pvSaveMem);
990
991 // Optimisation for things that are being moved in whole region file sector
992 // (4K chunks). We could generalise this a bit more but seems safest at the
993 // moment to identify this particular type of move and code explicitly for
994 // this situation
995 if ((nNumberOfBytesToWrite & 4095) == 0) {
996 if (nNumberOfBytesToWrite > 0) {
997 // Get addresses for start & end of the region we are copying from
998 // as uintptr_t, for easier maths
999 uintptr_t uiFromStart = (uintptr_t)spaceStartOffset;
1000 uintptr_t uiFromEnd = (uintptr_t)beginEndOfDataOffset;
1001
1002 // Round both of these values to get 4096 byte chunks that we will
1003 // need to at least partially move
1004 uintptr_t uiFromStartChunk = uiFromStart & ~((uintptr_t)4095);
1005 uintptr_t uiFromEndChunk = (uiFromEnd - 1) & ~((uintptr_t)4095);
1006
1007 // Loop through all the affected source 4096 chunks, going backwards
1008 // so we don't overwrite anything we'll need in the future
1009 for (uintptr_t uiCurrentChunk = uiFromEndChunk;
1010 uiCurrentChunk >= uiFromStartChunk; uiCurrentChunk -= 4096) {
1011 // Establish chunk we'll need to copy
1012 uintptr_t uiCopyStart = uiCurrentChunk;
1013 uintptr_t uiCopyEnd = uiCurrentChunk + 4096;
1014 // Clamp chunk to the bounds of the full region we are trying to
1015 // copy
1016 if (uiCopyStart < uiFromStart) {
1017 // Needs to be clampged against the start of our region
1018 uiCopyStart = uiFromStart;
1019 }
1020 if (uiCopyEnd > uiFromEnd) {
1021 // Needs to be clamped to the end of our region
1022 uiCopyEnd = uiFromEnd;
1023 }
1024 memcpy((void*)(uiCopyStart + nNumberOfBytesToWrite),
1025 (void*)uiCopyStart, uiCopyEnd - uiCopyStart);
1026 }
1027 }
1028 } else {
1029 while (true) {
1030 // Copy buffer 1 to buffer 2
1031 memcpy(buffer2, buffer1, buffer1Size);
1032 buffer2Size = buffer1Size;
1033
1034 // Fill buffer 1 from file
1035 if ((readStartOffset - bufferSize) < spaceStartOffset) {
1036 amountToRead = static_cast<unsigned int>(readStartOffset -
1037 spaceStartOffset);
1038 } else {
1039 amountToRead = bufferSize;
1040 }
1041
1042 // Push the read point back by the amount of bytes that we are going
1043 // to read
1044 readStartOffset -= amountToRead;
1045
1046 // printf("About to read %u from %d\n", amountToRead,
1047 // readStartOffset - (char *)pvSaveMem );
1048
1049 memcpy(buffer1, readStartOffset, amountToRead);
1050 numberOfBytesRead = amountToRead;
1051
1052 buffer1Size = amountToRead;
1053
1054 // Move back the write pointer by the amount of bytes we are going
1055 // to write
1056 writeStartOffset -= buffer2Size;
1057
1058 // Write buffer 2 to file
1059 if ((writeStartOffset + buffer2Size) <= finishEndOfDataOffset) {
1060 // printf("About to write %u to %d\n", buffer2Size,
1061 // writeStartOffset - (char *)pvSaveMem );
1062 memcpy((void*)writeStartOffset, buffer2, buffer2Size);
1063 numberOfBytesWritten = buffer2Size;
1064 } else {
1065 assert((writeStartOffset + buffer2Size) <=
1066 finishEndOfDataOffset);
1067 numberOfBytesWritten = 0;
1068 }
1069
1070 if (numberOfBytesRead == 0) {
1071 // printf("\n************** MOVE COMPLETED ***************
1072 // \n\n");
1073 assert(writeStartOffset == spaceEndOffset);
1074 break;
1075 }
1076 }
1077 }
1078
1079 header.AdjustStartOffsets(file, nNumberOfBytesToWrite);
1080}

1082// Attempt to convert a filename into a numeric identifier, which we use for
1083// region files. File names supported are of the form:
1084//
1085// Filename Encoded as
1086//
1087// r.x.z.mcr 00 00 xx zz
1088// DIM-1r.x.z.mcr 00 01 xx zz
1089// DIM1/r.x.z.mcr 00 02 xx zz
1090
1091bool ConsoleSaveFileSplit::GetNumericIdentifierFromName(
1092 const std::string& fileName, unsigned int* idOut) {
1093 // Determine whether it is one of our region file names if the file
1094 // extension is ".mbr"
1095 if (fileName.length() < 4) return false;
1096 std::string extension = fileName.substr(fileName.length() - 4, 4);
1097 if (extension != std::string(".mcr")) return false;
1098
1099 unsigned int id = 0;
1100 int x, z;
1101
1102 const char* cstr = fileName.c_str();
1103 const char* body = cstr + 2;
1104
1105 // If this filename starts with a "r" then assume it is of the format
1106 // "r.x.z.mcr" - don't do anything as default value we've set are correct
1107 if (cstr[0] != 'r') {
1108 // Must be prefixed by "DIM-1r." or "DIM1/r."
1109 body = cstr + 7;
1110 // Differentiate between these 2 options
1111 if (cstr[3] == '-') {
1112 // "DIM-1r."
1113 id = 0x00010000;
1114 } else {
1115 // "DIM/1r."
1116 id = 0x00020000;
1117 }
1118 }
1119 // Get x/z coords
1120 sscanf(body, "%d.%d.mcr", &x, &z);
1121
1122 // Pack full id
1123 // 4jcraft added cast to unsigned
1124 id |= (((unsigned int)x << 8) & 0x0000ff00);
1125 id |= (z & 0x000000ff);
1126
1127 *idOut = id;
1128
1129 return true;
1130}

1132// Convert a numeric file identifier (for region files) back into a normal
1133// filename. See comment above.
1134
1135std::string ConsoleSaveFileSplit::GetNameFromNumericIdentifier(
1136 unsigned int idIn) {
1137 std::string prefix;
1138
1139 switch (idIn & 0x00ff0000) {
1140 case 0:
1141 prefix = "";
1142 break;
1143 case 1:
1144 prefix = "DIM-1";
1145 break;
1146 case 2:
1147 prefix = "DIM1/";
1148 break;
1149 }
1150 signed char regionX = (idIn >> 8) & 255;
1151 signed char regionZ = idIn & 255;
1152 std::string region = (prefix + std::string("r.") + toWString(regionX) +
1153 "." + toWString(regionZ) + ".mcr");
1154
1155 return region;

1192void ConsoleSaveFileSplit::Flush(bool autosave, bool updateThumbnail) {
1193 LockSaveAccess();
1194
1195 // The storage manage might potentially be busy doing a sub-file write
1196 // initiated from the tick. Wait until this is totally processed.
1197 while (PlatformStorage.GetSaveState() != IPlatformStorage::ESaveGame_Idle) {
1198 Log::info("Flush wait\n");
1199 std::this_thread::sleep_for(std::chrono::milliseconds(10));
1200 }
1201
1202 finalizeWrite();
1203
1204 m_autosave = autosave;
1205 if (!m_autosave) processSubfilesForWrite();
1206
1207 time_util::Timer timer;
1208
1209 unsigned int fileSize = header.GetFileSize();
1210
1211 // Assume that the compression will make it smaller so initially attempt to
1212 // allocate the current file size We add 4 bytes to the start so that we can
1213 // signal compressed data And another 4 bytes to store the decompressed data
1214 // size
1215 unsigned int compLength = fileSize + 8;
1216
1217 // 4J Stu - Added TU-1 interim
1218
1219 // Attempt to allocate the required memory
1220 // We do not own this, it belongs to the PlatformStorage
1221 std::uint8_t* compData =
1222 (std::uint8_t*)PlatformStorage.AllocateSaveData(compLength);
1223
1224 // If we failed to allocate then compData will be nullptr
1225 // Pre-calculate the compressed data size so that we can attempt to allocate
1226 // a smaller buffer
1227 if (compData == nullptr) {
1228 // Length should be 0 here so that the compression call knows that we
1229 // want to know the length back
1230 compLength = 0;
1231
1232 // Pre-calculate the buffer size required for the compressed data
1233 timer.reset();
1234 Compression::getCompression()->Compress(nullptr, &compLength, pvSaveMem,
1235 fileSize);
1236
1237 Log::info("Check buffer size: Elapsed time %f\n",
1238 static_cast<float>(timer.elapsed_seconds()));
1239
1240 // We add 4 bytes to the start so that we can signal compressed data
1241 // And another 4 bytes to store the decompressed data size
1242 compLength = compLength + 8;
1243
1244 // Attempt to allocate the required memory
1245 compData = (std::uint8_t*)PlatformStorage.AllocateSaveData(compLength);
1246 }
1247
1248 if (compData != nullptr) {
1249 // Re-compress all save data before we save it to disk
1250 timer.reset();
1251 Compression::getCompression()->Compress(compData + 8, &compLength,
1252 pvSaveMem, fileSize);
1253
1254 Log::info("Compress: Elapsed time %f\n",
1255 static_cast<float>(timer.elapsed_seconds()));
1256
1257 memset(compData, 0, 8);
1258 int saveVer = 0;
1259 memcpy(compData, &saveVer, sizeof(int));
1260 memcpy(compData + 4, &fileSize, sizeof(int));
1261
1262 Log::info("Save data compressed from %d to %d\n", fileSize, compLength);
1263
1264 if (updateThumbnail) {
1265 std::uint8_t* pbThumbnailData = nullptr;
1266 unsigned int dwThumbnailDataSize = 0;
1267
1268 std::uint8_t* pbDataSaveImage = nullptr;
1269 unsigned int dwDataSizeSaveImage = 0;
1270
1271 std::uint8_t bTextMetadata[88];
1272 memset(bTextMetadata, 0, 88);
1273
1274 int64_t seed = 0;
1275 bool hasSeed = false;
1276 if (MinecraftServer::getInstance() != nullptr &&
1277 MinecraftServer::getInstance()->levels[0] != nullptr) {
1278 seed = MinecraftServer::getInstance()
1279 ->levels[0]
1280 ->getLevelData()
1281 ->getSeed();
1282 hasSeed = true;
1283 }
1284
1285 int iTextMetadataBytes = gameServices().createImageTextData(
1286 bTextMetadata, seed, hasSeed,
1287 gameServices().getGameHostOption(eGameHostOption_All),
1288 Minecraft::GetInstance()->getCurrentTexturePackId());
1289
1290 // set the icon and save image
1291 PlatformStorage.SetSaveImages(pbThumbnailData, dwThumbnailDataSize,
1292 pbDataSaveImage, dwDataSizeSaveImage,
1293 bTextMetadata, iTextMetadataBytes);
1294 Log::info("Save thumbnail size %d\n", dwThumbnailDataSize);
1295 }
1296
1297 int32_t saveOrCheckpointId = 0;
1298 bool validSave =
1299 PlatformStorage.GetSaveUniqueNumber(&saveOrCheckpointId);
1300
1301 // save the data
1302 PlatformStorage.SaveSaveData(
1303 [this](bool bRes) { return SaveSaveDataCallback(this, bRes); });
1304#if !defined(_CONTENT_PACKAGE)
1305 if (gameServices().debugSettingsOn()) {
1306 if (gameServices().getWriteSavesToFolderEnabled()) {
1307 DebugFlushToFile(compData, compLength + 8);
1308 }
1309 }
1310#endif
1311 ReleaseSaveAccess();
1312 }
1313}
