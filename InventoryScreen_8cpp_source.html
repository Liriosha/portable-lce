44void InventoryScreen::renderBg(float a) {

50 int yo = (height - imageHeight) / 2;

53 glEnable(GL_RESCALE_NORMAL);

56 glPushMatrix();

62 float oybr = minecraft->player->yBodyRot;
63 float oyr = minecraft->player->yRot;

65 float oyh = minecraft->player->yHeadRot;
66 float oyhp = minecraft->player->yHeadRotO;
67
68 float xd = (xo + 51) - xMouse;
69 float yd = (yo + 75 - 50) - yMouse;
70
71 glRotatef(45 + 90, 0, 1, 0);
72 Lighting::turnOn();
73 glRotatef(-45 - 90, 0, 1, 0);
74
75 glRotatef(-(float)atan(yd / 40.0f) * 20, 1, 0, 0);
76
77 minecraft->player->yBodyRot = (float)atan(xd / 40.0f) * 20;
78 minecraft->player->yRot = (float)atan(xd / 40.0f) * 40;
79 minecraft->player->xRot = -(float)atan(yd / 40.0f) * 20;
80 minecraft->player->yHeadRot = (float)atan(xd / 40.0f) * 40;
81 minecraft->player->yHeadRotO = (float)atan(xd / 40.0f) * 40;
82 glTranslatef(0, minecraft->player->heightOffset, 0);
83 EntityRenderDispatcher::instance->playerRotY = 180;
84 EntityRenderDispatcher::instance->render(minecraft->player, 0, 0, 0, 0, 1);
85 minecraft->player->yBodyRot = oybr;
86 minecraft->player->yRot = oyr;
87 minecraft->player->xRot = oxr;
88 minecraft->player->yHeadRot = oyh;
