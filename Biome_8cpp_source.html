315Biome::Biome(int id) : id(id) {
316 // 4J Stu Default inits
317 color = 0;
318 // snowCovered = false; // 4J - this isn't set by the java game any more
319 // so removing to save confusion
320
321 topMaterial = (uint8_t)Tile::grass_Id;
322 material = (uint8_t)Tile::dirt_Id;
323 leafColor = 0x4EE031;
324 _hasRain = true;
325 depth = 0.1f;
326 scale = 0.3f;
327 temperature = 0.5f;
328 downfall = 0.5f;
329 // waterColor = 0xffffff; // 4J Stu - Not used
330 decorator = nullptr;
331
332 m_grassColor = eMinecraftColour_NOT_SET;
333 m_foliageColor = eMinecraftColour_NOT_SET;
334 m_waterColor = eMinecraftColour_NOT_SET;
335
336 /* 4J - removing these so that we can consistently return newly created
337 trees via getTreeFeature, and let the calling function be resposible for
338 deleting the returned tree normalTree = new TreeFeature(); fancyTree = new
339 BasicTree(); birchTree = new BirchFeature(); swampTree = new
340 SwampTreeFeature();
341 */
342
343 biomes[id] = this;
344 decorator = createDecorator();
345
346 friendlies.push_back(new MobSpawnerData(eTYPE_SHEEP, 12, 4, 4));

349 eTYPE_CHICKEN, 10, 4, 4)); // 4J - moved chickens to their own category

351
352 enemies.push_back(new MobSpawnerData(eTYPE_SPIDER, 10, 4, 4));
353 enemies.push_back(new MobSpawnerData(eTYPE_ZOMBIE, 10, 4, 4));
354 enemies.push_back(new MobSpawnerData(eTYPE_SKELETON, 10, 4, 4));
355 enemies.push_back(new MobSpawnerData(eTYPE_CREEPER, 10, 4, 4));
356 enemies.push_back(new MobSpawnerData(eTYPE_SLIME, 10, 4, 4));
357 enemies.push_back(new MobSpawnerData(eTYPE_ENDERMAN, 1, 1, 4));
358
359 // wolves are added to forests and taigas

449std::vector<Biome::MobSpawnerData*>* Biome::getMobs(MobCategory* category) {
450 if (category == MobCategory::monster) return &enemies;
451 if (category == MobCategory::creature) return &friendlies;
452 if (category == MobCategory::waterCreature) return &waterFriendlies;
453 if (category == MobCategory::creature_chicken) return &friendlies_chicken;
454 if (category == MobCategory::creature_wolf) return &friendlies_wolf;
455 if (category == MobCategory::creature_mushroomcow)
456 return &friendlies_mushroomcow;
457 if (category == MobCategory::ambient) return &ambientFriendlies;
458 return nullptr;
459}

461bool Biome::hasSnow() {
462 // 4J - snowCovered flag removed as it wasn't being set by the game anymore
463 // - snow is now temperature dependent to match code in rain rendering,
