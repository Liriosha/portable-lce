78void Painting::PaintingPostConstructor(int dir, int motive) {
79#ifndef _CONTENT_PACKAGE
80 if (gameServices().debugArtToolsOn() && motive >= 0) {
81 this->motive = (Motive*)Motive::values[motive];
82 setDir(dir);
83 } else
84#endif
85 {
86 std::vector<Motive*>* survivableMotives = new std::vector<Motive*>();
87 for (int i = 0; i < LAST_VALUE; i++) {
88 this->motive = (Motive*)Motive::values[i];
89 setDir(dir);
90 if (survives()) {
91 survivableMotives->push_back(this->motive);
92 }
93 }
94 if (!survivableMotives->empty()) {
95 this->motive = survivableMotives->at(
96 random->nextInt((int)survivableMotives->size()));
97 }
98 setDir(dir);
99 }
100}
