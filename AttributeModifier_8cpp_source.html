72HtmlString AttributeModifier::getHoverText(eATTRIBUTE_ID attribute) {

74 double displayAmount;

76 if (getOperation() == AttributeModifier::OPERATION_MULTIPLY_BASE ||
77 getOperation() == AttributeModifier::OPERATION_MULTIPLY_TOTAL) {
78 displayAmount = getAmount() * 100.0f;
79 } else {
80 displayAmount = getAmount();
81 }
82
83 eMinecraftColour color;
84
85 if (amount > 0) {
86 color = eHTMLColor_9;
87 } else if (amount < 0) {
88 displayAmount *= -1;
89 color = eHTMLColor_c;
90 }
91
92 bool percentage = false;
93 switch (getOperation()) {
94 case AttributeModifier::OPERATION_ADDITION:
95 percentage = false;
96 break;
97 case AttributeModifier::OPERATION_MULTIPLY_BASE:
98 case AttributeModifier::OPERATION_MULTIPLY_TOTAL:
99 percentage = true;
100 break;
101 default:
102 // No other operations
103 assert(0);
104 }
105
106 char formatted[256];
107 snprintf(formatted, 256, "%s%d%s %s", (amount > 0 ? "+" : "-"),
108 (int)displayAmount, (percentage ? "%" : ""),
109 gameServices().getString(Attribute::getName(attribute)));

111 return HtmlString(formatted, color);
