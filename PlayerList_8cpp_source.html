411// Add this function to take some of the code from the PlayerList::add function
412// with the fixes for checking spawn area, especially in the nether. These
413// needed to be done in a different order from before Fix for #13150 - When a

416void PlayerList::validatePlayerSpawnPosition(
417 std::shared_ptr<ServerPlayer> player) {
418 // 4J Stu - Some adjustments to make sure the current players position is
419 // correct Make sure that the player is on the ground, and in the centre x/z
420 // of the current column
421 Log::info("Original pos is %f, %f, %f in dimension %d\n", player->x,
422 player->y, player->z, player->dimension);
423
424 bool spawnForced = player->isRespawnForced();
425
426 double targetX = 0;
427 if (player->x < 0)
428 targetX = std::ceil(player->x) - 0.5;
429 else
430 targetX = std::floor(player->x) + 0.5;
431
432 double targetY = floor(player->y);
433
434 double targetZ = 0;
435 if (player->z < 0)
436 targetZ = std::ceil(player->z) - 0.5;
437 else
438 targetZ = std::floor(player->z) + 0.5;
439
440 player->setPos(targetX, targetY, targetZ);
441
442 Log::info("New pos is %f, %f, %f in dimension %d\n", player->x, player->y,
443 player->z, player->dimension);
444
445 ServerLevel* level = server->getLevel(player->dimension);
446 while (level->getCubes(player, &player->bb)->size() != 0) {
447 player->setPos(player->x, player->y + 1, player->z);
448 }
449 Log::info("Final pos is %f, %f, %f in dimension %d\n", player->x, player->y,
450 player->z, player->dimension);
451
452 // 4J Stu - If we are in the nether and the above while loop has put us
453 // above the nether then we have a problem Finding a valid, safe spawn point
454 // is potentially computationally expensive (may have to hunt through a
455 // large part of the nether) so move the player to their spawn position in
456 // the overworld so that they do not lose their inventory 4J Stu - We also
457 // use this mechanism to force a spawn point in the overworld for players
458 // who were in the save when the reset nether option was applied
459 if (level->dimension->id == -1 && player->y > 125) {
460 Log::info(
461 "Player in the nether tried to spawn at y = %f, moving to "
462 "overworld\n",
463 player->y);
464 player->setLevel(server->getLevel(0));
465 player->gameMode->setLevel(server->getLevel(0));
466 player->dimension = 0;
467
468 level = server->getLevel(player->dimension);
469
470 Pos* levelSpawn = level->getSharedSpawnPos();
471 player->setPos(levelSpawn->x, levelSpawn->y, levelSpawn->z);
472 delete levelSpawn;
473
474 Pos* bedPosition = player->getRespawnPosition();
475 if (bedPosition != nullptr) {
476 Pos* respawnPosition = Player::checkBedValidRespawnPosition(
477 server->getLevel(player->dimension), bedPosition, spawnForced);
478 if (respawnPosition != nullptr) {
479 player->moveTo(respawnPosition->x + 0.5f,
480 respawnPosition->y + 0.1f,
481 respawnPosition->z + 0.5f, 0, 0);
482 player->setRespawnPosition(bedPosition, spawnForced);
483 }
484 delete bedPosition;
485 }
486 while (level->getCubes(player, &player->bb)->size() != 0) {
487 player->setPos(player->x, player->y + 1, player->z);
488 }
489
490 Log::info("Updated pos is %f, %f, %f in dimension %d\n", player->x,
491 player->y, player->z, player->dimension);
492 }

950void PlayerList::repositionAcrossDimension(std::shared_ptr<Entity> entity,
951 int lastDimension,
952 ServerLevel* oldLevel,
953 ServerLevel* newLevel) {
954 double xt = entity->x;
955 double zt = entity->z;

961 newLevel->getLevelData()
962 ->getHellScale(); // 4J Scale was 8 but this is all we can fit in
963 if (entity->dimension == -1) {
964 xt /= scale;
965 zt /= scale;
966 entity->moveTo(xt, entity->y, zt, entity->yRot, entity->xRot);
967 if (entity->isAlive()) {

971 xt *= scale;
972 zt *= scale;
973 entity->moveTo(xt, entity->y, zt, entity->yRot, entity->xRot);
974 if (entity->isAlive()) {
975 oldLevel->tick(entity, false);

977 } else {
978 Pos* p;
979
980 if (lastDimension == 1) {
981 // Coming from the end
982 p = newLevel->getSharedSpawnPos();

984 // Going to the end
985 p = newLevel->getDimensionSpecificSpawn();

988 xt = p->x;
989 entity->y = p->y;
990 zt = p->z;
991 delete p;

994 oldLevel->tick(entity, false);

1005 if (lastDimension != 1) {
1006 xt = (double)std::clamp((int)xt, -Level::MAX_LEVEL_SIZE + 128,
1007 Level::MAX_LEVEL_SIZE - 128);
1008 zt = (double)std::clamp((int)zt, -Level::MAX_LEVEL_SIZE + 128,
1009 Level::MAX_LEVEL_SIZE - 128);
1010 if (entity->isAlive()) {
1011 newLevel->addEntity(entity);
1012 entity->moveTo(xt, entity->y, zt, entity->yRot, entity->xRot);
1013 newLevel->tick(entity, false);
