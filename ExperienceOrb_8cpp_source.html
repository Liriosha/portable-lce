67int ExperienceOrb::getLightColor(float a) {
68 float l = 0.5f;
69 if (l < 0) l = 0;
70 if (l > 1) l = 1;
71 int br = Entity::getLightColor(a);
72
73 int br1 = (br) & 0xff;
74 int br2 = (br >> 16) & 0xff;
75 br1 += (int)(l * 15 * 16);
76 if (br1 > 15 * 16) br1 = 15 * 16;
77 // br2 = 15*16;
78 return br1 | br2 << 16;

81void ExperienceOrb::tick() {
82 Entity::tick();
83 if (throwTime > 0) throwTime--;
84 xo = x;
85 yo = y;
86 zo = z;
87
88 yd -= 0.03f;
89 if (level->getMaterial(Mth::floor(x), Mth::floor(y), Mth::floor(z)) ==
90 Material::lava) {
91 yd = 0.2f;
92 xd = (random->nextFloat() - random->nextFloat()) * 0.2f;
93 zd = (random->nextFloat() - random->nextFloat()) * 0.2f;
94 playSound(eSoundType_RANDOM_FIZZ, 0.4f,
95 2.0f + random->nextFloat() * 0.4f);
96 }
97 checkInTile(x, (bb.y0 + bb.y1) / 2, z);
98
99 double maxDist = 8;
100 // 4J - PC Comment
101 // Usually exp orbs will get created at the same time so smoothen the
102 // lagspikes
103 if (followingTime <
104 tickCount - SharedConstants::TICKS_PER_SECOND + (entityId % 100)) {
105 if (followingPlayer == nullptr ||
106 followingPlayer->distanceToSqr(shared_from_this()) >
107 maxDist * maxDist) {
108 followingPlayer =
109 level->getNearestPlayer(shared_from_this(), maxDist);
110 }
111 followingTime = tickCount;
112 }
113 if (followingPlayer != nullptr) {
114 double xdd = (followingPlayer->x - x) / maxDist;
115 double ydd =
116 (followingPlayer->y + followingPlayer->getHeadHeight() - y) /
117 maxDist;
118 double zdd = (followingPlayer->z - z) / maxDist;
119 double dd = sqrt(xdd * xdd + ydd * ydd + zdd * zdd);
120 double power = 1 - dd;
121 if (power > 0) {
122 power = power * power;
123 xd += xdd / dd * power * 0.1;
124 yd += ydd / dd * power * 0.1;
125 zd += zdd / dd * power * 0.1;
126 }
127 }
128
129 move(xd, yd, zd);
130
131 float friction = 0.98f;
132 if (onGround) {
133 friction = 0.6f * 0.98f;
134 int t =
135 level->getTile(Mth::floor(x), Mth::floor(bb.y0) - 1, Mth::floor(z));
136 if (t > 0) {
137 friction = Tile::tiles[t]->friction * 0.98f;
138 }
139 }
140
141 xd *= friction;
142 yd *= 0.98f;
143 zd *= friction;
144
145 if (onGround) {
146 yd *= -0.9f;
147 }
148
149 tickCount++;
150
151 age++;
152 if (age >= LIFETIME) {
153 remove();
154 }
155}

265bool ExperienceOrb::shouldRender(Vec3* c) {
266 double xd = x - c->x;
267 double yd = y - c->y;
268 double zd = z - c->z;
269 double distance = xd * xd + yd * yd + zd * zd;

271 // 4J - don't render experience orbs that are less than 2 metres away, to
