101LevelChunk* McRegionChunkStorage::load(Level* level, int x, int z) {
102 DataInputStream* regionChunkInputStream =
103 RegionFileCache::getChunkDataInputStream(m_saveFile, m_prefix, x, z);
104
105#if defined(SPLIT_SAVES)
106 // If we can't find the chunk in the save file, then we should remove any
107 // entities we might have for that chunk
108 if (regionChunkInputStream == nullptr) {
109 // 4jcraft fixed cast from int to int64 and taking the mask of the upper
110 // bits and cast to unsigned
111 uint64_t index =
112 ((uint64_t)(uint32_t)(x) << 32) | (((uint64_t)(uint32_t)(z)));
113
114 auto it = m_entityData.find(index);
115 if (it != m_entityData.end()) {
116 m_entityData.erase(it);
117 }
118 }
119#endif
120
121 LevelChunk* levelChunk = nullptr;
122
123 if (m_saveFile->getOriginalSaveVersion() >=
124 SAVE_FILE_VERSION_COMPRESSED_CHUNK_STORAGE) {
125 if (regionChunkInputStream != nullptr) {
126 levelChunk = OldChunkStorage::load(level, regionChunkInputStream);
127 loadEntities(level, levelChunk);
128 regionChunkInputStream->deleteChildStream();
129 delete regionChunkInputStream;
130 }
131 } else {
132 CompoundTag* chunkData;
133 if (regionChunkInputStream != nullptr) {
134 chunkData = NbtIo::read((DataInput*)regionChunkInputStream);
135 } else {
136 return nullptr;
137 }
138
139 regionChunkInputStream->deleteChildStream();
140 delete regionChunkInputStream;
141
142 if (!chunkData->contains("Level")) {
143 char buf[256];
144 sprintf(buf,
145 "Chunk file at %d, %d is missing level data, skipping\n", x,
146 z);
147 Log::info("%s", buf);
148 delete chunkData;
149 return nullptr;
150 }
151 if (!chunkData->getCompound("Level")->contains("Blocks")) {
152 char buf[256];
153 sprintf(buf,
154 "Chunk file at %d, %d is missing block data, skipping\n", x,
155 z);
156 Log::info("%s", buf);
157 delete chunkData;
158 return nullptr;
159 }
160 levelChunk =
161 OldChunkStorage::load(level, chunkData->getCompound("Level"));
162 if (!levelChunk->isAt(x, z)) {
163 char buf[256];
164 sprintf(buf,
165 "Chunk file at %d, %d is in the wrong location; "
166 "relocating. Expected %d, %d, got %d, %d\n",
167 x, z, x, z, levelChunk->x, levelChunk->z);
168 Log::info("%s", buf);
169 delete levelChunk;
170 delete chunkData;
171 return nullptr;
172
173 // 4J Stu - We delete the data within OldChunkStorage::load, so we
174 // can never reload from it
175 // chunkData->putInt("xPos", x);
176 // chunkData->putInt("zPos", z);
177 // levelChunk = OldChunkStorage::load(level,
178 }
179#if defined(SPLIT_SAVES)
180 loadEntities(level, levelChunk);
181#endif
182 delete chunkData;
183 }
184#if !defined(_CONTENT_PACKAGE)
185 if (levelChunk && gameServices().debugSettingsOn() &&
186 gameServices().debugGetMask(PlatformInput.GetPrimaryPad()) &
187 (1L << eDebugSetting_EnableBiomeOverride)) {
188 // 4J Stu - This will force an update of the chunk's biome array
189 levelChunk->reloadBiomes();
190 }
191#endif
192 return levelChunk;
193}

344// 4jcraft: removed the wasting 100ms chunk loading part.
345int McRegionChunkStorage::runSaveThreadProc(void* lpParam) {
346 Compression::CreateNewThreadStorage();
347
348 bool running = true;
349 DataOutputStream* dos = nullptr;
350 while (running) {
351 {
352 std::unique_lock<std::mutex> lock(cs_memory);
353 s_queueCondition.wait(lock,
354 [] { return !s_chunkDataQueue.empty(); });
355 dos = s_chunkDataQueue.front();
356 s_chunkDataQueue.pop_front();
357 s_runningThreadCount++;
358 } // Unlock so the main thread can keep working
359
360 if (dos) {
361 dos->close();
362 dos->deleteChildStream();
363 delete dos;
364 dos = nullptr;
365 }
366
367 {
368 std::lock_guard<std::mutex> lock(cs_memory);
369 s_runningThreadCount--;
370 }
371
372 // Tell the main thread we finished a chunk
373 s_waitCondition.notify_all();
374 }
375
376 Compression::ReleaseThreadStorage();
377 return 0;
378}
