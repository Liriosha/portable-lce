52ScatteredFeaturePieces::ScatteredFeaturePiece::ScatteredFeaturePiece(
53 Random* random, int west, int floor, int north, int width, int height,
54 int depth)
55 : StructurePiece(0) {
56 heightPosition = -1;
57 this->width = width;
58 this->height = height;
59 this->depth = depth;
60
61 orientation = random->nextInt(4);
62
63 LevelGenerationOptions* levelGenOptions =
64 gameServices().getLevelGenerationOptions();
65 if (levelGenOptions != nullptr) {
66 int tempOrientation = 0;
67 if (levelGenOptions->isFeatureChunk(west >> 4, north >> 4,
68 StructureFeature::eFeature_Temples,
69 &tempOrientation)) {
70 orientation = tempOrientation;
71 }
72 }
73
74 switch (orientation) {
75 case Direction::NORTH:
76 case Direction::SOUTH:
77 boundingBox =
78 new BoundingBox(west, floor, north, west + width - 1,
79 floor + height - 1, north + depth - 1);
80 break;
81 default:
82 boundingBox =
83 new BoundingBox(west, floor, north, west + depth - 1,
84 floor + height - 1, north + width - 1);
85 break;
86 }
87}

105bool ScatteredFeaturePieces::ScatteredFeaturePiece::updateAverageGroundHeight(
106 Level* level, BoundingBox* chunkBB, int offset) {
107 if (heightPosition >= 0) {
108 return true;
109 }
110
111 int total = 0;
112 int count = 0;
113 for (int z = boundingBox->z0; z <= boundingBox->z1; z++) {
114 for (int x = boundingBox->x0; x <= boundingBox->x1; x++) {
115 if (chunkBB->isInside(x, 64, z)) {
116 total += std::max(level->getTopSolidBlock(x, z),
117 level->dimension->getSpawnYPosition());
118 count++;
119 }
120 }
121 }
122
123 if (count == 0) {
124 return false;
125 }
126 heightPosition = total / count;
127 boundingBox->move(0, heightPosition - boundingBox->y0 + offset, 0);
128 return true;
129}
