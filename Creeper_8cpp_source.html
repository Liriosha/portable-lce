109void Creeper::tick() {
110 oldSwell = swell;
111 if (isAlive()) {
112 int swellDir = getSwellDir();
113 if (swellDir > 0 && swell == 0) {
114 playSound(eSoundType_RANDOM_FUSE, 1, 0.5f);
115 }
116 swell += swellDir;
117 if (swell < 0) swell = 0;
118 if (swell >= maxSwell) {
119 swell = maxSwell;
120 if (!level->isClientSide) {
121 bool destroyBlocks = level->getGameRules()->getBoolean(
122 GameRules::RULE_MOBGRIEFING);
123 if (isPowered())
124 level->explode(shared_from_this(), x, y, z,
125 explosionRadius * 2, destroyBlocks);
126 else
127 level->explode(shared_from_this(), x, y, z, explosionRadius,
128 destroyBlocks);
129 remove();
130 }
131 }
132 }
133 Monster::tick();
134}
