24MultiPlayerChunkCache::MultiPlayerChunkCache(Level* level) {
25 XZSIZE = level->dimension->getXZSize(); // 4J Added
26 XZOFFSET = XZSIZE / 2; // 4J Added
27 m_XZSize = XZSIZE;
28 hasData = new bool[XZSIZE * XZSIZE];
29 memset(hasData, 0, sizeof(bool) * XZSIZE * XZSIZE);
30
31 std::vector<uint8_t> emptyBlocks(16 * 16 * Level::maxBuildHeight);
32 emptyChunk = new EmptyLevelChunk(level, emptyBlocks, 0, 0);
33
34 // For normal world dimension, create a chunk that can be used to create the
35 // illusion of infinite water at the edge of the world
36 if (level->dimension->id == 0) {
37 std::vector<uint8_t> bytes = std::vector<uint8_t>(16 * 16 * 128);
38
39 // Superflat.... make grass, not water...
40 if (level->getLevelData()->getGenerator() == LevelType::lvl_flat) {
41 for (int x = 0; x < 16; x++)
42 for (int y = 0; y < 128; y++)
43 for (int z = 0; z < 16; z++) {
44 unsigned char tileId = 0;
45 if (y == 3)
46 tileId = Tile::grass_Id;
47 else if (y <= 2)
48 tileId = Tile::dirt_Id;
49
50 bytes[x << 11 | z << 7 | y] = tileId;
51 }
52 } else {
53 for (int x = 0; x < 16; x++)
54 for (int y = 0; y < 128; y++)
55 for (int z = 0; z < 16; z++) {
56 unsigned char tileId = 0;
57 if (y <= (level->getSeaLevel() - 10))
58 tileId = Tile::stone_Id;
59 else if (y < level->getSeaLevel())
60 tileId = Tile::calmWater_Id;
61
62 bytes[x << 11 | z << 7 | y] = tileId;
63 }
64 }
65
66 waterChunk = new WaterLevelChunk(level, bytes, 0, 0);
67
68 if (level->getLevelData()->getGenerator() == LevelType::lvl_flat) {
69 for (int x = 0; x < 16; x++)
70 for (int y = 0; y < 128; y++)
71 for (int z = 0; z < 16; z++) {
72 if (y >= 3) {
73 ((WaterLevelChunk*)waterChunk)
74 ->setLevelChunkBrightness(LightLayer::Sky, x, y,
75 z, 15);
76 }
77 }
78 } else {
79 for (int x = 0; x < 16; x++)
80 for (int y = 0; y < 128; y++)
81 for (int z = 0; z < 16; z++) {
82 if (y >= (level->getSeaLevel() - 1)) {
83 ((WaterLevelChunk*)waterChunk)
84 ->setLevelChunkBrightness(LightLayer::Sky, x, y,
85 z, 15);
86 } else {
87 ((WaterLevelChunk*)waterChunk)
88 ->setLevelChunkBrightness(LightLayer::Sky, x, y,
89 z, 2);
90 }
91 }
92 }
93 } else {
94 waterChunk = nullptr;
95 }
96
97 this->level = level;
98
99 this->cache = new LevelChunk*[XZSIZE * XZSIZE];
100 memset(this->cache, 0, XZSIZE * XZSIZE * sizeof(LevelChunk*));
101}

119// 4J added - find out if we actually really do have a chunk in our cache
120bool MultiPlayerChunkCache::reallyHasChunk(int x, int z) {
121 int ix = x + XZOFFSET;
122 int iz = z + XZOFFSET;
123 // Check we're in range of the stored level - if we aren't, then consider
124 // that we do have that chunk as we'll be able to use the water chunk there
125 if ((ix < 0) || (ix >= XZSIZE)) return true;
126 if ((iz < 0) || (iz >= XZSIZE)) return true;
127 int idx = ix * XZSIZE + iz;
128
129 LevelChunk* chunk = cache[idx];
130 if (chunk == nullptr) {
131 return false;
132 }
133 return hasData[idx];
134}

136void MultiPlayerChunkCache::drop(int x, int z) {
137 // 4J Stu - We do want to drop any entities in the chunks, especially for
138 // the case when a player is dead as they will not get the RemoveEntity
139 // packet if an entity is removed.
140 LevelChunk* chunk = getChunk(x, z);
141 if (!chunk->isEmpty()) {
142 // Added parameter here specifies that we don't want to delete tile
143 // entities, as they won't get recreated unless they've got update
144 // packets The tile entities are in general only created on the client
145 // by virtue of the chunk rebuild
146 chunk->unload(false);
147
148 // 4J - We just want to clear out the entities in the chunk, but
149 // everything else should be valid
150 chunk->loaded = true;
151 }
152}

154LevelChunk* MultiPlayerChunkCache::create(int x, int z) {
155 int ix = x + XZOFFSET;
156 int iz = z + XZOFFSET;
157 // Check we're in range of the stored level
158 if ((ix < 0) || (ix >= XZSIZE))
159 return (waterChunk ? waterChunk : emptyChunk);
160 if ((iz < 0) || (iz >= XZSIZE))
161 return (waterChunk ? waterChunk : emptyChunk);
162 int idx = ix * XZSIZE + iz;
163 LevelChunk* chunk = cache[idx];
164 LevelChunk* lastChunk = chunk;
165
166 if (chunk == nullptr) {
167 {
168 std::unique_lock<std::mutex> lock(m_csLoadCreate);
169
170 // LevelChunk *chunk;
171 if (NetworkService
172 .IsHost()) // force here to disable sharing of data
173 {
174 // 4J-JEV: We are about to use shared data, abort if the server
175 // is stopped and the data is deleted.
176 if (MinecraftServer::getInstance()->serverHalted())
177 return nullptr;
178
179 // If we're the host, then don't create the chunk, share
180 // data from the server's copy
181#ifdef _LARGE_WORLDS
182 LevelChunk* serverChunk =
183 MinecraftServer::getInstance()
184 ->getLevel(level->dimension->id)
185 ->cache->getChunkLoadedOrUnloaded(x, z);
186#else
187 LevelChunk* serverChunk = MinecraftServer::getInstance()
188 ->getLevel(level->dimension->id)
189 ->cache->getChunk(x, z);
190#endif
191 chunk = new LevelChunk(level, x, z, serverChunk);
192 // Let renderer know that this chunk has been created - it might
193 // have made render data from the EmptyChunk if it got to a
194 // chunk before the server sent it
195 level->setTilesDirty(x * 16, 0, z * 16, x * 16 + 15, 127,
196 z * 16 + 15);
197 hasData[idx] = true;
198 } else {
199 // Passing an empty array into the LevelChunk ctor, which it now
200 // detects and sets up the chunk as compressed & empty
201 std::vector<uint8_t> bytes;
202
203 chunk = new LevelChunk(level, bytes, x, z);
204
205 // 4J - changed to use new methods for lighting
206 chunk->setSkyLightDataAllBright();
207 // Arrays::fill(chunk->skyLight->data,
208 //(byte) 255);
209 }
210
211 chunk->loaded = true;
212 }
213
214 LevelChunk* expected = lastChunk;
215 if (std::atomic_ref<LevelChunk*>(cache[idx])
216 .compare_exchange_strong(expected, chunk,
217 std::memory_order_release)) {
218 // If we're sharing with the server, we'll need to calculate our
219 // heightmap now, which isn't shared. If we aren't sharing with the
220 // server, then this will be calculated when the chunk data arrives.
221 if (NetworkService.IsHost()) {
222 chunk->recalcHeightmapOnly();
223 }
224
225 // Successfully updated the cache
226 {
227 std::lock_guard<std::mutex> lock(m_csLoadCreate);
228 loadedChunkList.push_back(chunk);
229 }
230 } else {
231 // Something else must have updated the cache. Return that chunk and
232 // discard this one. This really shouldn't be happening in
233 // multiplayer
234 delete chunk;
235 return cache[idx];
236 }
237
238 } else {
239 chunk->load();
240 }
241
242 return chunk;
243}
