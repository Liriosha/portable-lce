177 dimension = tag->getByte("dimension");
178 x = tag->getInt("xCenter");
179 z = tag->getInt("zCenter");
180 scale = tag->getByte("scale");
181 if (scale < 0) scale = 0;
182 if (scale > MAX_SCALE) scale = MAX_SCALE;
183
184 int width = tag->getShort("width");
185 int height = tag->getShort("height");
186 if (width == MapItem::IMAGE_WIDTH && height == MapItem::IMAGE_HEIGHT) {
187 colors = tag->getByteArray("colors");
188 } else {
189 std::vector<uint8_t> newColors = tag->getByteArray("colors");
190 // 4J - vector manages its own memory, no need to delete old colors
191 // End4J
192 colors =
193 std::vector<uint8_t>(MapItem::IMAGE_WIDTH * MapItem::IMAGE_HEIGHT);
194 int xo = (MapItem::IMAGE_WIDTH - width) / 2;
195 int yo = (MapItem::IMAGE_HEIGHT - height) / 2;
196 for (int y = 0; y < height; y++) {
197 int yt = y + yo;
198 if (yt < 0 && yt >= MapItem::IMAGE_HEIGHT) continue;
199 for (int x = 0; x < width; x++) {
200 int xt = x + xo;
201 if (xt < 0 && xt >= MapItem::IMAGE_WIDTH) continue;
202 colors[xt + yt * MapItem::IMAGE_WIDTH] =
203 newColors[x + y * width];
204 }
205 }
206 }

543void MapItemSavedData::mergeInMapData(
544 std::shared_ptr<MapItemSavedData> dataToAdd) {
545 int w = MapItem::IMAGE_WIDTH;
546 int h = MapItem::IMAGE_HEIGHT;
547
548 for (int x = 0; x < w; x++) {
549 int yd0 = 255;
550 int yd1 = 0;
551
552 for (int z = 0; z < h; z++) {
553 uint8_t oldColor = colors[x + z * w];
554 uint8_t newColor = dataToAdd->colors[x + z * w];
555 if (oldColor == 0 && oldColor != newColor) {
556 if (yd0 > z) yd0 = z;
557 if (yd1 < z) yd1 = z;
558 colors[x + z * w] = newColor;
559 }
560 }
561 if (yd0 <= yd1) {
562 setDirty(x, yd0, yd1);
