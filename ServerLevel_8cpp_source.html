438// which tiles in the current chunks to tick, and is very cache unfriendly by
439// nature) This code now has a thread of its own so it can wait all it wants on
440// the cache without holding the main game thread up. This slightly changes how
441// things are processed, as we now tick the tiles that were determined in the
442// previous tick. Have also limited the amount of tiles to be ticked to 256 (it
443// never seemed to creep up much beyond this in normal play anyway, and we need
444// some finite limit).
445void ServerLevel::tickTiles() {
446 // Index into the arrays used by the update thread
447 int iLev = 0;
448 if (dimension->id == -1) {
449 iLev = 1;
450 } else if (dimension->id == 1) {
451 iLev = 2;
452 }
453 chunksToPoll.clear();
454
455 unsigned int tickCount = 0;
456
457 {
458 std::lock_guard<std::recursive_mutex> lock(m_updateCS[iLev]);
459 // This section processes the tiles that need to be ticked, which we
460 // worked out in the previous tick (or haven't yet, if this is the first
461 // frame)
462 /*int grassTicks = 0;
463 int lavaTicks = 0;
464 int otherTicks = 0;*/
465 for (int i = 0; i < m_updateTileCount[iLev]; i++) {
466 int x = m_updateTileX[iLev][i];
467 int y = m_updateTileY[iLev][i];
468 int z = m_updateTileZ[iLev][i];
469 if (hasChunkAt(x, y, z)) {
470 int id = getTile(x, y, z);
471 if (Tile::tiles[id] != nullptr &&
472 Tile::tiles[id]->isTicking()) {
473 /*if(id == 2) ++grassTicks;
474 else if(id == 11) ++lavaTicks;
475 else ++otherTicks;*/
476 Tile::tiles[id]->tick(this, x, y, z, random);
477 }
478 }
479 }
480 // printf("Total ticks - Grass: %d, Lava: %d, Other: %d, Total: %d\n",
481 // grassTicks, lavaTicks, otherTicks, grassTicks + lavaTicks +
482 // otherTicks);
483 m_updateTileCount[iLev] = 0;
484 m_updateChunkCount[iLev] = 0;
485 }
486
487 Level::tickTiles();
488
489 // AP moved this outside of the loop
490 int prob = 100000;
491 if (gameServices().debugGetMask() & (1L << eDebugSetting_RegularLightning))
492 prob = 100;
493
494 auto itEndCtp = chunksToPoll.end();
495 for (auto it = chunksToPoll.begin(); it != itEndCtp; it++) {
496 ChunkPos cp = *it;
497 int xo = cp.x * 16;
498 int zo = cp.z * 16;
499
500 // 4J added - don't let this actually load/create any chunks, we'll let
501 // the normal updateDirtyChunks etc. processes do that, so it can happen
502 // on another thread
503 if (!this->hasChunk(cp.x, cp.z)) continue;
504
505 // 4J Stu - When adding a 5th player to the game, the number of
506 // chunksToPoll is greater than the size of the m_updateChunkX &
507 // m_updateChunkZ arrays (19*19*4 at time of writing). It doesn't seem

511
512 // The cause of this is largely because the chunksToPoll vector does not
513 // enforce unique elements The java version used a HashSet which would,
514 // although if our world size gets a lot larger then we may have no
515 // overlaps of players surrounding chunks
516 // assert(false);
517
518 // If you hit this assert, then a memory overwrite will occur when you
519 // continue
520 assert(m_updateChunkCount[iLev] < LEVEL_CHUNKS_TO_UPDATE_MAX);
521
522 m_updateChunkX[iLev][m_updateChunkCount[iLev]] = cp.x;
523 m_updateChunkZ[iLev][m_updateChunkCount[iLev]++] = cp.z;
524
525 LevelChunk* lc = getChunk(cp.x, cp.z);
526 tickClientSideTiles(xo, zo, lc);
527
528 if (random->nextInt(prob) == 0 && isRaining() && isThundering()) {
529 randValue = randValue * 3 + addend;
530 int val = (randValue >> 2);
531 int x = xo + (val & 15);
532 int z = zo + ((val >> 8) & 15);
533 int y = getTopRainBlock(x, z);
534
535 if (isRainingAt(x, y, z)) {
536 addGlobalEntity(std::shared_ptr<LightningBolt>(
537 new LightningBolt(this, x, y, z)));
538 }
539 }
540
541 // 4J - changes here brought forrward from 1.2.3
542 if (random->nextInt(16) == 0) {
543 randValue = randValue * 3 + addend;
544 int val = (randValue >> 2);
545 int x = (val & 15);
546 int z = ((val >> 8) & 15);
547 int yy = getTopRainBlock(x + xo, z + zo);
548 if (shouldFreeze(x + xo, yy - 1, z + zo)) {
549 setTileAndUpdate(x + xo, yy - 1, z + zo, Tile::ice_Id);
550 }
551 if (isRaining() && shouldSnow(x + xo, yy, z + zo)) {
552 setTileAndUpdate(x + xo, yy, z + zo, Tile::topSnow_Id);
553 }
554 if (isRaining()) {
555 Biome* b = getBiome(x + xo, z + zo);
556 if (b->hasRain()) {
557 int tile = getTile(x + xo, yy - 1, z + zo);

559 Tile::tiles[tile]->handleRain(this, x + xo, yy - 1,
560 z + zo);
561 }
562 }
563 }
564 }
565
566 // 4J - lighting change brought forward from 1.8.2
567 checkLight(xo + random->nextInt(16), random->nextInt(128),
568 zo + random->nextInt(16));
569 }
570
571 m_level[iLev] = this;
572 m_randValue[iLev] = randValue;
573 // We've set up everything that the udpate thread needs, so kick it off

575}
