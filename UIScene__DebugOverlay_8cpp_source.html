147void UIScene_DebugOverlay::customDraw(IggyCustomDrawCallbackRegion* region) {
148 Minecraft* pMinecraft = Minecraft::GetInstance();
149 if (pMinecraft->localplayers[m_iPad] == nullptr ||
150 pMinecraft->localgameModes[m_iPad] == nullptr)
151 return;
152
153 int itemId = -1;
154 // 4jcraft TODO: UB on our platform since this casts char16_t* to char*
155 sscanf((char*)region->name, "item_%d", &itemId);
156 if (itemId == -1 || itemId > Item::ITEM_NUM_COUNT ||
157 Item::items[itemId] == nullptr) {
158 app.DebugPrintf("This is not the control we are looking for\n");
159 } else {
160 std::shared_ptr<ItemInstance> item =
161 std::shared_ptr<ItemInstance>(new ItemInstance(itemId, 1, 0));
162 if (item != nullptr)
163 customDrawSlotControl(region, m_iPad, item, 1.0f, false, false);
164 }
165}

192void UIScene_DebugOverlay::handlePress(F64 controlId, F64 childId) {
193 switch ((int)controlId) {
194 case eControl_Items: {
195 app.DebugPrintf(
196 "UIScene_DebugOverlay::handlePress for itemsList: %f\n",
197 childId);
198 int id = childId;
199 // app.SetXuiServerAction(m_iPad, eXuiServerAction_DropItem, (void
200 // *)m_itemIds[id]);
201 ClientConnection* conn = Minecraft::GetInstance()->getConnection(
202 PlatformProfile.GetPrimaryPad());
203 conn->send(GiveItemCommand::preparePacket(
204 std::dynamic_pointer_cast<Player>(
205 Minecraft::GetInstance()
206 ->localplayers[PlatformProfile.GetPrimaryPad()]),
207 m_itemIds[id]));
208 } break;
209 case eControl_Mobs: {
210 int id = childId;
211 if (id < m_mobFactories.size()) {
212 MinecraftServer::getInstance()->queueServerAction(
213 minecraft::server::SpawnDebugMob{
214 0, static_cast<int>(m_mobFactories[id])});
215 }
216 } break;
217 case eControl_Enchantments: {
218 int id = childId;
219 ClientConnection* conn = Minecraft::GetInstance()->getConnection(
220 PlatformProfile.GetPrimaryPad());
221 conn->send(EnchantItemCommand::preparePacket(
222 std::dynamic_pointer_cast<Player>(
223 Minecraft::GetInstance()
224 ->localplayers[PlatformProfile.GetPrimaryPad()]),
225 m_enchantmentIdAndLevels[id].first,
226 m_enchantmentIdAndLevels[id].second));
227 } break;
228 case eControl_Schematic: {
229#ifndef _CONTENT_PACKAGE
230 ui.NavigateToScene(PlatformProfile.GetPrimaryPad(),
231 eUIScene_DebugCreateSchematic, nullptr,
232 eUILayer_Debug);
233#endif
234 } break;
235 case eControl_SetCamera: {
236#ifndef _CONTENT_PACKAGE
237 ui.NavigateToScene(PlatformProfile.GetPrimaryPad(),
238 eUIScene_DebugSetCamera, nullptr,
239 eUILayer_Debug);
240#endif
241 } break;
242 case eControl_Rain: {
243 // app.SetXuiServerAction(PlatformProfile.GetPrimaryPad(),eXuiServerAction_ToggleRain);
244 ClientConnection* conn = Minecraft::GetInstance()->getConnection(
245 PlatformProfile.GetPrimaryPad());
246 conn->send(ToggleDownfallCommand::preparePacket());
247 } break;
248 case eControl_Thunder:
249 MinecraftServer::getInstance()->queueServerAction(
250 minecraft::server::ToggleThunder{});
251 break;
252 case eControl_ResetTutorial:
253 Tutorial::debugResetPlayerSavedProgress(
254 PlatformProfile.GetPrimaryPad());
255 break;
256 case eControl_SetDay: {
257 ClientConnection* conn = Minecraft::GetInstance()->getConnection(
258 PlatformProfile.GetPrimaryPad());
259 conn->send(TimeCommand::preparePacket(false));
260 } break;
261 case eControl_SetNight: {
262 ClientConnection* conn = Minecraft::GetInstance()->getConnection(
263 PlatformProfile.GetPrimaryPad());
264 conn->send(TimeCommand::preparePacket(true));
265 } break;
266 };
267}

269void UIScene_DebugOverlay::handleSliderMove(F64 sliderId, F64 currentValue) {
270 switch ((int)sliderId) {
271 case eControl_Time: {
272 Minecraft* pMinecraft = Minecraft::GetInstance();
273
274 // Need to set the time on both levels to stop the flickering as the
275 // local level tries to predict the time Only works if we are on the
276 // host machine, but shouldn't break if not
277 MinecraftServer::SetTime(currentValue * 100);
278 pMinecraft->level->getLevelData()->setGameTime(currentValue * 100);
279
280 char TempString[256];
281 float currentTime = currentValue * 100;
282 snprintf(TempString, 256, "Set time (unsafe) (%d)",
283 (int)currentTime);
284 m_sliderTime.setLabel(TempString);
285 } break;
286 case eControl_FOV: {
287 Minecraft* pMinecraft = Minecraft::GetInstance();
288 pMinecraft->gameRenderer->SetFovVal((float)currentValue);
289
290 char TempString[256];
291 snprintf(TempString, 256, "Set fov (%d)", (int)currentValue);
292 m_sliderFov.setLabel(TempString);
293 } break;
294 };
