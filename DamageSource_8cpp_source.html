214std::shared_ptr<ChatPacket> DamageSource::getDeathMessagePacket(
215 std::shared_ptr<LivingEntity> player) {
216 std::shared_ptr<LivingEntity> source = player->getKillCredit();
217 if (source != nullptr) {
218 return std::make_shared<ChatPacket>(
219 player->getNetworkName(),
220 m_msgWithItemId != ChatPacket::e_ChatCustom ? m_msgWithItemId
221 : m_msgId,
222 source->GetType(), source->getNetworkName());
223 } else {
224 return std::shared_ptr<ChatPacket>(
225 new ChatPacket(player->getNetworkName(), m_msgId));
226 }
227}
