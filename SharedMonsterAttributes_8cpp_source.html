101void SharedMonsterAttributes::loadAttribute(AttributeInstance* instance,
102 CompoundTag* tag) {
103 instance->setBaseValue(tag->getDouble("Base"));
104
105 if (tag->contains("Modifiers")) {
106 ListTag<CompoundTag>* list =
107 (ListTag<CompoundTag>*)tag->getList("Modifiers");
108
109 for (int i = 0; i < list->size(); i++) {
110 AttributeModifier* modifier = loadAttributeModifier(list->get(i));
111 AttributeModifier* old = instance->getModifier(modifier->getId());
112 if (old != nullptr) instance->removeModifier(old);
113 instance->addModifier(modifier);
114 }
115 }
