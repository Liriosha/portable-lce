264void MobEffect::applyEffectTick(std::shared_ptr<LivingEntity> mob,
265 int amplification) {
266 // Maybe move this to separate class implementations in the future?
267 if (id == regeneration->id) {
268 if (mob->getHealth() < mob->getMaxHealth()) {
269 mob->heal(1);
270 }
271 } else if (id == poison->id) {
272 if (mob->getHealth() > 1) {
273 mob->hurt(DamageSource::magic, 1);
274 }
275 } else if (id == wither->id) {
276 mob->hurt(DamageSource::wither, 1);
277 } else if ((id == hunger->id) && mob->instanceof(eTYPE_PLAYER)) {
278 // every tick, cause the same amount of exhaustion as when removing
279 // a block, times amplification
280 std::dynamic_pointer_cast<Player>(mob)->causeFoodExhaustion(
281 FoodConstants::EXHAUSTION_MINE * (amplification + 1));
282 } else if ((id == saturation->id) && mob->instanceof(eTYPE_PLAYER)) {
283 if (!mob->level->isClientSide) {
284 std::dynamic_pointer_cast<Player>(mob)->getFoodData()->eat(
285 amplification + 1, FoodConstants::FOOD_SATURATION_MAX);
286 }
287 } else if ((id == heal->id && !mob->isInvertedHealAndHarm()) ||
288 (id == harm->id && mob->isInvertedHealAndHarm())) {
289 mob->heal(std::max(4 << amplification, 0));
290 } else if ((id == harm->id && !mob->isInvertedHealAndHarm()) ||
291 (id == heal->id && mob->isInvertedHealAndHarm())) {
292 mob->hurt(DamageSource::magic, 6 << amplification);
293 }
294}

296void MobEffect::applyInstantenousEffect(std::shared_ptr<LivingEntity> source,
297 std::shared_ptr<LivingEntity> mob,
298 int amplification, double scale) {
299 if ((id == heal->id && !mob->isInvertedHealAndHarm()) ||
300 (id == harm->id && mob->isInvertedHealAndHarm())) {
301 int amount = (int)(scale * (double)(4 << amplification) + .5);
302 mob->heal(amount);
303 } else if ((id == harm->id && !mob->isInvertedHealAndHarm()) ||
304 (id == heal->id && mob->isInvertedHealAndHarm())) {
305 int amount = (int)(scale * (double)(6 << amplification) + .5);
306 if (source == nullptr) {
307 mob->hurt(DamageSource::magic, amount);
308 } else {
309 DamageSource* damageSource =
310 DamageSource::indirectMagic(mob, source);
311 mob->hurt(damageSource, amount);
312 delete damageSource;
313 }
314 }

329bool MobEffect::isDurationEffectTick(int remainingDuration, int amplification) {
330 // Maybe move this to separate class implementations in the future?
331 if (id == regeneration->id) {
332 // tick intervals are 50, 25, 12, 6..
333 int interval = 50 >> amplification;
334 if (interval > 0) {
335 return (remainingDuration % interval) == 0;
336 }
337 return true;
338 } else if (id == poison->id) {
339 // tick intervals are 25, 12, 6..
340 int interval = 25 >> amplification;
341 if (interval > 0) {
342 return (remainingDuration % interval) == 0;
343 }
344 return true;
345 } else if (id == wither->id) {
346 int interval = 40 >> amplification;
347 if (interval > 0) {
348 return (remainingDuration % interval) == 0;
349 }
350 return true;
351 } else if (id == hunger->id) {
352 return true;
353 }
354
355 return false;
356}

380std::string MobEffect::formatDuration(MobEffectInstance* instance) {
381 if (instance->isNoCounter()) {
382 return "**:**";
383 }
384 int duration = instance->getDuration();
385
386 int seconds = duration / SharedConstants::TICKS_PER_SECOND;
387 int minutes = seconds / 60;
388 seconds %= 60;
389
390 char temp[8];
391 memset(&temp, 0, 8 * (sizeof(char)));
392
393 if (seconds < 10) {
394 snprintf(temp, 8, "%d:0%d", minutes, seconds);
395 // return minutes + ":0" + seconds;
396 } else {
397 snprintf(temp, 8, "%d:%d", minutes, seconds);
398 // return minutes + ":" + seconds;
399 }
400
401 return temp;
402}
