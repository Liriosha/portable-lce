40void FlatLevelSource::prepareHeights(std::vector<uint8_t>& blocks) {
41 int height = blocks.size() / (16 * 16);
42
43 for (int xc = 0; xc < 16; xc++) {
44 for (int zc = 0; zc < 16; zc++) {
45 for (int yc = 0; yc < height; yc++) {
46 int block = 0;
47 if (yc == 0) {
48 block = Tile::unbreakable_Id;
49 } else if (yc <= 2) {
50 block = Tile::dirt_Id;
51 } else if (yc == 3) {
52 block = Tile::grass_Id;
53 }
54 blocks[xc << 11 | zc << 7 | yc] = (uint8_t)block;
55 }
56 }
57 }
58}

62LevelChunk* FlatLevelSource::getChunk(int xOffs, int zOffs) {
63 // 4J - now allocating this with a physical alloc & bypassing general memory
64 // management so that it will get cleanly freed
65 int chunksSize = Level::genDepth * 16 * 16;
66 uint8_t* tileData = (uint8_t*)malloc(chunksSize);
67 memset(tileData, 0, chunksSize);
68 std::vector<uint8_t> blocks =
69 std::vector<uint8_t>(tileData, tileData + chunksSize);
70 // std::vector<uint8_t> blocks = std::vector<uint8_t>(16 * level->depth *
71 // 16);
72 prepareHeights(blocks);
73
74 // LevelChunk *levelChunk = new LevelChunk(level, blocks, xOffs, zOffs);
76 // double[] temperatures = level.getBiomeSource().temperatures;
77
78 if (generateStructures) {
79 villageFeature->apply(this, level, xOffs, zOffs, blocks);
80 }
81
82 // 4J - this now creates compressed block data from the blocks array passed
83 // in, so moved it until after the blocks are actually finalised. We also
84 // now need to free the passed in blocks as the LevelChunk doesn't use the
85 // passed in allocation anymore.
86 LevelChunk* levelChunk = new LevelChunk(level, blocks, xOffs, zOffs);
87 free(tileData);

96void FlatLevelSource::postProcess(ChunkSource* parent, int xt, int zt) {
97 // 4J - changed from random to pprandom so we can run in parallel with
98 // getChunk etc.
99 pprandom->setSeed(level->getSeed());
100 int64_t xScale = pprandom->nextLong() / 2 * 2 + 1;
101 int64_t zScale = pprandom->nextLong() / 2 * 2 + 1;
102 pprandom->setSeed(((xt * xScale) + (zt * zScale)) ^ level->getSeed());
103
104 if (generateStructures) {
105 villageFeature->postProcess(level, pprandom, xt, zt);
106 }
107
108 gameServices().processSchematics(parent->getChunk(xt, zt));
109}
