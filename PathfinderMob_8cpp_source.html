186void PathfinderMob::findRandomStrollLocation(
187 int quadrant /*=-1*/) // 4J - added quadrant
188{
189 bool hasBest = false;
190 int xBest = -1;
191 int yBest = -1;
192 int zBest = -1;
193 float best = -99999;
194 for (int i = 0; i < 10; i++) {
195 // 4J - added quadrant parameter to this method so that the caller can
196 // request that only stroll locations in one quadrant be found. If -1 is
197 // passed then behaviour is the same as the java game
198 int xt, zt;
199 int yt = Mth::floor(y + random->nextInt(7) - 3);
200 if (quadrant == -1) {
201 xt = Mth::floor(x + random->nextInt(13) - 6);
202 zt = Mth::floor(z + random->nextInt(13) - 6);
203 } else {
204 int sx = ((quadrant & 1) ? -1 : 1);
205 int sz = ((quadrant & 2) ? -1 : 1);
206 xt = Mth::floor(x + random->nextInt(7) * sx);
207 zt = Mth::floor(z + random->nextInt(7) * sz);
208 }
209 float value = getWalkTargetValue(xt, yt, zt);
210 if (value > best) {
211 best = value;
212 xBest = xt;
213 yBest = yt;
214 zBest = zt;
215 hasBest = true;
216 }
217 }
218 if (hasBest) {
219 setPath(level->findPath(shared_from_this(), xBest, yBest, zBest, 10,
220 true, false, false,
221 true)); // 4J - changed to setPath from path =
222 }

278void PathfinderMob::tickLeash() {
279 Mob::tickLeash();
280
281 if (isLeashed() && getLeashHolder() != nullptr &&
282 getLeashHolder()->level == this->level) {
283 // soft restriction
284 std::shared_ptr<Entity> leashHolder = getLeashHolder();
285 restrictTo((int)leashHolder->x, (int)leashHolder->y,
286 (int)leashHolder->z, 5);
287
288 float _distanceTo = distanceTo(leashHolder);
289
290 std::shared_ptr<TamableAnimal> tamabaleAnimal =
291 shared_from_this()->instanceof(eTYPE_TAMABLE_ANIMAL)
292 ? std::dynamic_pointer_cast<TamableAnimal>(shared_from_this())
293 : nullptr;
294 if ((tamabaleAnimal != nullptr) && tamabaleAnimal->isSitting()) {
295 if (_distanceTo > 10) {
296 dropLeash(true, true);
297 }
298 return;
299 }
300
301 if (!addedLeashRestrictionGoal) {
302 goalSelector.addGoal(2, &leashRestrictionGoal, false);
303 getNavigation()->setAvoidWater(false);
304 addedLeashRestrictionGoal = true;
305 }
306
307 onLeashDistance(_distanceTo);
308
309 if (_distanceTo > 4) {
310 // harder restriction
311 getNavigation()->moveTo(leashHolder, 1.0);
312 }
313 if (_distanceTo > 6) {
314 // hardest restriction
315 double dx = (leashHolder->x - x) / _distanceTo;
316 double dy = (leashHolder->y - y) / _distanceTo;
317 double dz = (leashHolder->z - z) / _distanceTo;
318
319 xd += dx * std::abs(dx) * .4;
320 yd += dy * std::abs(dy) * .4;
321 zd += dz * std::abs(dz) * .4;
322 }
323 if (_distanceTo > 10) {
324 dropLeash(true, true);
325 }
326
327 } else if (!isLeashed() && addedLeashRestrictionGoal) {
328 addedLeashRestrictionGoal = false;
329 goalSelector.removeGoal(&leashRestrictionGoal);
330 getNavigation()->setAvoidWater(true);
331 clearRestriction();
332 }
333}
