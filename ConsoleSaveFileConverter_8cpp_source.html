29void ConsoleSaveFileConverter::ProcessSimpleFile(ConsoleSaveFile* sourceSave,
30 FileEntry* sourceFileEntry,
31 ConsoleSaveFile* targetSave,
32 FileEntry* targetFileEntry) {
33 unsigned int numberOfBytesRead = 0;
34 unsigned int numberOfBytesWritten = 0;
35
36 std::uint8_t* data = new std::uint8_t[sourceFileEntry->getFileSize()];
37
38 // Read from source
39 sourceSave->readFile(sourceFileEntry, data, sourceFileEntry->getFileSize(),
40 &numberOfBytesRead);
41
42 // Write back to target
43 targetSave->writeFile(targetFileEntry, data, numberOfBytesRead,
44 &numberOfBytesWritten);
45
46 delete[] data;
47}

49void ConsoleSaveFileConverter::ProcessStandardRegionFile(
50 ConsoleSaveFile* sourceSave, File sourceFile, ConsoleSaveFile* targetSave,
51 File targetFile) {
52 unsigned int numberOfBytesWritten = 0;
53 unsigned int numberOfBytesRead = 0;
54
55 RegionFile sourceRegionFile(sourceSave, &sourceFile);
56 RegionFile targetRegionFile(targetSave, &targetFile);
57
58 for (unsigned int x = 0; x < 32; ++x) {
59 for (unsigned int z = 0; z < 32; ++z) {
60 DataInputStream* dis =
61 sourceRegionFile.getChunkDataInputStream(x, z);
62
63 if (dis) {
64 int read = dis->read();
65 DataOutputStream* dos =
66 targetRegionFile.getChunkDataOutputStream(x, z);
67 while (read != -1) {
68 dos->write(read & 0xff);
69
70 read = dis->read();
71 }
72 dos->close();
73 dos->deleteChildStream();
74 delete dos;
75 }
76
77 delete dis;
78 }
