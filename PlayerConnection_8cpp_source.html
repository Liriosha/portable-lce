444void PlayerConnection::handlePlayerAction(
445 std::shared_ptr<PlayerActionPacket> packet) {
446 ServerLevel* level = server->getLevel(player->dimension);
447 player->resetLastActionTime();
448
449 if (packet->action == PlayerActionPacket::DROP_ITEM) {
450 player->drop(false);
451 return;
452 } else if (packet->action == PlayerActionPacket::DROP_ALL_ITEMS) {
453 player->drop(true);
454 return;
455 } else if (packet->action == PlayerActionPacket::RELEASE_USE_ITEM) {
456 player->releaseUsingItem();
457 return;
458 }
459
460 bool shouldVerifyLocation = false;
461 if (packet->action == PlayerActionPacket::START_DESTROY_BLOCK)
462 shouldVerifyLocation = true;
463 if (packet->action == PlayerActionPacket::ABORT_DESTROY_BLOCK)
464 shouldVerifyLocation = true;
465 if (packet->action == PlayerActionPacket::STOP_DESTROY_BLOCK)
466 shouldVerifyLocation = true;
467
468 int x = packet->x;
469 int y = packet->y;
470 int z = packet->z;
471 if (shouldVerifyLocation) {
472 double xDist = player->x - (x + 0.5);
473 // there is a mismatch between the player's camera and the player's
474 // position, so add 1.5 blocks
475 double yDist = player->y - (y + 0.5) + 1.5;
476 double zDist = player->z - (z + 0.5);
477 double dist = xDist * xDist + yDist * yDist + zDist * zDist;
478 if (dist > 6 * 6) {
479 return;
480 }
481 if (y >= server->getMaxBuildHeight()) {
482 return;
483 }
484 }
485
486 if (packet->action == PlayerActionPacket::START_DESTROY_BLOCK) {
487 if (true)
488 player->gameMode->startDestroyBlock(
489 x, y, z,
490 packet->face); // 4J - condition was
491 // !server->isUnderSpawnProtection(level,
492 // x, y, z, player) (from Java 1.6.4)
493 // but putting back to old behaviour
494 else
495 player->connection->send(std::shared_ptr<TileUpdatePacket>(
496 new TileUpdatePacket(x, y, z, level)));
497
498 } else if (packet->action == PlayerActionPacket::STOP_DESTROY_BLOCK) {
499 player->gameMode->stopDestroyBlock(x, y, z);
500 server->getPlayers()->prioritiseTileChanges(
501 x, y, z,
502 level->dimension
503 ->id); // 4J added - make sure that the update packets for this
504 // get prioritised over other general world updates
505 if (level->getTile(x, y, z) != 0)
506 player->connection->send(std::shared_ptr<TileUpdatePacket>(
507 new TileUpdatePacket(x, y, z, level)));
508 } else if (packet->action == PlayerActionPacket::ABORT_DESTROY_BLOCK) {
509 player->gameMode->abortDestroyBlock(x, y, z);
510 if (level->getTile(x, y, z) != 0)
511 player->connection->send(std::shared_ptr<TileUpdatePacket>(
512 new TileUpdatePacket(x, y, z, level)));
513 }

570 // pistonBase_Id to pistonMovingPiece_Id directly sent to the client,
571 // as this will happen on the client as a result of it actioning (via a
572 // tile event) the retraction of the piston locally. However, by putting
573 // a switch beside a piston and then performing an action on the side of
574 // it facing a piston, the following line of code will send a
575 // TileUpdatePacket containing the change to pistonMovingPiece_Id to the
576 // client, and this packet is received before the piston retract action
577 // happens - when the piston retract then occurs, it doesn't work
578 // properly because the piston tile isn't what it is expecting.

580 player->connection->send(std::shared_ptr<TileUpdatePacket>(
581 new TileUpdatePacket(x, y, z, level)));
582 }
583 }
584
585 item = player->inventory->getSelected();
586
587 bool forceClientUpdate = false;

589 forceClientUpdate = true;
590 }

592 player->inventory->items[player->inventory->selected] = nullptr;
593 item = nullptr;
594 }
595

604 player->ignoreSlotUpdateHack = false;
605

608 packet->getItem())) {
609 send(std::shared_ptr<ContainerSetSlotPacket>(
610 new ContainerSetSlotPacket(player->containerMenu->containerId,
611 s->index,
612 player->inventory->getSelected())));
613 }
