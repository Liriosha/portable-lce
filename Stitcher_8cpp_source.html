105// Based on: http://graphics.stanford.edu/~seander/bithacks.html#RoundUpPowerOf2
106int Stitcher::smallestEncompassingPowerOfTwo(int input) {
107 int result = input - 1;
108 result |= result >> 1;
109 result |= result >> 2;
110 result |= result >> 4;
111 result |= result >> 8;
112 result |= result >> 16;
113 return result + 1;
114}

116bool Stitcher::addToStorage(TextureHolder* textureHolder) {
117 for (int i = 0; i < storage.size(); i++) {
118 if (storage.at(i)->add(textureHolder)) {
119 return true;
120 }
121
122 // Try rotated
123 textureHolder->rotate();
124 if (storage.at(i)->add(textureHolder)) {
125 return true;
126 }
127
128 // Undo rotation
129 textureHolder->rotate();
130 }
131
132 return expand(textureHolder);
133}

142bool Stitcher::expand(TextureHolder* textureHolder) {
143 int minDistance =
144 std::min(textureHolder->getHeight(), textureHolder->getWidth());
145 bool firstAddition = storageX == 0 && storageY == 0;
146
147 // It couldn't fit, decide which direction to grow to
148 bool growOnX;
149 if (forcePowerOfTwo) {
150 int xCurrentSize = smallestEncompassingPowerOfTwo(storageX);
151 int yCurrentSize = smallestEncompassingPowerOfTwo(storageY);
152 int xNewSize = smallestEncompassingPowerOfTwo(storageX + minDistance);
153 int yNewSize = smallestEncompassingPowerOfTwo(storageY + minDistance);
154
155 bool xCanGrow = xNewSize <= maxWidth;
156 bool yCanGrow = yNewSize <= maxHeight;
157
158 if (!xCanGrow && !yCanGrow) {
159 return false;
160 }
161
162 // Even if the smallest side fits the larger might not >.>
163 int maxDistance =
164 std::max(textureHolder->getHeight(), textureHolder->getWidth());
165 // TODO: This seems wrong ...
166 if (firstAddition && !xCanGrow &&
167 !(smallestEncompassingPowerOfTwo(storageY + maxDistance) <=
168 maxHeight)) {
169 return false;
170 }
171
172 bool xWillGrow = xCurrentSize != xNewSize;
173 bool yWillGrow = yCurrentSize != yNewSize;
174
175 if (xWillGrow ^ yWillGrow) {
176 // Either grows
177 // only pick X if it can grow AND it wanted to grow
178 // if !xCanGrow then yCanGrow
179
180 growOnX = xWillGrow && xCanGrow;
181 } else {
182 // Both or Neither grow -- smallest side wins
183 growOnX = xCanGrow && xCurrentSize <= yCurrentSize;
184 }
185 } else {
186 // We need to figure out to either expand
187 bool xCanGrow = (storageX + minDistance) <= maxWidth;
188 bool yCanGrow = (storageY + minDistance) <= maxHeight;
189
190 if (!xCanGrow && !yCanGrow) {
191 return false;
192 }
193
194 // Prefer growing on X when its: first addition *or* its the smaller of
195 // the two sides
196 growOnX = (firstAddition || storageX <= storageY) && xCanGrow;
197 }
198
199 StitchSlot* slot;
200 if (growOnX) {
201 if (textureHolder->getWidth() > textureHolder->getHeight()) {
202 textureHolder->rotate();
203 }
204
205 // Grow the 'Y' when it has no size yet
206 if (storageY == 0) {
207 storageY = textureHolder->getHeight();
208 }
209
210 int newSlotWidth = textureHolder->getWidth();
211 // 4J Stu - If we are expanding the texture, then allocate the full
212 // powerOfTwo size that we are going to eventually create
213 if (forcePowerOfTwo) {
214 newSlotWidth =
215 smallestEncompassingPowerOfTwo(storageX + newSlotWidth) -
216 storageX;
217 }
218 slot = new StitchSlot(storageX, 0, newSlotWidth, storageY);
219 // storageX += textureHolder->getWidth();
220 storageX += newSlotWidth;
221 } else {
222 int newSlotHeight = textureHolder->getHeight();
223 // 4J Stu - If we are expanding the texture, then allocate the full
224 // powerOfTwo size that we are going to eventually create
225 if (forcePowerOfTwo) {
226 newSlotHeight =
227 smallestEncompassingPowerOfTwo(storageY + newSlotHeight) -
228 storageY;
229 }
230
231 // grow on Y
232 slot = new StitchSlot(0, storageY, storageX, newSlotHeight);
233 // storageY += textureHolder->getHeight();
234 storageY += newSlotHeight;
235 }
236
237 slot->add(textureHolder);
238 storage.push_back(slot);
239
240 return true;
241}
