95void PendingConnection::sendPreLoginResponse() {
96 // 4J Stu - Calculate the players with UGC privileges set
97 PlayerUID* ugcXuids = new PlayerUID[MINECRAFT_NET_MAX_PLAYERS];
98 std::uint8_t ugcXuidCount = 0;
99 std::uint8_t hostIndex = 0;
100 std::uint8_t ugcFriendsOnlyBits = 0;
101 char szUniqueMapName[14];
102
103 PlatformStorage.GetSaveUniqueFilename(szUniqueMapName);
104
105 PlayerList* playerList = MinecraftServer::getInstance()->getPlayers();
106 for (auto it = playerList->players.begin(); it != playerList->players.end();
107 ++it) {
108 std::shared_ptr<ServerPlayer> player = *it;
109 // If the offline Xuid is invalid but the online one is not then that's
110 // guest which we should ignore If the online Xuid is invalid but the
111 // offline one is not then we are definitely an offline game so dont
112 // care about UGC
113
114 // PADDY - this is failing when a local player with chat restrictions
115 // joins an online game
116
117 if (player != nullptr &&
118 player->connection->m_offlineXUID != INVALID_XUID &&
119 player->connection->m_onlineXUID != INVALID_XUID) {
120 if (player->connection->m_friendsOnlyUGC) {
121 ugcFriendsOnlyBits |= (1 << ugcXuidCount);
122 }
123 // Need to use the online XUID otherwise friend checks will fail on
124 // the client
125 ugcXuids[ugcXuidCount] = player->connection->m_onlineXUID;
126
127 if (player->connection->getNetworkPlayer() != nullptr &&
128 player->connection->getNetworkPlayer()->IsHost())
129 hostIndex = ugcXuidCount;
130
131 ++ugcXuidCount;
132 }
133 }
134
135 {
136 connection->send(std::shared_ptr<PreLoginPacket>(new PreLoginPacket(
137 "-", ugcXuids, ugcXuidCount, ugcFriendsOnlyBits,
138 server->m_ugcPlayersVersion, szUniqueMapName,
139 gameServices().getGameHostOption(eGameHostOption_All), hostIndex,
140 server->m_texturePackId)));
141 }
142}

176void PendingConnection::handleAcceptedLogin(
177 std::shared_ptr<LoginPacket> packet) {
178 if (packet->m_ugcPlayersVersion != server->m_ugcPlayersVersion) {
179 // Send the pre-login packet again with the new list of players
180 sendPreLoginResponse();
181 return;
182 }
183
184 // Guests use the online xuid, everyone else uses the offline one
185 PlayerUID playerXuid = packet->m_offlineXuid;
186 if (playerXuid == INVALID_XUID) playerXuid = packet->m_onlineXuid;
187
188 std::shared_ptr<ServerPlayer> playerEntity =
189 server->getPlayers()->getPlayerForLogin(this, name, playerXuid,
190 packet->m_onlineXuid);
191 if (playerEntity != nullptr) {
192 server->getPlayers()->placeNewPlayer(connection, playerEntity, packet);
193 connection = nullptr; // We've moved responsibility for this over to
194 // the new PlayerConnection, nullptr so we don't
195 // delete our reference to it here in our dtor
196 }
197 done = true;
198}
