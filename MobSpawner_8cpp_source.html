44const int MobSpawner::tick(ServerLevel* level, bool spawnEnemies,
45 bool spawnFriendlies, bool spawnPersistent) {
46 if (!spawnEnemies && !spawnFriendlies && !spawnPersistent) {
47 return 0;
48 }
49 chunksToPoll.clear();
50
51 // 4J - rewritten to add chunks interleaved by player, and to add them from
52 // the centre outwards. We're going to be potentially adding less creatures
53 // than the original so that our count stays consistent with number of
54 // players added, so we want to make sure as best we can that the ones we do
55 // add are near the active players
56 int playerCount = (int)level->players.size();
57 int* xx = new int[playerCount];
58 int* zz = new int[playerCount];
59 for (int i = 0; i < playerCount; i++) {
60 std::shared_ptr<Player> player = level->players[i];
61 xx[i] = Mth::floor(player->x / 16);
62 zz[i] = Mth::floor(player->z / 16);
63 chunksToPoll.insert(
64 std::pair<ChunkPos, bool>(ChunkPos(xx[i], zz[i]), false));
65 }
66
67 for (int r = 1; r <= 8; r++) {
68 for (int l = 0; l < (r * 2); l++) {
69 for (int i = 0; i < playerCount; i++) {
70 bool edgeChunk = (r == 8);
71
72 // If this chunk isn't at the edge of the region for this
73 // player, then always store with a flag of false so that if it
74 // was at the edge of another player, then this will remove that
75 if (!edgeChunk) {
76 chunksToPoll.insert(std::pair<ChunkPos, bool>(
77 ChunkPos((xx[i] - r) + l, (zz[i] - r)), false));
78 chunksToPoll.insert(std::pair<ChunkPos, bool>(
79 ChunkPos((xx[i] + r), (zz[i] - r) + l), false));
80 chunksToPoll.insert(std::pair<ChunkPos, bool>(
81 ChunkPos((xx[i] + r) - l, (zz[i] + r)), false));
82 chunksToPoll.insert(std::pair<ChunkPos, bool>(
83 ChunkPos((xx[i] - r), (zz[i] + r) - l), false));
84 } else {
85 ChunkPos cp = ChunkPos((xx[i] - r) + l, (zz[i] - r));
86 if (chunksToPoll.find(cp) == chunksToPoll.end())
87 chunksToPoll.insert(
88 std::pair<ChunkPos, bool>(cp, true));
89 cp = ChunkPos((xx[i] + r), (zz[i] - r) + l);
90 if (chunksToPoll.find(cp) == chunksToPoll.end())
91 chunksToPoll.insert(
92 std::pair<ChunkPos, bool>(cp, true));
93 cp = ChunkPos((xx[i] + r) - l, (zz[i] + r));
94 if (chunksToPoll.find(cp) == chunksToPoll.end())
95 chunksToPoll.insert(
96 std::pair<ChunkPos, bool>(cp, true));
97 cp = ChunkPos((xx[i] - r), (zz[i] + r) - l);
98 if (chunksToPoll.find(cp) == chunksToPoll.end())
99 chunksToPoll.insert(
100 std::pair<ChunkPos, bool>(cp, true));
101 }
102 }
103 }
104 }
105 delete[] xx;
106 delete[] zz;
107 int count = 0;
108 Pos* spawnPos = level->getSharedSpawnPos();
109
110 for (unsigned int i = 0; i < MobCategory::values.size(); i++) {
111 MobCategory* mobCategory = MobCategory::values[i];
112 if ((mobCategory->isFriendly() && !spawnFriendlies) ||
113 (!mobCategory->isFriendly() && !spawnEnemies) ||
114 (mobCategory->isPersistent() && !spawnPersistent)) {
115 continue;
116 }
117
118 // 4J - early out for non-main dimensions, if spawning anything friendly
119 if (mobCategory->isFriendly()) {
120 if (level->dimension->id != 0) {
121 continue;
122 }
123 }
124
125 // 4J - this is now quite different to the java version. We just have
126 // global max counts for the level whereas the original has a max per
127 // chunk that scales with the number of chunks to be polled.
128 int categoryCount = level->countInstanceOf(
129 mobCategory->getEnumBaseClass(), mobCategory->isSingleType());
130 if (categoryCount >= mobCategory->getMaxInstancesPerLevel()) {
131 continue;
132 }
133
134 auto itEndCTP = chunksToPoll.end();
135 for (auto it = chunksToPoll.begin(); it != itEndCTP; it++) {
136 if (it->second) {
137 // don't add mobs to edge chunks, to prevent adding mobs
138 // "outside" of the active playground
139 continue;
140 }
141 ChunkPos* cp = (ChunkPos*)(&it->first);
142
143 // 4J - don't let this actually create/load a chunk that isn't here
144 // already - we'll let the normal updateDirtyChunks etc. processes
145 // do that, so it can happen on another thread
146 if (!level->hasChunk(cp->x, cp->z)) continue;
147
148 TilePos start = getRandomPosWithin(level, cp->x, cp->z);
149 int xStart = start.x;
150 int yStart = start.y;
151 int zStart = start.z;
152
153 if (level->isSolidBlockingTile(xStart, yStart, zStart)) continue;
154 if (level->getMaterial(xStart, yStart, zStart) !=
155 mobCategory->getSpawnPositionMaterial())
156 continue;
157 int clusterSize = 0;
158
159 for (int dd = 0; dd < 3; dd++) {
160 int x = xStart;
161 int y = yStart;
162 int z = zStart;
163 int ss = 6;
164
165 Biome::MobSpawnerData* currentMobType = nullptr;
166 MobGroupData* groupData = nullptr;
167
168 for (int ll = 0; ll < 4; ll++) {
169 x +=
170 level->random->nextInt(ss) - level->random->nextInt(ss);
171 y += level->random->nextInt(1) - level->random->nextInt(1);
172 z +=
173 level->random->nextInt(ss) - level->random->nextInt(ss);
174 // int y = heightMap[x + z * w] + 1;
175
176 // 4J - don't let this actually create/load a chunk that
177 // isn't here already - we'll let the normal
178 // updateDirtyChunks etc. processes do that, so it can
179 // happen on another thread
180 if (!level->hasChunkAt(x, y, z)) continue;
181
182 if (isSpawnPositionOk(mobCategory, level, x, y, z)) {
183 float xx = x + 0.5f;
184 float yy = (float)y;
185 float zz = z + 0.5f;
186 if (level->getNearestPlayer(
187 xx, yy, zz, MIN_SPAWN_DISTANCE) != nullptr) {
188 continue;
189 } else {
190 float xd = xx - spawnPos->x;
191 float yd = yy - spawnPos->y;
192 float zd = zz - spawnPos->z;
193 float sd = xd * xd + yd * yd + zd * zd;
194 if (sd < MIN_SPAWN_DISTANCE * MIN_SPAWN_DISTANCE) {
195 continue;
196 }
197 }
198
199 if (currentMobType == nullptr) {
200 currentMobType = level->getRandomMobSpawnAt(
201 mobCategory, x, y, z);
202 if (currentMobType == nullptr) {
203 break;
204 }
205 }
206
207 std::shared_ptr<Mob> mob;
208 // 4J - removed try/catch
209 // try
210 // {
211 // mob =
212 // type.mobClass.getConstructor(Level.class).newInstance(level);
213 mob = std::dynamic_pointer_cast<Mob>(
214 EntityIO::newByEnumType(currentMobType->mobClass,
215 level));
216 // }
217 // catch
218 //(exception e)
219 // {
220 // //
221 // TODO 4J We can't print a stack trace, and the
222 // newInstance function doesn't throw an exception just
223 // now anyway
224 // //e.printStackTrace();
225 // return
226 // count;
227 // }
228
229 // 4J - If it is an animal or a monster, don't let any
230 // one type of mob represent more than 50% of the total
231 // amount of these things. This was added initially to
232 // stop flat lands being totally populated with slimes
233 // but seems like a generally good rule.
234 eINSTANCEOF mobType = mob->GetType();
235
236 if ((mobType & eTYPE_ANIMALS_SPAWN_LIMIT_CHECK) ||
237 (mobType & eTYPE_MONSTER)) {
238 // even more special rule for ghasts, because
239 // filling up the nether with 25 of them is a bit
240 // unpleasant. In the java version they are only
241 // limited by the fact that the world fills up with
242 // pig zombies (the only other type of enemy mob in
243 // the nether) before them - they aren't actually
244 // even counted properly themselves
245 if (mobType == eTYPE_GHAST) {
246 if (level->countInstanceOf(mobType, true) >= 4)
247 continue;
248 } else if (mobType == eTYPE_ENDERMAN &&
249 level->dimension->id == 1) {
250 // Special rule for the end, as we only have
251 // Endermen (plus the dragon). Increase the
252 // spawnable counts based on level difficulty
253 int maxEndermen =
254 mobCategory->getMaxInstancesPerLevel();
255
256 if (level->difficulty == Difficulty::NORMAL) {
257 maxEndermen -=
258 mobCategory->getMaxInstancesPerLevel() /
259 4;
260 } else if (level->difficulty <=
261 Difficulty::EASY) {
262 maxEndermen -=
263 mobCategory->getMaxInstancesPerLevel() /
264 2;
265 }
266
267 if (level->countInstanceOf(mobType, true) >=
268 maxEndermen)
269 continue;
270 } else if (level->countInstanceOf(mobType, true) >=
271 (mobCategory->getMaxInstancesPerLevel() /
272 2))
273 continue;
274 }
275
276 mob->moveTo(xx, yy, zz,
277 level->random->nextFloat() * 360, 0);
278
279 if (mob->canSpawn()) {
280 // 4J - check if we are going to despawn straight
281 // away too, and don't add if we will - otherwise
282 // we'll be sending network packets for adding &
283 // removal that we don't need
284 mob->checkDespawn();
285 if (!mob->removed) {
286 clusterSize++;
287 categoryCount++;
288 mob->setDespawnProtected(); // 4J added -
289 // default to
290 // protected
291 // against
292 // despawning
293 level->addEntity(mob);
294 groupData = mob->finalizeMobSpawn(groupData);
295 // 4J - change here so that we can't ever make
296 // more than the desired amount of entities in
297 // each priority. In the original java version
298 // depending on the random spawn positions being
299 // considered the only limit as to the number of
300 // entities created per category is the number
301 // of chunks to poll.
302 if (categoryCount >=
303 mobCategory->getMaxInstancesPerLevel())
304 goto categoryLoop;
305 if (clusterSize >=
306 mob->getMaxSpawnClusterSize())
307 goto chunkLoop;
308 }
309 }
310 count += clusterSize;
311 }
312 }
313 }
314 chunkLoop:
315 continue;
316 }
317 categoryLoop:
318 continue;
319 }
320 delete spawnPos;
321
322 return count;
323}

325bool MobSpawner::isSpawnPositionOk(MobCategory* category, Level* level, int x,
326 int y, int z) {
327 // 4J - don't let this actually create/load a chunk that isn't here already
328 // - we'll let the normal updateDirtyChunks etc. processes do that, so it
329 // can happen on another thread
330 if (!level->hasChunkAt(x, y, z)) return false;
331
332 if (category->getSpawnPositionMaterial() == Material::water) {
333 // 4J - changed to spawn water things only in deep water
334 int yo = 0;
335 int liquidCount = 0;
336
337 while ((y - yo) >= 0 && (yo < 5)) {
338 if (level->getMaterial(x, y - yo, z)->isLiquid()) liquidCount++;
339 yo++;
340 }
341
342 // 4J - Sometimes deep water could be just a waterfall, so check that
343 // it's wide as well
344 bool inEnoughWater = false;
345 if (liquidCount == 5) {
346 if (level->getMaterial(x + 5, y, z)->isLiquid() &&
347 level->getMaterial(x - 5, y, z)->isLiquid() &&
348 level->getMaterial(x, y, z + 5)->isLiquid() &&
349 level->getMaterial(x, y, z - 5)->isLiquid()) {
350 inEnoughWater = true;
351 }
352 }
353
354 return inEnoughWater && !level->isSolidBlockingTile(x, y + 1, z);
355 } else {
356 if (!level->isTopSolidBlocking(x, y - 1, z)) return false;
357 int tt = level->getTile(x, y - 1, z);
358 return tt != Tile::unbreakable_Id &&
359 !level->isSolidBlockingTile(x, y, z) &&
360 !level->getMaterial(x, y, z)->isLiquid() &&
361 !level->isSolidBlockingTile(x, y + 1, z);
362 }
363}
