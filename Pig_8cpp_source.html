101bool Pig::mobInteract(std::shared_ptr<Player> player) {
102 if (!Animal::mobInteract(player)) {
103 if (hasSaddle() && !level->isClientSide &&
104 (rider.lock() == nullptr || rider.lock() == player)) {
105 // 4J HEG - Fixed issue with player not being able to dismount pig
106 // (issue #4479)
107 player->ride(rider.lock() == player ? nullptr : shared_from_this());
108 return true;
109 }
110 return false;
111 }
112 return true;
113}
