275 SharedConstants::TEXTURE_LIGHTING

278 1)); // 4J - change brought forward from 1.8.2

280 Icon* icon = mob->getItemInHandIcon(item, layer);

282 glPopMatrix();

290 Tesselator* t = Tesselator::getInstance();

292 // Consider forcing the mipmap LOD level to use, if this is to be

498
499 float swing1 = sinf(swing * std::numbers::pi);

504 }

510
511 glTranslatef(0.0f, 0.0f * d - (1 - h) * 1.2f - tilt * 0.5f + 0.04f,
512 -0.9f * d);
513

515 glRotatef((tilt) * -85, 0, 0, 1);

951void ItemInHandRenderer::tick() {
952 oHeight = height;
953
954 std::shared_ptr<Player> player = minecraft->player;
955 std::shared_ptr<ItemInstance> nextTile = player->inventory->getSelected();
956
957 bool matches =
958 lastSlot == player->inventory->selected && nextTile == selectedItem;
959 if (selectedItem == nullptr && nextTile == nullptr) {
960 matches = true;
961 }
962 if (nextTile != nullptr && selectedItem != nullptr &&
963 nextTile != selectedItem && nextTile->id == selectedItem->id &&
964 nextTile->getAuxValue() == selectedItem->getAuxValue()) {
965 selectedItem = nextTile;
966 matches = true;
967 }
968
969 float max = 0.4f;
970 float tHeight = matches ? 1.0f : 0;
971 float dd = tHeight - height;
972 if (dd < -max) dd = -max;
973 if (dd > max) dd = max;
974
975 height += dd;
976 if (height < 0.1f) {
977 selectedItem = nextTile;
978 lastSlot = player->inventory->selected;
979 }
980}
