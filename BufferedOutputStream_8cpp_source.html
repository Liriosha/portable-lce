53// Writes len bytes from the specified byte array starting at offset off to this
54// buffered output stream. Ordinarily this method stores bytes from the given
55// array into this stream's buffer, flushing the buffer to the underlying output
56// stream as needed. If the requested length is at least as large as this
57// stream's buffer, however, then this method will flush the buffer and write
58// the bytes directly to the underlying output stream. Thus redundant
59// BufferedOutputStreams will not copy data unnecessarily.
60//
61// Overrides:
62// write in class FilterOutputStream
63// Parameters:
64// b - the data.
65// off - the start offset in the data.
66// len - the number of bytes to write.
67void BufferedOutputStream::write(const std::vector<uint8_t>& b,
68 unsigned int offset, unsigned int length) {
69 // Over the length of what we can store in our buffer - just flush the
70 // buffer and output directly
71 if (length >= buf.size()) {
72 flush();
73 stream->write(b, offset, length);
74 } else {
75 for (unsigned int i = 0; i < length; i++) {
76 write(static_cast<unsigned int>(b[offset + i]));
77 }
78 }
79}
