81void EntityRenderer::renderFlame(std::shared_ptr<Entity> e, double x, double y,
82 double z, float a) {
83 glDisable(GL_LIGHTING);
84
85 Icon* fire1 = Tile::fire->getTextureLayer(0);
86 Icon* fire2 = Tile::fire->getTextureLayer(1);
87
88 glPushMatrix();
89 glTranslatef((float)x, (float)y, (float)z);
90
91 float s = e->bbWidth * 1.4f;
92 glScalef(s, s, s);
93 bindTexture(&TextureAtlas::LOCATION_BLOCKS);
94 Tesselator* t = Tesselator::getInstance();
95
96 float r = 0.5f;
97 float xo = 0.0f;
98
99 float h = e->bbHeight / s;
100 float yo = (float)(e->y - e->bb.y0);
101
102 glRotatef(-entityRenderDispatcher->playerRotY, 0, 1, 0);
103
104 glTranslatef(0, 0, -0.3f + ((int)h) * 0.02f);
105 glColor4f(1, 1, 1, 1);
106 float zo = 0;
107 int ss = 0;
108 t->begin();
109 while (h > 0) {
110 Icon* tex = nullptr;
111 if (ss % 2 == 0) {
112 tex = fire1;
113 } else {
114 tex = fire2;
115 }
116
117 float u0 = tex->getU0();
118 float v0 = tex->getV0();
119 float u1 = tex->getU1();
120 float v1 = tex->getV1();
121
122 if (ss / 2 % 2 == 0) {
123 float tmp = u1;
124 u1 = u0;
125 u0 = tmp;
126 }
127 t->vertexUV((float)(r - xo), (float)(0 - yo), (float)(zo), (float)(u1),
128 (float)(v1));
129 t->vertexUV((float)(-r - xo), (float)(0 - yo), (float)(zo), (float)(u0),
130 (float)(v1));
131 t->vertexUV((float)(-r - xo), (float)(1.4f - yo), (float)(zo),
132 (float)(u0), (float)(v0));
133 t->vertexUV((float)(r - xo), (float)(1.4f - yo), (float)(zo),
134 (float)(u1), (float)(v0));
135 h -= 0.45f;
136 yo -= 0.45f;
137 r *= 0.9f;
138 zo += 0.03f;
139 ss++;
140 }
141 t->end();
142 glPopMatrix();
143 glEnable(GL_LIGHTING);
144}
145void EntityRenderer::renderShadow(std::shared_ptr<Entity> e, double x, double y,
146 double z, float pow, float a) {
147 glDisable(GL_LIGHTING);
148 glEnable(GL_BLEND);
149 glBlendFunc(GL_SRC_ALPHA, GL_ONE_MINUS_SRC_ALPHA);
150
151 entityRenderDispatcher->textures->bindTexture(&SHADOW_LOCATION);
152
153 Level* level = getLevel();
154
155 glDepthMask(false);
156 float r = shadowRadius;
157 float fYLocalPlayerShadowOffset = 0.0f;
158
159 if (e->instanceof(eTYPE_MOB)) {
160 std::shared_ptr<Mob> mob = std::dynamic_pointer_cast<Mob>(e);
161 r *= mob->getSizeScale();
162
163 if (mob->instanceof(eTYPE_ANIMAL)) {
164 if (std::dynamic_pointer_cast<Animal>(mob)->isBaby()) {
165 r *= 0.5f;
166 }
167 }
168 }
169
170 double ex = e->xOld + (e->x - e->xOld) * a;
171 double ey = e->yOld + (e->y - e->yOld) * a + e->getShadowHeightOffs();
172
173 // 4J-PB - local players seem to have a position at their head, and remote
174 // players have a foot position. get the shadow to render by changing the
175 // check here depending on the player type
176 if (e->instanceof(eTYPE_LOCALPLAYER)) {
177 ey -= 1.62;
178 fYLocalPlayerShadowOffset = -1.62f;
179 }
180 double ez = e->zOld + (e->z - e->zOld) * a;
181
182 int x0 = std::floor(ex - r);
183 int x1 = std::floor(ex + r);
184 int y0 = std::floor(ey - r);
185 int y1 = std::floor(ey);
186 int z0 = std::floor(ez - r);
187 int z1 = std::floor(ez + r);
188
189 double xo = x - ex;
190 double yo = y - ey;
191 double zo = z - ez;
192
193 Tesselator* tt = Tesselator::getInstance();
194 tt->begin();
195 for (int xt = x0; xt <= x1; xt++)
196 for (int yt = y0; yt <= y1; yt++)
197 for (int zt = z0; zt <= z1; zt++) {
198 int t = level->getTile(xt, yt - 1, zt);
199 if (t > 0 && level->getRawBrightness(xt, yt, zt) > 3) {
200 renderTileShadow(Tile::tiles[t], x,
201 y + e->getShadowHeightOffs() +
202 fYLocalPlayerShadowOffset,
203 z, xt, yt, zt, pow, r, xo,
204 yo + e->getShadowHeightOffs() +
205 fYLocalPlayerShadowOffset,
206 zo);
207 }
208 }
209 tt->end();
210
211 glColor4f(1, 1, 1, 1);
212 glDisable(GL_BLEND);
213 glDepthMask(true);
214 glEnable(GL_LIGHTING);
215}

219void EntityRenderer::renderTileShadow(Tile* tt, double x, double y, double z,
220 int xt, int yt, int zt, float pow,
221 float r, double xo, double yo,
222 double zo) {
223 Tesselator* t = Tesselator::getInstance();
224 if (!tt->isCubeShaped()) return;
225
226 double a = ((pow - (y - (yt + yo)) / 2) * 0.5f) *
227 getLevel()->getBrightness(xt, yt, zt);
228 if (a < 0) return;
229 if (a > 1) a = 1;
230
231 t->color(1.0f, 1.0f, 1.0f, (float)a);
232 // glColor4f(1, 1, 1, (float) a);
233
234 double x0 = xt + tt->getShapeX0() + xo;
235 double x1 = xt + tt->getShapeX1() + xo;
236 double y0 = yt + tt->getShapeY0() + yo + 1.0 / 64.0f;
237 double z0 = zt + tt->getShapeZ0() + zo;
238 double z1 = zt + tt->getShapeZ1() + zo;
239
240 float u0 = (float)((x - (x0)) / 2 / r + 0.5f);
241 float u1 = (float)((x - (x1)) / 2 / r + 0.5f);
242 float v0 = (float)((z - (z0)) / 2 / r + 0.5f);
243 float v1 = (float)((z - (z1)) / 2 / r + 0.5f);
244
245 // u0 = 0;
246 // v0 = 0;
247 // u1 = 1;
248 // v1 = 1;
249
250 t->vertexUV((float)(x0), (float)(y0), (float)(z0), (float)(u0),
251 (float)(v0));
252 t->vertexUV((float)(x0), (float)(y0), (float)(z1), (float)(u0),
253 (float)(v1));
254 t->vertexUV((float)(x1), (float)(y0), (float)(z1), (float)(u1),
255 (float)(v1));
256 t->vertexUV((float)(x1), (float)(y0), (float)(z0), (float)(u1),
257 (float)(v0));
258}

372void EntityRenderer::postRender(std::shared_ptr<Entity> entity, double x,
373 double y, double z, float rot, float a,
374 bool bRenderPlayerShadow) {
375 if (!entityRenderDispatcher
376 ->isGuiRender) // 4J - added, don't render shadow in gui as it
377 // uses its own blending, and we have globally
378 // enabled blending for interface opacity
379 {
380 if (bRenderPlayerShadow &&
381 entityRenderDispatcher->options->fancyGraphics &&
382 shadowRadius > 0 && !entity->isInvisible()) {
383 double dist = entityRenderDispatcher->distanceToSqr(
384 entity->x, entity->y, entity->z);
385 float pow = (float)((1 - dist / (16.0f * 16.0f)) * shadowStrength);
386 if (pow > 0) {
387 renderShadow(entity, x, y, z, pow, a);
388 }
389 }
390 }
391 if (entity->isOnFire()) renderFlame(entity, x, y, z, a);
392}
