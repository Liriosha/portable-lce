73bool PortalTile::trySpawnPortal(Level* level, int x, int y, int z,
74 bool actuallySpawn) {
75 int xd = 0;
76 int zd = 0;
77 if (level->getTile(x - 1, y, z) == Tile::obsidian_Id ||
78 level->getTile(x + 1, y, z) == Tile::obsidian_Id)
79 xd = 1;
80 if (level->getTile(x, y, z - 1) == Tile::obsidian_Id ||
81 level->getTile(x, y, z + 1) == Tile::obsidian_Id)
82 zd = 1;
83
84 if (xd == zd) return false;
85
86 if (level->getTile(x - xd, y, z - zd) == 0) {
87 x -= xd;
88 z -= zd;
89 }
90
91 for (int xx = -1; xx <= 2; xx++) {
92 for (int yy = -1; yy <= 3; yy++) {
93 bool edge = (xx == -1) || (xx == 2) || (yy == -1) || (yy == 3);
94 if ((xx == -1 || xx == 2) && (yy == -1 || yy == 3)) continue;
95
96 int t = level->getTile(x + xd * xx, y + yy, z + zd * xx);
97
98 if (edge) {
99 if (t != Tile::obsidian_Id) return false;
100 } else {
101 if (t != 0 && t != Tile::fire_Id) return false;
102 }
103 }
104 }
105
106 if (!actuallySpawn) return true;
107
108 for (int xx = 0; xx < 2; xx++) {
109 for (int yy = 0; yy < 3; yy++) {
110 level->setTileAndData(x + xd * xx, y + yy, z + zd * xx,
111 Tile::portalTile_Id, 0, Tile::UPDATE_CLIENTS);
112 }
113 }
114
115 return true;
116}

164bool PortalTile::shouldRenderFace(LevelSource* level, int x, int y, int z,
165 int face) {
166 if (level->getTile(x, y, z) == id) return false;
167
168 bool w =
169 level->getTile(x - 1, y, z) == id && level->getTile(x - 2, y, z) != id;
170 bool e =
171 level->getTile(x + 1, y, z) == id && level->getTile(x + 2, y, z) != id;
172
173 bool n =
174 level->getTile(x, y, z - 1) == id && level->getTile(x, y, z - 2) != id;
175 bool s =
176 level->getTile(x, y, z + 1) == id && level->getTile(x, y, z + 2) != id;
177
178 bool we = w || e;
179 bool ns = n || s;
180
181 if (we && face == 4) return true;
182 if (we && face == 5) return true;
183 if (ns && face == 2) return true;
184 if (ns && face == 3) return true;
185
