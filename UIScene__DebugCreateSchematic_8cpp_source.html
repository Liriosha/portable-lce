87void UIScene_DebugCreateSchematic::handlePress(F64 controlId, F64 childId) {
88 switch ((int)controlId) {
89 case eControl_Create: {
90 // We want the start to be even
91 if (m_data.startX > 0 && m_data.startX % 2 != 0)
92 m_data.startX -= 1;
93 else if (m_data.startX < 0 && m_data.startX % 2 != 0)
94 m_data.startX -= 1;
95 if (m_data.startY < 0)
96 m_data.startY = 0;
97 else if (m_data.startY > 0 && m_data.startY % 2 != 0)
98 m_data.startY -= 1;
99 if (m_data.startZ > 0 && m_data.startZ % 2 != 0)
100 m_data.startZ -= 1;
101 else if (m_data.startZ < 0 && m_data.startZ % 2 != 0)
102 m_data.startZ -= 1;
103
104 // We want the end to be odd to have a total size that is even
105 if (m_data.endX > 0 && m_data.endX % 2 == 0)
106 m_data.endX += 1;
107 else if (m_data.endX < 0 && m_data.endX % 2 == 0)
108 m_data.endX += 1;
109 if (m_data.endY > Level::maxBuildHeight)
110 m_data.endY = Level::maxBuildHeight;
111 else if (m_data.endY > 0 && m_data.endY % 2 == 0)
112 m_data.endY += 1;
113 else if (m_data.endY < 0 && m_data.endY % 2 == 0)
114 m_data.endY += 1;
115 if (m_data.endZ > 0 && m_data.endZ % 2 == 0)
116 m_data.endZ += 1;
117 else if (m_data.endZ < 0 && m_data.endZ % 2 == 0)
118 m_data.endZ += 1;
119
120 MinecraftServer::getInstance()->queueServerAction(m_data);
121
122 navigateBack();
123 } break;
124 case eControl_Name:
125 case eControl_StartX:
126 case eControl_StartY:
127 case eControl_StartZ:
128 case eControl_EndX:
129 case eControl_EndY:
130 case eControl_EndZ:
131 m_keyboardCallbackControl = (eControls)((int)controlId);
132 PlatformInput.RequestKeyboard(
133 "Enter something", "", 0, 25,
134 [this](bool bRes) -> int {
135 return handleKeyboardComplete(bRes);
136 },
137 IPlatformInput::EKeyboardMode_Default);
138 break;
139 };
140}

157int UIScene_DebugCreateSchematic::handleKeyboardComplete(bool bRes) {
158 const char* text = PlatformInput.GetText();
159 if (text[0] != '\0') {
160 std::string value = text;
161 int iVal = 0;
162 if (!value.empty()) iVal = fromWString<int>(value);
163 switch (m_keyboardCallbackControl) {
164 case eControl_Name:
165 m_textInputName.setLabel(value);
166 if (!value.empty()) {
167 snprintf(m_data.name, 64, "%s", value.c_str());

169 snprintf(m_data.name, 64, "schematic");
170 }
171 break;

179 break;
