2054 ThreadStorage* tls = m_tlsShape;
2055 // 4J Stu - Added this so that the TLS shape is correct for this tile
2056 if (tls->tileId != this->id) updateDefaultShape();
2057 if (face == 0 && tls->yy0 > 0) return true;
2058 if (face == 1 && tls->yy1 < 1) return true;
2059 if (face == 2 && tls->zz0 > 0) return true;
2060 if (face == 3 && tls->zz1 < 1) return true;
2061 if (face == 4 && tls->xx0 > 0) return true;

2063 return (!level->isSolidRenderTile(x, y, z));
