15static void sigsegv_handler(int sig) {
16 const char msg[] = "\n=== SIGNAL CAUGHT: ";
17 write(STDERR_FILENO, msg, sizeof(msg) - 1);
18 char signum[8];
19 int len = 0;
20 int s = sig;
21 if (s == 0) {
22 signum[len++] = '0';
23 } else {
24 char tmp[8];
25 int tl = 0;
26 while (s > 0) {
27 tmp[tl++] = '0' + (s % 10);
28 s /= 10;
29 }
30 for (int i = tl - 1; i >= 0; i--) signum[len++] = tmp[i];
31 }
32 write(STDERR_FILENO, signum, len);
33 const char msg1b[] = " ===\n";
34 write(STDERR_FILENO, msg1b, sizeof(msg1b) - 1);
35 void* array[64];
36 int size = backtrace(array, 64);
37 backtrace_symbols_fd(array, size, STDERR_FILENO);
38 const char msg2[] = "=== END BACKTRACE ===\n";
39 write(STDERR_FILENO, msg2, sizeof(msg2) - 1);
40 _exit(139);
41}
