17std::vector<int> FuzzyZoomLayer::getArea(int xo, int yo, int w, int h) {
18 int px = xo >> 1;
19 int py = yo >> 1;
20 int pw = (w >> 1) + 3;
21 int ph = (h >> 1) + 3;
22 std::vector<int> p = parent->getArea(px, py, pw, ph);
23
24 // 4jcraft added casts to unsigned to prevent shift of neg value
25 std::vector<int> tmp(pw * ph * 4);
26 int ww = ((unsigned int)pw << 1);
27 for (int y = 0; y < ph - 1; y++) {
28 int ry = (unsigned int)y << 1;
29 int pp = ry * ww;
30 int ul = p[(0 + 0) + (y + 0) * pw];
31 int dl = p[(0 + 0) + (y + 1) * pw];
32 for (int x = 0; x < pw - 1; x++) {
33 initRandom((unsigned int)(x + px) << 1, (unsigned int)(y + py)
34 << 1);
35 int ur = p[(x + 1) + (y + 0) * pw];
36 int dr = p[(x + 1) + (y + 1) * pw];
37
38 tmp[pp] = ul;
39 tmp[pp++ + ww] = random(ul, dl);
40 tmp[pp] = random(ul, ur);
41 tmp[pp++ + ww] = random(ul, ur, dl, dr);
42
43 ul = ur;
44 dl = dr;
45 }
46 }
47 std::vector<int> result(w * h);
48 for (int y = 0; y < h; y++) {
49 std::copy(
50 tmp.begin() + (y + (yo & 1)) * ((unsigned int)pw << 1) + (xo & 1),
51 tmp.begin() + (y + (yo & 1)) * ((unsigned int)pw << 1) + (xo & 1) +
52 w,
53 result.begin() + y * w);
54 }
55
56 return result;
57}
