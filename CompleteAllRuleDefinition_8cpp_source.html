35void CompleteAllRuleDefinition::updateStatus(GameRule* rule) {
36 int goal = 0;
37 int progress = 0;
38 for (auto it = rule->m_parameters.begin(); it != rule->m_parameters.end();
39 ++it) {
40 if (it->second.isPointer) {
41 goal += it->second.gr->getGameRuleDefinition()->getGoal();
42 progress += it->second.gr->getGameRuleDefinition()->getProgress(
43 it->second.gr);
44 }
45 }
46 if (rule->getConnection() != nullptr) {
47 PacketData data;
48 data.goal = goal;
49 data.progress = progress;
50
51 int icon = -1;
52 int auxValue = 0;
53
54 if (m_lastRuleStatusChanged != nullptr) {
55 icon = m_lastRuleStatusChanged->getIcon();
56 auxValue = m_lastRuleStatusChanged->getAuxValue();
57 m_lastRuleStatusChanged = nullptr;
58 }
59 rule->getConnection()->send(
60 std::shared_ptr<UpdateGameRuleProgressPacket>(
61 new UpdateGameRuleProgressPacket(
62 getActionType(), this->m_descriptionId, icon, auxValue, 0,
63 &data, sizeof(PacketData))));
64 }
65 app.DebugPrintf("Updated CompleteAllRule - Completed %d of %d\n", progress,
66 goal);
67}
