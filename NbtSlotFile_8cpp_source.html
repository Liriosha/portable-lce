174void NbtSlotFile::replaceSlot(int slot, std::vector<CompoundTag*>* tags) {
175 toReplace = fileSlotMap[slot];
176 fileSlotMap[slot] = new std::vector<int>();
177
178 auto itEndTags = tags->end();
179 for (auto it = tags->begin(); it != itEndTags; it++) {
180 CompoundTag* tag = *it; // tags->at(i);
181 std::vector<uint8_t> compressed = NbtIo::compress(tag);
182 if (compressed.size() > largest) {
183 char buf[256];
184 largest = compressed.size();
185#ifndef _CONTENT_PACKAGE
186 snprintf(buf, 256, "New largest: %I64d (%s)\n", largest,
187 tag->getString("id").c_str());
188 OutputDebugStringW(buf);
189#endif
190 }
191
192 int pos = 0;
193 int remaining = compressed.size();
194 if (remaining == 0) continue;
195
196 int nextFileSlot = getFreeSlot();
197 short currentSlot = slot + 1;
198 int lastFileSlot = -1;
199
200 while (remaining > 0) {
201 int fileSlot = nextFileSlot;
202 fileSlotMap[slot]->push_back(fileSlot);
203
204 short toWrite = remaining;
205 if (toWrite > FILE_SLOT_SIZE) {
206 toWrite = FILE_SLOT_SIZE;
207 }
208
209 remaining -= toWrite;
210 if (remaining > 0) {
211 nextFileSlot = getFreeSlot();
212 } else {
213 nextFileSlot = -1;
214 }
215
216 seekSlotHeader(fileSlot);
217 WriteExact(raf, &currentSlot, sizeof(currentSlot));
218 WriteExact(raf, &toWrite, sizeof(toWrite));
219 WriteExact(raf, &nextFileSlot, sizeof(nextFileSlot));
220 WriteExact(raf, &lastFileSlot, sizeof(lastFileSlot));
221
222 seekSlot(fileSlot);
223 WriteExact(raf, compressed.data() + pos, toWrite);
224
225 if (remaining > 0) {
226 lastFileSlot = fileSlot;
227 pos += toWrite;
228 currentSlot = -slot - 1;
229 }
230 }
231 }
232
233 auto itEndToRep = toReplace->end();
234 for (auto it = toReplace->begin(); it != itEndToRep; it++) {
235 int c = *it; // toReplace->at(i);
236
237 freeFileSlots.push_back(c);
238
239 seekSlotHeader(c);
240 short zero = 0;
241 WriteExact(raf, &zero, sizeof(zero));
242 }
243
244 toReplace->clear();
245}
