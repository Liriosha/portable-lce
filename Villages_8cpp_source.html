86 // for (Village village : villages)

137void Villages::addDoorInfos(Pos* pos) {
138 int scanX = 16, scanY = 4, scanZ = 16;
139 for (int xx = pos->x - scanX; xx < pos->x + scanX; xx++) {
140 for (int yy = pos->y - scanY; yy < pos->y + scanY; yy++) {
141 for (int zz = pos->z - scanZ; zz < pos->z + scanZ; zz++) {
142 if (isDoor(xx, yy, zz)) {
143 std::shared_ptr<DoorInfo> currentDoor =
144 getDoorInfo(xx, yy, zz);
145 if (currentDoor == nullptr)
146 createDoorInfo(xx, yy, zz);
147 else
148 currentDoor->timeStamp = _tick;
149 }
150 }
151 }
152 }

170void Villages::createDoorInfo(int x, int y, int z) {
171 int dir = ((DoorTile*)Tile::door_wood)->getDir(level, x, y, z);
172 if (dir == 0 || dir == 2) {
173 int canSeeX = 0;
174 for (int i = -5; i < 0; ++i)
175 if (level->canSeeSky(x + i, y, z)) canSeeX--;
176 for (int i = 1; i <= 5; ++i)
177 if (level->canSeeSky(x + i, y, z)) canSeeX++;
178 if (canSeeX != 0)
179 unclustered.push_back(std::shared_ptr<DoorInfo>(
180 new DoorInfo(x, y, z, canSeeX > 0 ? -2 : 2, 0, _tick)));
181 } else {
182 int canSeeZ = 0;
183 for (int i = -5; i < 0; ++i)
184 if (level->canSeeSky(x, y, z + i)) canSeeZ--;
185 for (int i = 1; i <= 5; ++i)
186 if (level->canSeeSky(x, y, z + i)) canSeeZ++;
187 if (canSeeZ != 0)
