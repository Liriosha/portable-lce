33void LiquidTileDynamic::iterativeTick(Level* level, int x, int y, int z,
34 Random* random) {
35 m_tilesToTick.push_back(LiquidTickData(level, x, y, z, random));
36
37 int failsafe = 100;
38 while ((m_tilesToTick.size() > 0) && (failsafe > 0)) {
39 LiquidTickData tickData = m_tilesToTick.front();
40 m_tilesToTick.pop_front();
41 mainTick(tickData.level, tickData.x, tickData.y, tickData.z,
42 tickData.random);
43 failsafe--;
44 }
45 m_tilesToTick.clear();
46}

64void LiquidTileDynamic::mainTick(Level* level, int x, int y, int z,
65 Random* random) {
66 int depth = getDepth(level, x, y, z);
67
68 int dropOff = 1;
69 if (material == Material::lava && !level->dimension->ultraWarm) dropOff = 2;
70
71 bool becomeStatic = true;
72 int tickDelay = getTickDelay(level);
73 if (depth > 0) {
74 int highest = -100;
75 maxCount = 0;
76 highest = getHighest(level, x - 1, y, z, highest);
77 highest = getHighest(level, x + 1, y, z, highest);
78 highest = getHighest(level, x, y, z - 1, highest);
79 highest = getHighest(level, x, y, z + 1, highest);
80
81 int newDepth = highest + dropOff;
82 if (newDepth >= 8 || highest < 0) {
83 newDepth = -1;
84 }
85 if (getDepth(level, x, y + 1, z) >= 0) {
86 int above = getDepth(level, x, y + 1, z);
87 if (above >= 8)
88 newDepth = above;
89 else
90 newDepth = above + 8;
91 }
92 if (maxCount >= 2 && material == Material::water) {
93 // Only spread spring if it's on top of an existing spring, or
94 // on top of solid ground.
95 if (level->getMaterial(x, y - 1, z)->isSolid()) {
96 newDepth = 0;
97 } else if (level->getMaterial(x, y - 1, z) == material &&
98 level->getData(x, y - 1, z) == 0) {
99 newDepth = 0;
100 }
101 }
102 if (material == Material::lava) {
103 if (depth < 8 && newDepth < 8) {
104 if (newDepth > depth) {
105 if (random->nextInt(4) != 0) {
106 tickDelay = tickDelay * 4;
107 }
108 }
109 }
110 }
111 if (newDepth == depth) {
112 if (becomeStatic) {
113 setStatic(level, x, y, z);
114 }
115 } else {
116 depth = newDepth;
117 if (depth < 0) {
118 level->removeTile(x, y, z);
119 } else {
120 level->setData(x, y, z, depth, Tile::UPDATE_CLIENTS);
121 level->addToTickNextTick(x, y, z, id, tickDelay);
122 level->updateNeighborsAt(x, y, z, id);
123 }
124 }
125 } else {
126 setStatic(level, x, y, z);
127 }
128 if (canSpreadTo(level, x, y - 1, z)) {
129 if (material == Material::lava) {
130 if (level->getMaterial(x, y - 1, z) == Material::water) {
131 level->setTileAndUpdate(x, y - 1, z, Tile::stone_Id);
132 fizz(level, x, y - 1, z);
133 return;
134 }
135 }
136
137 if (depth >= 8)
138 trySpreadTo(level, x, y - 1, z, depth);
139 else
140 trySpreadTo(level, x, y - 1, z, depth + 8);
141 } else if (depth >= 0 &&
142 (depth == 0 || isWaterBlocking(level, x, y - 1, z))) {
143 bool* spreads = getSpread(level, x, y, z);
144 int neighbor = depth + dropOff;
145 if (depth >= 8) {
146 neighbor = 1;
147 }
148 if (neighbor >= 8) return;
149 if (spreads[0]) trySpreadTo(level, x - 1, y, z, neighbor);
150 if (spreads[1]) trySpreadTo(level, x + 1, y, z, neighbor);
151 if (spreads[2]) trySpreadTo(level, x, y, z - 1, neighbor);
152 if (spreads[3]) trySpreadTo(level, x, y, z + 1, neighbor);
153 }
154}

174int LiquidTileDynamic::getSlopeDistance(Level* level, int x, int y, int z,
175 int pass, int from) {
176 int lowest = 1000;
177 for (int d = 0; d < 4; d++) {
178 if (d == 0 && from == 1) continue;
179 if (d == 1 && from == 0) continue;
180 if (d == 2 && from == 3) continue;
181 if (d == 3 && from == 2) continue;
182
183 int xx = x;
184 int yy = y;
185 int zz = z;
186
187 if (d == 0) xx--;
188 if (d == 1) xx++;
189 if (d == 2) zz--;
190 if (d == 3) zz++;
191
192 if (isWaterBlocking(level, xx, yy, zz)) {
193 continue;
194 } else if (level->getMaterial(xx, yy, zz) == material &&
195 level->getData(xx, yy, zz) == 0) {
196 continue;
197 } else {
198 if (isWaterBlocking(level, xx, yy - 1, zz)) {
199 if (pass < 4) {
200 int v = getSlopeDistance(level, xx, yy, zz, pass + 1, d);
201 if (v < lowest) lowest = v;
202 }
203 } else {
204 return pass;
205 }
206 }
207 }
208 return lowest;
209}

211bool* LiquidTileDynamic::getSpread(Level* level, int x, int y, int z) {
212 for (int d = 0; d < 4; d++) {
213 dist[d] = 1000;
214 int xx = x;
215 int yy = y;
216 int zz = z;
217
218 if (d == 0) xx--;
219 if (d == 1) xx++;
220 if (d == 2) zz--;
221 if (d == 3) zz++;
222 if (isWaterBlocking(level, xx, yy, zz)) {
223 continue;
224 } else if (level->getMaterial(xx, yy, zz) == material &&
225 level->getData(xx, yy, zz) == 0) {
226 continue;
227 }
228
229 {
230 if (isWaterBlocking(level, xx, yy - 1, zz)) {
231 dist[d] = getSlopeDistance(level, xx, yy, zz, 1, d);
232 } else {
233 dist[d] = 0;
234 }
235 }
236 }
237
238 int lowest = dist[0];
239 for (int d = 1; d < 4; d++) {
240 if (dist[d] < lowest) lowest = dist[d];
241 }
242
243 for (int d = 0; d < 4; d++) {
244 result[d] = (dist[d] == lowest);
245 }
246 return result;
247}

249bool LiquidTileDynamic::isWaterBlocking(Level* level, int x, int y, int z) {
250 int t = level->getTile(x, y, z);

273bool LiquidTileDynamic::canSpreadTo(Level* level, int x, int y, int z) {
274 // 4J added - don't try and spread out of our restricted map. If we don't do
275 // this check then tiles at the edge of the world will try and spread
276 // outside as the outside tiles report that they contain only air. The fact
277 // that this successfully spreads then updates the neighbours of the tile
278 // outside of the map, one of which is the original tile just inside the
279 // map, which gets set back to being dynamic, and added to the pending ticks
280 // array.
281 int xc = x >> 4;
282 int zc = z >> 4;
283 int ix = xc + (level->chunkSourceXZSize / 2);
284 int iz = zc + (level->chunkSourceXZSize / 2);
285 if ((ix < 0) || (ix >= level->chunkSourceXZSize)) return false;
286 if ((iz < 0) || (iz >= level->chunkSourceXZSize)) return false;
287
288 Material* target = level->getMaterial(x, y, z);
289 if (target == material) return false;
290 if (target == Material::lava) return false;
291 return !isWaterBlocking(level, x, y, z);
292}
