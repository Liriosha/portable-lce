23float WeaponItem::getDestroySpeed(std::shared_ptr<ItemInstance> itemInstance,
24 Tile* tile) {
25 if (tile->id == Tile::web_Id) {
26 // swords can quickly cut web
27 return 15;
28 }
29 // this change modifies which tiles the swords can destroy in creative
30 // mode (>1 == yes)
31 Material* material = tile->material;
32 if (material == Material::plant ||
33 material == Material::replaceable_plant ||
34 material == Material::coral || material == Material::leaves ||
35 material == Material::vegetable) {
36 return 1.5f;
37 }
38 return 1.0f;
39}
