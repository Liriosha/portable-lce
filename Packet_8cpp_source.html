381// sendToAnyClient - true - send to anyone, false - Sends to one person per
382// dimension per machine
383void Packet::map(int id, bool receiveOnClient, bool receiveOnServer,
384 bool sendToAnyClient, bool renderStats,
385 const std::type_info& clazz, packetCreateFn createFn) {
386 idToCreateMap.insert(
387 std::unordered_map<int, packetCreateFn>::value_type(id, createFn));
388
389#if !defined(_CONTENT_PACKAGE)
390#if PACKET_ENABLE_STAT_TRACKING
391 Packet::PacketStatistics* packetStatistics = new PacketStatistics(id);
392 outgoingStatistics[id] = packetStatistics;
393
394 if (renderStats) {
395 renderableStats.push_back(packetStatistics);
396 }
397#endif
398#endif
399
400 if (receiveOnClient) {
401 clientReceivedPackets.insert(id);
402 }
403 if (receiveOnServer) {
404 serverReceivedPackets.insert(id);
405 }
406 if (sendToAnyClient) {
407 sendToAnyClientPackets.insert(id);
408 }
409}
