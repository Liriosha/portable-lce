489 ptrdiff_t delta = (ptrdiff_t)rep_len - (ptrdiff_t)find_len;
490 size_t new_len = src_len + 1;
491 if (delta > 0)
492 new_len += (size_t)delta * count;
493 else
494 new_len -= (size_t)(-delta) * count;
495 result = (char*)malloc(new_len);
496 if (!result) return src;
497
498 tmp = result;
499 while ((pos = strstr(src, find))) {
500 size_t before = (size_t)(pos - src);
501 memcpy(tmp, src, before);
502 tmp += before;
503 memcpy(tmp, rep, rep_len);
504 tmp += rep_len;
505 src = pos + find_len;
506 }
507 memcpy(tmp, src, strlen(src) + 1);
508 free(base);
509 return result;

512static void gdraw_ShaderSourceUpgraded(GLuint shader, GLsizei count,
513 const GLchar** strings,
514 const GLint* lengths) {
515 size_t total = 0;
516 for (int i = 0; i < count; i++)
517 total += lengths ? (lengths[i] >= 0 ? (size_t)lengths[i]
518 : strlen(strings[i]))
519 : strlen(strings[i]);
520
521 char* src = (char*)malloc(total + 1);
522 if (!src) {
523 gdraw_real_shadersource(shader, count, strings, lengths);
524 return;
525 }
526
527 char* dst = src;
528 for (int i = 0; i < count; i++) {
529 size_t len = lengths ? (lengths[i] >= 0 ? (size_t)lengths[i]
530 : strlen(strings[i]))
531 : strlen(strings[i]);
532 memcpy(dst, strings[i], len);
533 dst += len;
534 }
535 *dst = '\0';
536
537 int is_vert = (gdraw_get_shader_type(shader) == GL_VERTEX_SHADER);
538
539 // Strip any existing #version directive as i'll add our own
540 {
541 char* vp = strstr(src, "#version");
542 if (vp) {
543 char* nl = strchr(vp, '\n');
544 if (nl)
545 memmove(vp, nl + 1, strlen(nl + 1) + 1);
546 else
547 *vp = '\0';
548 }
549 }
550
551 // Texture built-ins
552 src = gdraw_strreplace(src, "texture2DRect", "texture");
553 src = gdraw_strreplace(src, "texture2D", "texture");
554
555 // Attribute -> in
556 src = gdraw_strreplace(src, "attribute ", "in ");
557 src = gdraw_strreplace(src, "attribute\t", "in\t");
558 src = gdraw_strreplace(src, "attribute\n", "in\n");
559
560 // Varying -> out (vert) / in (frag)
561 if (is_vert) {
562 src = gdraw_strreplace(src, "varying ", "out ");
563 src = gdraw_strreplace(src, "varying\t", "out\t");
564 src = gdraw_strreplace(src, "varying\n", "out\n");
565 } else {
566 src = gdraw_strreplace(src, "varying ", "in ");
567 src = gdraw_strreplace(src, "varying\t", "in\t");
568 src = gdraw_strreplace(src, "varying\n", "in\n");
569 src = gdraw_strreplace(src, "gl_FragData[0]", "_gdraw_frag_out");
570 src = gdraw_strreplace(src, "gl_FragColor", "_gdraw_frag_out");
571 }
572
573 const char* header = is_vert
574 ? "#version 330 core\n"
575 : "#version 330 core\nout vec4 _gdraw_frag_out;\n";
576 char* patched = (char*)malloc(strlen(header) + strlen(src) + 2);
577 if (!patched) {
578 free(src);
579 gdraw_real_shadersource(shader, count, strings, lengths);
580 return;
581 }
582 strcpy(patched, header);
583 strcat(patched, src);
584 free(src);
585
586 const GLchar* patched_ptr = (const GLchar*)patched;
587 gdraw_real_shadersource(shader, 1, &patched_ptr, NULL);
588 free(patched);
589}
