84void PistonPieceEntity::moveCollidedEntities(float progress, float amount) {
85 if (extending) {
86 progress = 1.0f - progress;
87 } else {
88 progress = progress - 1.0f;
89 }
90
91 auto aabb =
92 Tile::pistonMovingPiece->getAABB(level, x, y, z, id, progress, facing);
93 if (aabb.has_value()) {
94 std::vector<std::shared_ptr<Entity> >* entities =
95 level->getEntities(nullptr, &*aabb);
96 if (!entities->empty()) {
97 std::vector<std::shared_ptr<Entity> > collisionHolder;
98 for (auto it = entities->begin(); it != entities->end(); it++) {
99 collisionHolder.push_back(*it);
100 }
101
102 for (auto it = collisionHolder.begin(); it != collisionHolder.end();
103 it++) {
104 (*it)->move(amount * Facing::STEP_X[facing],
105 amount * Facing::STEP_Y[facing],

124void PistonPieceEntity::tick() {
125 progressO = progress;
126
127 if (progressO >= 1) {
128 moveCollidedEntities(1, 4 / 16.f);
129 level->removeTileEntity(x, y, z);
130 setRemoved();
131 if (level->getTile(x, y, z) == Tile::pistonMovingPiece_Id) {
132 level->setTileAndData(x, y, z, id, data, Tile::UPDATE_ALL);
133 level->neighborChanged(x, y, z, id);
134 }
135 return;
136 }
137
138 progress += .5f;
139 if (progress >= 1) {
140 progress = 1;
141 }
142
143 if (extending) {
144 moveCollidedEntities(progress, (progress - progressO) + 1.0f / 16.0f);
145 }
146}
