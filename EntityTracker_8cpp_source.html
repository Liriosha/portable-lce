97void EntityTracker::addEntity(std::shared_ptr<Entity> e, int range,
98 int updateInterval, bool trackDeltas) {
99 if (range > maxRange) range = maxRange;
100 if (entityMap.find(e->entityId) != entityMap.end()) {
101 assert(false); // Entity already tracked
102 }
103 if (e->entityId >= 2048) {
104 assert(0);
105 }
106 std::shared_ptr<TrackedEntity> te = std::shared_ptr<TrackedEntity>(
107 new TrackedEntity(e, range, updateInterval, trackDeltas));
108 entities.insert(te);
109 entityMap[e->entityId] = te;
110 te->updatePlayers(this, &level->players);
111}

113// 4J - have split removeEntity into two bits - it used to do the equivalent of
114// EntityTracker::removePlayer followed by EntityTracker::removeEntity. This is

117void EntityTracker::removeEntity(std::shared_ptr<Entity> e) {

120 std::shared_ptr<TrackedEntity> te = it->second;
121 entityMap.erase(it);
122 entities.erase(te);

141void EntityTracker::tick() {
142 std::vector<std::shared_ptr<ServerPlayer> > movedPlayers;
143 for (auto it = entities.begin(); it != entities.end(); it++) {
144 std::shared_ptr<TrackedEntity> te = *it;
145 te->tick(this, &level->players);
146 if (te->moved && te->e->GetType() == eTYPE_SERVERPLAYER) {
147 movedPlayers.push_back(
148 std::dynamic_pointer_cast<ServerPlayer>(te->e));
149 }
150 }
151
152 // 4J Stu - If one player on a system is updated, then make sure they all
153 // are as they all have their range extended to include entities visible by
154 // any other player on the system Fix for #11194 - Gameplay: Host player and
155 // their split-screen avatars can become invisible and invulnerable to
156 // client.
157 MinecraftServer* server = MinecraftServer::getInstance();
158 for (unsigned int i = 0; i < server->getPlayers()->players.size(); i++) {
159 std::shared_ptr<ServerPlayer> ep = server->getPlayers()->players[i];
160 if (ep->dimension != level->dimension->id) continue;
161
162 if (ep->connection == nullptr) continue;
163 INetworkPlayer* thisPlayer = ep->connection->getNetworkPlayer();
164 if (thisPlayer == nullptr) continue;
165
166 bool addPlayer = false;
167 for (unsigned int j = 0; j < movedPlayers.size(); j++) {
168 std::shared_ptr<ServerPlayer> sp = movedPlayers[j];
169
170 if (sp == ep) break;
171
172 if (sp->connection == nullptr) continue;
173 INetworkPlayer* otherPlayer = sp->connection->getNetworkPlayer();
174 if (otherPlayer != nullptr &&
175 thisPlayer->IsSameSystem(otherPlayer)) {
176 addPlayer = true;
177 break;
178 }
179 }
180 if (addPlayer) movedPlayers.push_back(ep);
181 }
182
183 for (unsigned int i = 0; i < movedPlayers.size(); i++) {
184 std::shared_ptr<ServerPlayer> player = movedPlayers[i];
185 if (player->connection == nullptr) continue;
186 for (auto it = entities.begin(); it != entities.end(); it++) {
187 std::shared_ptr<TrackedEntity> te = *it;
188 if (te->e != player) {
189 te->updatePlayer(this, player);
190 }
191 }
192 }
193
194 // 4J Stu - We want to do this for dead players as they don't tick normally
195 for (auto it = level->players.begin(); it != level->players.end(); ++it) {
196 std::shared_ptr<ServerPlayer> player =
197 std::dynamic_pointer_cast<ServerPlayer>(*it);
198 if (!player->isAlive()) {
199 player->flushEntitiesToRemove();
200 }
201 }
202}
