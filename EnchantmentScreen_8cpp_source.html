261void EnchantmentScreen::tick() {
262 AbstractContainerScreen::tick();
263
264 // 4jcraft: brought over from UIControl_EnchantmentBook
265 oFlip = flip;
266 oOpen = open;
267
268 std::shared_ptr<ItemInstance> current =
269 enchantMenu->getSlot(EnchantmentMenu::INGREDIENT_SLOT)->getItem();
270 if (!ItemInstance::matches(current, last)) {
271 last = current;
272
273 if (current) {
274 do {
275 flipT += random.nextInt(4) - random.nextInt(4);
276 } while (flip <= flipT + 1 && flip >= flipT - 1);
277 } else {
278 flipT = 0.0f;
279 }
280 }
281
282 bool shouldBeOpen = false;
283 for (int i = 0; i < 3; ++i) {
284 if (enchantMenu->costs[i] != 0) {
285 shouldBeOpen = true;
286 break;
287 }
288 }
289
290 if (shouldBeOpen)
291 open += 0.2f;
292 else
293 open -= 0.2f;
294
295 if (open < 0.0f) open = 0.0f;
296 if (open > 1.0f) open = 1.0f;
297
298 float diff = (flipT - flip) * 0.4f;
299 float max = 0.2f;
300 if (diff < -max) diff = -max;
301 if (diff > max) diff = max;
302 flipA += (diff - flipA) * 0.9f;
303 flip = flip + flipA;
304}
