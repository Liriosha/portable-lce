42void Slot::swap(Slot* other) {
43 std::shared_ptr<ItemInstance> item1 = container->getItem(slot);
44 std::shared_ptr<ItemInstance> item2 =
45 other->container->getItem(other->slot);
46
47 if (item1 != nullptr && item1->count > other->getMaxStackSize()) {
48 if (item2 != nullptr) return;
49 item2 = item1->remove(item1->count - other->getMaxStackSize());
50 }
51 if (item2 != nullptr && item2->count > getMaxStackSize()) {
52 if (item1 != nullptr) return;
53 item1 = item2->remove(item2->count - getMaxStackSize());
54 }
55 other->container->setItem(other->slot, item1);
56
57 container->setItem(slot, item2);
58 setChanged();
59}

97bool Slot::mayCombine(std::shared_ptr<ItemInstance> second) {
98 std::shared_ptr<ItemInstance> first = getItem();
99
100 if (first == nullptr || second == nullptr) return false;
101
102 ArmorItem* thisItem = dynamic_cast<ArmorItem*>(first->getItem());
103 if (thisItem) {
104 bool thisIsDyableArmor =
105 thisItem->getMaterial() == ArmorItem::ArmorMaterial::CLOTH;
106 bool itemIsDye = second->id == Item::dye_powder_Id;
107 return thisIsDyableArmor && itemIsDye;
108 }
109 // 4J Stu - This condition taken from Recipes::getItemFor to repair items,
110 // but added the damaged check to skip when the result is pointless
111 else if (first != nullptr && second != nullptr && first->id == second->id &&
112 first->count == 1 && second->count == 1 &&
113 Item::items[first->id]->canBeDepleted() &&
114 (first->isDamaged() || second->isDamaged())) {
115 // 4J Stu - Don't allow combinining enchanted items, the enchantment
116 // will be lost. They can use the anvil for this
117 return !first->isEnchanted() && !second->isEnchanted();
118 }
119 return false;
120}

122std::shared_ptr<ItemInstance> Slot::combine(
123 std::shared_ptr<ItemInstance> item) {
124 std::shared_ptr<ItemInstance> result = nullptr;
125 std::shared_ptr<ItemInstance> first = getItem();
126
127 std::shared_ptr<CraftingContainer> craftSlots =
128 std::shared_ptr<CraftingContainer>(
129 new CraftingContainer(nullptr, 2, 2));
130 craftSlots->setItem(0, item);
131 craftSlots->setItem(1, first);
132
133 ArmorItem* thisItem = dynamic_cast<ArmorItem*>(first->getItem());
134 if (thisItem) {
135 result = ArmorDyeRecipe::assembleDyedArmor(craftSlots);
136 } else {
137 result = Recipes::getInstance()->getItemFor(craftSlots, nullptr);
138 }
139
140 craftSlots->setItem(0, nullptr);
141 craftSlots->setItem(1, nullptr);
142 return result;
143}
