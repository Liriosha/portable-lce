87void UIScene_SignEntryMenu::tick() {
88 UIScene::tick();
89
90 if (m_bConfirmed) {
91 m_bConfirmed = false;
92
93 // Set the sign text here so we on;y call the verify once it has been
94 // set, not while we're typing in to it
95 for (int i = 0; i < 4; i++) {
96 std::string temp = m_textInputLines[i].getLabel();
97 m_sign->SetMessage(i, temp);
98 }
99
100 m_sign->setChanged();
101
102 Minecraft* pMinecraft = Minecraft::GetInstance();
103 // need to send the new data
104 if (pMinecraft->level->isClientSide) {
105 std::shared_ptr<MultiplayerLocalPlayer> player =
106 pMinecraft->localplayers[m_iPad];
107 if (player != nullptr && player->connection &&
108 player->connection->isStarted()) {
109 player->connection->send(
110 std::shared_ptr<SignUpdatePacket>(new SignUpdatePacket(
111 m_sign->x, m_sign->y, m_sign->z, m_sign->IsVerified(),
112 m_sign->IsCensored(), m_sign->GetMessages())));
113 }
114 }
115 ui.CloseUIScenes(m_iPad);
116 }

119void UIScene_SignEntryMenu::handleInput(int iPad, int key, bool repeat,
120 bool pressed, bool released,
121 bool& handled) {
122 if (m_bConfirmed || m_bIgnoreInput) return;
123
124 ui.AnimateKeyPress(iPad, key, repeat, pressed, released);
125
126 switch (key) {
127 case ACTION_MENU_CANCEL:
128 if (pressed) {
129 // user backed out, so wipe the sign
130 std::string temp = "";
131
132 for (int i = 0; i < 4; i++) {
133 m_sign->SetMessage(i, temp);
134 }
135
136 navigateBack();
137 ui.PlayUISFX(eSFX_Back);
138 handled = true;
139 }
140 break;
141 case ACTION_MENU_OK:
142 case ACTION_MENU_UP:
143 case ACTION_MENU_DOWN:
144 sendInputToMovie(key, repeat, pressed, released);
145 handled = true;
146 break;
147 }
148}

150void UIScene_SignEntryMenu::handlePress(F64 controlId, F64 childId) {
151 switch ((int)controlId) {
152 case eControl_Confirm: {
153 m_bConfirmed = true;
154 } break;
155 case eControl_Line1:
156 case eControl_Line2:
157 case eControl_Line3:
158 case eControl_Line4: {
159 m_iEditingLine = (int)controlId;
160 m_bIgnoreInput = true;
161 PlatformInput.RequestKeyboard(
162 app.GetString(IDS_SIGN_TITLE),
163 m_textInputLines[m_iEditingLine].getLabel(), m_iPad, 15,
164 [this](bool bRes) -> int {
165 // 4J HEG - No reason to set value if keyboard was cancelled
166 m_bIgnoreInput = false;
167 if (bRes && m_iEditingLine >= 0 && m_iEditingLine < 4) {
168 std::string str = PlatformInput.GetText();
169 if (str.size() > 15) str.resize(15);
170 m_textInputLines[m_iEditingLine].setLabel(str);
171 }
172 return 0;
173 },
174 IPlatformInput::EKeyboardMode_Alphabet);
175 } break;
176 }
177}
