139UIScene_Credits::UIScene_Credits(int iPad, void* initData, UILayer* parentLayer)
140 : UIScene(iPad, parentLayer) {
141 // Setup all the Iggy references we need for this scene
142 initialiseMovie();
143
144 m_bAddNextLabel = false;
145
146 // How many lines of text are in the credits?
147 m_iNumTextDefs = MAX_CREDIT_STRINGS;
148
149 // Are there any additional lines needed for the DLC credits?
150 m_iNumTextDefs += app.GetDLCCreditsCount();
151
152 m_iCurrDefIndex = -1;
153
154 // Add the first 20 Flash can cope with
155 for (unsigned int i = 0; i < 20; ++i) {
156 ++m_iCurrDefIndex;
157
158 // Set up the new text element.
159 if (gs_aCreditDefs[i].m_iStringID[0] == NO_TRANSLATED_STRING) {
160 setNextLabel(gs_aCreditDefs[i].m_Text, gs_aCreditDefs[i].m_eType);
161 } else // using additional translated string.
162 {
163 char* creditsString = new char[128];
164 if (gs_aCreditDefs[i].m_iStringID[1] != NO_TRANSLATED_STRING) {
165 snprintf(creditsString, 128, gs_aCreditDefs[i].m_Text,
166 app.GetString(gs_aCreditDefs[i].m_iStringID[0]),
167 app.GetString(gs_aCreditDefs[i].m_iStringID[1]));
168 } else {
169 snprintf(creditsString, 128, gs_aCreditDefs[i].m_Text,
170 app.GetString(gs_aCreditDefs[i].m_iStringID[0]));
171 }
172 setNextLabel(creditsString, gs_aCreditDefs[i].m_eType);
173 delete[] creditsString;
174 }
175 }
176}

193void UIScene_Credits::tick() {
194 UIScene::tick();
195
196 if (m_bAddNextLabel) {
197 m_bAddNextLabel = false;
198
199 const SCreditTextItemDef* pDef;
200
201 // Time to create next text item.
202 ++m_iCurrDefIndex;
203
204 // Wrap back to start.
205 if (m_iCurrDefIndex >= m_iNumTextDefs) {
206 m_iCurrDefIndex = 0;
207 }
208
209 if (m_iCurrDefIndex >= MAX_CREDIT_STRINGS) {
210 app.DebugPrintf("DLC credit %d\n",
211 m_iCurrDefIndex - MAX_CREDIT_STRINGS);
212 // DLC credit
213 pDef = app.GetDLCCredits(m_iCurrDefIndex - MAX_CREDIT_STRINGS);
214 } else {
215 // Get text def for this item.
216 pDef = &(gs_aCreditDefs[m_iCurrDefIndex]);
217 }
218
219 // Set up the new text element.
220 if (pDef->m_Text != nullptr) // 4J-PB - think the RAD logo ones aren't
221 // set up yet and are coming is as null
222 {
223 if (pDef->m_iStringID[0] == CREDIT_ICON) {
224 addImage((ECreditIcons)pDef->m_iStringID[1]);
225 } else // using additional translated string.
226 {
227 std::string sanitisedString = std::string(pDef->m_Text);
228
229 // 4J-JEV: Some DLC credits contain copyright or registered
230 // symbols that are not rendered in some fonts.
231 if (!ui.UsingBitmapFont()) {
232 sanitisedString =
233 replaceAll(sanitisedString, "\u00A9", "(C)");
234 sanitisedString =
235 replaceAll(sanitisedString, "\u00AE", "(R)");
236 sanitisedString =
237 replaceAll(sanitisedString, "\u2013", "-");
238 }
239
240 char* creditsString = new char[128];
241 if (pDef->m_iStringID[0] == NO_TRANSLATED_STRING) {
242 memset(creditsString, 0, 128);
243 memcpy(creditsString, sanitisedString.c_str(),
244 sizeof(char) * sanitisedString.length());
245 } else if (pDef->m_iStringID[1] != NO_TRANSLATED_STRING) {
246 snprintf(creditsString, 128, sanitisedString.c_str(),
247 app.GetString(pDef->m_iStringID[0]),
248 app.GetString(pDef->m_iStringID[1]));
249 } else {
250 snprintf(creditsString, 128, sanitisedString.c_str(),
251 app.GetString(pDef->m_iStringID[0]));
252 }
253
254 setNextLabel(creditsString, pDef->m_eType);
255 delete[] creditsString;
256 }
257 }
258 }

283void UIScene_Credits::setNextLabel(const std::string& label,
284 ECreditTextTypes size) {
285 IggyDataValue result;
286 IggyDataValue value[3];
287
288 IggyStringUTF8 stringVal;
289 stringVal.string = const_cast<char*>(label.c_str());
290 stringVal.length = label.length();
291 value[0].type = IGGY_DATATYPE_string_UTF8;
292 value[0].string8 = stringVal;
293
294 value[1].type = IGGY_DATATYPE_number;
295 value[1].number = (int)size;
296
297 value[2].type = IGGY_DATATYPE_boolean;
298 value[2].boolval = (m_iCurrDefIndex == (m_iNumTextDefs - 1));
299
300 IggyResult out = IggyPlayerCallMethodRS(getMovie(), &result,
301 IggyPlayerRootPath(getMovie()),
302 m_funcSetNextLabel, 3, value);
303}
