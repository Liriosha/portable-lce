63void ServerConnection::tick() {
64 {
65 // MGH - changed this so that the the CS lock doesn't cover the tick
66 // (was causing a lockup when 2 players tried to join)
67 std::vector<std::shared_ptr<PendingConnection> > tempPending;
68 {
69 std::lock_guard<std::mutex> lock(pending_cs);
70 tempPending = pending;
71 }
72
73 for (unsigned int i = 0; i < tempPending.size(); i++) {
74 std::shared_ptr<PendingConnection> uc = tempPending[i];
75 // try { // 4J - removed try/catch
76 uc->tick();
77 // } catch (Exception e) {
78 // uc.disconnect("Internal server error");
79 // logger.log(Level.WARNING, "Failed to handle packet: "
80 // + e, e);
81 // }
82 if (uc->connection != nullptr) uc->connection->flush();
83 }
84 }
85
86 // now remove from the pending list
87 {
88 std::lock_guard<std::mutex> lock(pending_cs);
89 for (unsigned int i = 0; i < pending.size(); i++)
90 if (pending[i]->done) {
91 pending.erase(pending.begin() + i);
92 i--;
93 }
94 }
95
96 for (unsigned int i = 0; i < players.size(); i++) {
97 std::shared_ptr<PlayerConnection> player = players[i];
98 std::shared_ptr<ServerPlayer> serverPlayer = player->getPlayer();
99 if (serverPlayer) {
100 serverPlayer->updateFrameTick();
101 serverPlayer->doChunkSendingTick(false);
102 }
103 player->tick();
104 if (player->done) {
105 players.erase(players.begin() + i);
106 i--;
107 }
108 player->connection->flush();
109 }
110}

112bool ServerConnection::addPendingTextureRequest(
113 const std::string& textureName) {
114 auto it = find(m_pendingTextureRequests.begin(),
115 m_pendingTextureRequests.end(), textureName);
116 if (it == m_pendingTextureRequests.end()) {
117 m_pendingTextureRequests.push_back(textureName);
118 return true;
119 }
120
121 // 4J Stu - We want to request this texture from everyone, if we have a
122 // duplicate it's most likely because the first person we asked for it
123 // didn't have it eg They selected a skin then deleted the skin pack. The
124 // side effect of this change is that in certain cases we can send a few
125 // more requests, and receive a few more responses if people join with the
126 // same skin in a short space of time
127 return true;
