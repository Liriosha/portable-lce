90void UIScene_Keyboard::handleInput(int iPad, int key, bool repeat, bool pressed,
91 bool released, bool& handled) {
92 IggyDataValue result;
93 IggyResult out;
94
95 if (repeat || pressed) {
96 switch (key) {
97 case ACTION_MENU_CANCEL:
98 navigateBack();
99 handled = true;
100 break;
101 case ACTION_MENU_X: // X
102 out = IggyPlayerCallMethodRS(
103 getMovie(), &result, IggyPlayerRootPath(getMovie()),
104 m_funcBackspaceButtonPressed, 0, nullptr);
105 handled = true;
106 break;
107 case ACTION_MENU_PAGEUP: // LT
108 out = IggyPlayerCallMethodRS(
109 getMovie(), &result, IggyPlayerRootPath(getMovie()),
110 m_funcSymbolButtonPressed, 0, nullptr);
111 handled = true;
112 break;
113 case ACTION_MENU_Y: // Y
114 out = IggyPlayerCallMethodRS(
115 getMovie(), &result, IggyPlayerRootPath(getMovie()),
116 m_funcSpaceButtonPressed, 0, nullptr);
117 handled = true;
118 break;
119 case ACTION_MENU_STICK_PRESS: // LS
120 out = IggyPlayerCallMethodRS(
121 getMovie(), &result, IggyPlayerRootPath(getMovie()),
122 m_funcCapsButtonPressed, 0, nullptr);
123 handled = true;
124 break;
125 case ACTION_MENU_LEFT_SCROLL: // LB
126 out = IggyPlayerCallMethodRS(
127 getMovie(), &result, IggyPlayerRootPath(getMovie()),
128 m_funcCursorLeftButtonPressed, 0, nullptr);
129 handled = true;
130 break;
131 case ACTION_MENU_RIGHT_SCROLL: // RB
132 out = IggyPlayerCallMethodRS(
133 getMovie(), &result, IggyPlayerRootPath(getMovie()),
134 m_funcCursorRightButtonPressed, 0, nullptr);
135 handled = true;
136 break;
137 case ACTION_MENU_PAUSEMENU: // Start
138 if (!m_bKeyboardDonePressed) {
139 out = IggyPlayerCallMethodRS(
140 getMovie(), &result, IggyPlayerRootPath(getMovie()),
141 m_funcDoneButtonPressed, 0, nullptr);
142
143 // kick off done timer
144 addTimer(KEYBOARD_DONE_TIMER_ID, KEYBOARD_DONE_TIMER_TIME);
145 m_bKeyboardDonePressed = true;
146 }
147 handled = true;
148 break;
149 }
150 }
151
152 switch (key) {
153 case ACTION_MENU_OK:
154 case ACTION_MENU_LEFT:
155 case ACTION_MENU_RIGHT:
156 case ACTION_MENU_UP:
157 case ACTION_MENU_DOWN:
158 sendInputToMovie(key, repeat, pressed, released);
159 handled = true;
160 break;
161 }
162}

164void UIScene_Keyboard::handlePress(F64 controlId, F64 childId) {
165 if ((int)controlId == 0) {
166 // Done has been pressed. At this point we can query for the input
167 // string and pass it on to wherever it is needed. we can not query for
168 // m_KeyboardTextInput.getLabel() here because we're in an iggy callback
169 // so we need to wait a frame.
170 if (!m_bKeyboardDonePressed) {
171 // kick off done timer
172 addTimer(KEYBOARD_DONE_TIMER_ID, KEYBOARD_DONE_TIMER_TIME);
173 m_bKeyboardDonePressed = true;
174 }
175 }
