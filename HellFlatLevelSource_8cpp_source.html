37void HellFlatLevelSource::prepareHeights(int xOffs, int zOffs,
38 std::vector<uint8_t>& blocks) {
39 int height = blocks.size() / (16 * 16);
40
41 for (int xc = 0; xc < 16; xc++) {
42 for (int zc = 0; zc < 16; zc++) {
43 for (int yc = 0; yc < height; yc++) {
44 int block = 0;
45 if ((yc <= 6) || (yc >= 121)) {
46 block = Tile::netherRack_Id;
47 }
48
49 blocks[xc << 11 | zc << 7 | yc] = (uint8_t)block;
50 }
51 }

55void HellFlatLevelSource::buildSurfaces(int xOffs, int zOffs,
56 std::vector<uint8_t>& blocks) {
57 for (int x = 0; x < 16; x++) {
58 for (int z = 0; z < 16; z++) {
59 for (int y = Level::genDepthMinusOne; y >= 0; y--) {
60 int offs = (z * 16 + x) * Level::genDepth + y;
61
62 // 4J Build walls around the level
63 bool blockSet = false;
64 if (xOffs <= -(m_XZSize / 2)) {
65 if (z - random->nextInt(4) <= 0 ||
66 xOffs < -(m_XZSize / 2)) {
67 blocks[offs] = (uint8_t)Tile::unbreakable_Id;
68 blockSet = true;
69 }
70 }
71 if (zOffs <= -(m_XZSize / 2)) {
72 if (x - random->nextInt(4) <= 0 ||
73 zOffs < -(m_XZSize / 2)) {
74 blocks[offs] = (uint8_t)Tile::unbreakable_Id;
75 blockSet = true;
76 }
77 }
78 if (xOffs >= (m_XZSize / 2) - 1) {
79 if (z + random->nextInt(4) >= 15 ||
80 xOffs > (m_XZSize / 2)) {
81 blocks[offs] = (uint8_t)Tile::unbreakable_Id;
82 blockSet = true;
83 }
84 }
85 if (zOffs >= (m_XZSize / 2) - 1) {
86 if (x + random->nextInt(4) >= 15 ||
87 zOffs > (m_XZSize / 2)) {
88 blocks[offs] = (uint8_t)Tile::unbreakable_Id;
89 blockSet = true;
90 }
91 }
92 if (blockSet) continue;
93 // End 4J Extra to build walls around the level
94
95 if (y >= Level::genDepthMinusOne - random->nextInt(5)) {
96 blocks[offs] = (uint8_t)Tile::unbreakable_Id;
97 } else if (y <= 0 + random->nextInt(5)) {
98 blocks[offs] = (uint8_t)Tile::unbreakable_Id;
99 }
100 }
101 }
102 }

107LevelChunk* HellFlatLevelSource::getChunk(int xOffs, int zOffs) {

149void HellFlatLevelSource::postProcess(ChunkSource* parent, int xt, int zt) {
150 HeavyTile::instaFall = true;
151 int xo = xt * 16;
152 int zo = zt * 16;
153
154 // 4J - added. The original java didn't do any setting of the random seed
155 // here. We'll be running our postProcess in parallel with getChunk etc. so
156 // we need to use a separate random - have used the same initialisation code
157 // as used in RandomLevelSource::postProcess to make sure this random value
158 // is consistent for each world generation. Also changed all uses of random
159 // here to pprandom.
160 pprandom->setSeed(level->getSeed());
161 int64_t xScale = pprandom->nextLong() / 2 * 2 + 1;
162 int64_t zScale = pprandom->nextLong() / 2 * 2 + 1;
163 pprandom->setSeed(((xt * xScale) + (zt * zScale)) ^ level->getSeed());
164
165 int count = pprandom->nextInt(pprandom->nextInt(10) + 1) + 1;
166
167 for (int i = 0; i < count; i++) {
168 int x = xo + pprandom->nextInt(16) + 8;
169 int y = pprandom->nextInt(Level::genDepth - 8) + 4;
170 int z = zo + pprandom->nextInt(16) + 8;
171 HellFireFeature().place(level, pprandom, x, y, z);
172 }
173
174 count = pprandom->nextInt(pprandom->nextInt(10) + 1);
175 for (int i = 0; i < count; i++) {
176 int x = xo + pprandom->nextInt(16) + 8;
177 int y = pprandom->nextInt(Level::genDepth - 8) + 4;
178 int z = zo + pprandom->nextInt(16) + 8;
179 LightGemFeature().place(level, pprandom, x, y, z);
180 }
181
182 HeavyTile::instaFall = false;
183
184 gameServices().processSchematics(parent->getChunk(xt, zt));
185}
