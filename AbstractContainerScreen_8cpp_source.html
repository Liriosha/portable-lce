128void AbstractContainerScreen::renderTooltipInternal(
129 const std::vector<std::string>& cleanedLines,
130 const std::vector<int>& lineColors, int xm, int ym) {
131 if (cleanedLines.empty()) return;
132
133 int tooltipWidth = 0;
134 for (const auto& line : cleanedLines) {
135 int lineWidth = font->width(line);
136 if (lineWidth > tooltipWidth) tooltipWidth = lineWidth;
137 }
138
139 int tooltipX = xm + 12;
140 int tooltipY = ym - 12;
141 int tooltipHeight = 8;
142
143 if (cleanedLines.size() > 1) {
144 tooltipHeight += 2 + (cleanedLines.size() - 1) * 10;
145 }
146
147 int bgColor = 0xf0100010;
148 fillGradient(tooltipX - 3, tooltipY - 4, tooltipX + tooltipWidth + 3,
149 tooltipY - 3, bgColor, bgColor);
150 fillGradient(tooltipX - 3, tooltipY + tooltipHeight + 3,
151 tooltipX + tooltipWidth + 3, tooltipY + tooltipHeight + 4,
152 bgColor, bgColor);
153 fillGradient(tooltipX - 3, tooltipY - 3, tooltipX + tooltipWidth + 3,
154 tooltipY + tooltipHeight + 3, bgColor, bgColor);
155 fillGradient(tooltipX - 4, tooltipY - 3, tooltipX - 3,
156 tooltipY + tooltipHeight + 3, bgColor, bgColor);
157 fillGradient(tooltipX + tooltipWidth + 3, tooltipY - 3,
158 tooltipX + tooltipWidth + 4, tooltipY + tooltipHeight + 3,
159 bgColor, bgColor);
160
161 int borderStart = 0x505000ff;
162 int borderFinish = (borderStart & 0xfefefe) >> 1 | borderStart & 0xff000000;
163 fillGradient(tooltipX - 3, (tooltipY - 3) + 1, (tooltipX - 3) + 1,
164 (tooltipY + tooltipHeight + 3) - 1, borderStart, borderFinish);
165 fillGradient(tooltipX + tooltipWidth + 2, (tooltipY - 3) + 1,
166 tooltipX + tooltipWidth + 3,
167 (tooltipY + tooltipHeight + 3) - 1, borderStart, borderFinish);
168 fillGradient(tooltipX - 3, tooltipY - 3, tooltipX + tooltipWidth + 3,
169 (tooltipY - 3) + 1, borderStart, borderStart);
170 fillGradient(tooltipX - 3, tooltipY + tooltipHeight + 2,
171 tooltipX + tooltipWidth + 3, tooltipY + tooltipHeight + 3,
172 borderFinish, borderFinish);
173
174 int currentY = tooltipY;
175 for (size_t lineIndex = 0; lineIndex < cleanedLines.size(); ++lineIndex) {
176 const std::string& currentLine = cleanedLines[lineIndex];
177 int textColor = lineColors[lineIndex];
178
179 font->drawShadow(currentLine, tooltipX, currentY, textColor);
180
181 if (lineIndex == 0) {
182 currentY += 2;
183 }
184 currentY += 10;
185 }
186}

188void AbstractContainerScreen::renderTooltip(std::shared_ptr<ItemInstance> item,
189 int xm, int ym) {
190 if (item == nullptr) return;
191
192 std::vector<std::string> elementName;
193 std::vector<std::string>* tooltipLines =
194 item->getHoverText(minecraft->player, false, elementName);
195
196 if (tooltipLines != nullptr && tooltipLines->size() > 0) {
197 std::vector<std::string> cleanedLines;
198 std::vector<int> lineColors;
199
200 for (int lineIndex = 0; lineIndex < (int)tooltipLines->size();
201 ++lineIndex) {
202 std::string rawLine = (*tooltipLines)[lineIndex];
203 std::string clean = "";
204 int lineColor = 0xffffffff;
205
206 // 4jcraft: LCE is using HTML font elements for its tooltip
207 // colors, so make sure to parse them for parity w iggy UI
208 //
209 // examples would be enchantment books, potions and music
210 // discs
211 size_t fontPos = rawLine.find("<font");
212 if (fontPos != std::string::npos) {
213 size_t colorPos = rawLine.find("color=\"", fontPos);
214 if (colorPos != std::string::npos) {
215 colorPos += 7;
216 size_t colorEnd = rawLine.find('"', colorPos);
217 if (colorEnd != std::string::npos) {
218 std::string colorStr =
219 rawLine.substr(colorPos, colorEnd - colorPos);
220 if (!colorStr.empty() && colorStr[0] == '#') {
221 colorStr = colorStr.substr(1);
222 }
223 if (!colorStr.empty()) {
224 char* endPtr;
225 long hexColor =
226 strtol(colorStr.c_str(), &endPtr, 16);
227 if (*endPtr == '\0') {
228 lineColor = 0xff000000 | (int)hexColor;
229 }
230 }
231 }
232 }
233 }
234
235 bool inTag = false;
236 for (char currentChar : rawLine) {
237 if (currentChar == '<') {
238 inTag = true;
239 } else if (currentChar == '>') {
240 inTag = false;
241 } else if (!inTag) {
242 clean += currentChar;
243 }
244 }
245
246 cleanedLines.push_back(clean);
247 lineColors.push_back(lineColor);
248 }
249
250 if (!cleanedLines.empty()) {
251 lineColors[0] =
252 gameServices().getHTMLColour(item->getRarity()->color);
253 }
254
255 renderTooltipInternal(cleanedLines, lineColors, xm, ym);
256 }
257}

325bool AbstractContainerScreen::isHoveringOver(int x, int y, int w, int h, int xm,
326 int ym) {
327 int xo = (width - imageWidth) / 2;
328 int yo = (height - imageHeight) / 2;
329 xm -= xo;
330 ym -= yo;
331
