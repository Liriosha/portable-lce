75void ItemEntity::tick() {
76 Entity::tick();
77
78 if (throwTime > 0) throwTime--;
79 xo = x;
80 yo = y;
81 zo = z;
82
83 yd -= 0.04f;
84 noPhysics = checkInTile(x, (bb.y0 + bb.y1) / 2, z);
85
86 // 4J - added parameter here so that these don't care about colliding with
87 // other entities
88 move(xd, yd, zd, true);
89
90 bool moved = (int)xo != (int)x || (int)yo != (int)y || (int)zo != (int)z;
91
92 if (moved || tickCount % 25 == 0) {
93 if (level->getMaterial(Mth::floor(x), Mth::floor(y), Mth::floor(z)) ==
94 Material::lava) {
95 yd = 0.2f;
96 xd = (random->nextFloat() - random->nextFloat()) * 0.2f;
97 zd = (random->nextFloat() - random->nextFloat()) * 0.2f;
98 playSound(eSoundType_RANDOM_FIZZ, 0.4f,
99 2.0f + random->nextFloat() * 0.4f);
100 }
101
102 if (!level->isClientSide) {
103 mergeWithNeighbours();
104 }
105 }
106
107 float friction = 0.98f;
108 if (onGround) {
109 friction = 0.6f * 0.98f;
110 int t =
111 level->getTile(Mth::floor(x), Mth::floor(bb.y0) - 1, Mth::floor(z));
112 if (t > 0) {
113 friction = Tile::tiles[t]->friction * 0.98f;
114 }
115 }
116
117 xd *= friction;
118 yd *= 0.98f;
119 zd *= friction;
120
121 if (onGround) {
122 yd *= -0.5f;
123 }
124
125 tickCount++;
126 age++;
127 if (!level->isClientSide && age >= LIFETIME) {
128 remove();
129 }
130}

132void ItemEntity::mergeWithNeighbours() {
133 AABB grown = bb.grow(0.5, 0, 0.5);
134 std::vector<std::shared_ptr<Entity> >* neighbours =
135 level->getEntitiesOfClass(typeid(*this), &grown);
136 for (auto it = neighbours->begin(); it != neighbours->end(); ++it) {
137 std::shared_ptr<ItemEntity> entity =

139 merge(entity);

144bool ItemEntity::merge(std::shared_ptr<ItemEntity> target) {
145 if (target == shared_from_this()) return false;
146 if (!target->isAlive() || !this->isAlive()) return false;
147 std::shared_ptr<ItemInstance> myItem = this->getItem();
148 std::shared_ptr<ItemInstance> targetItem = target->getItem();
149
150 if (targetItem->getItem() != myItem->getItem()) return false;
151 if (targetItem->hasTag() ^ myItem->hasTag()) return false;
152 if (targetItem->hasTag() && !targetItem->getTag()->equals(myItem->getTag()))
153 return false;
154 if (targetItem->getItem()->isStackedByData() &&
155 targetItem->getAuxValue() != myItem->getAuxValue())
156 return false;
157 if (targetItem->count < myItem->count)
158 return target->merge(
159 std::dynamic_pointer_cast<ItemEntity>(shared_from_this()));
160 if (targetItem->count + myItem->count > targetItem->getMaxStackSize())
161 return false;
162
163 targetItem->count += myItem->count;
164 target->throwTime = std::max(target->throwTime, this->throwTime);
165 target->age = std::min(target->age, this->age);
166 target->setItem(targetItem);
167 remove();
168
169 return true;
170}

183bool ItemEntity::hurt(DamageSource* source, float damage) {
184 // 4J - added next line: found whilst debugging an issue with item entities
185 // getting into a bad state when being created by a cactus, since entities
186 // insides cactuses get hurt and therefore depending on the timing of things
187 // they could get removed from the client when they weren't supposed to be.
188 // Are there really any cases were we would want an itemEntity to be locally
189 // hurt?
190 if (level->isClientSide) return false;
191
192 if (isInvulnerable()) return false;
193 if (getItem() != nullptr && getItem()->id == Item::netherStar_Id &&
194 source->isExplosion())
195 return false;
196 markHurt();
197 health -= damage;
198 if (health <= 0) {
199 remove();
200 }
201 return false;
202}
