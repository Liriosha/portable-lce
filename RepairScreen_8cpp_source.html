157void RepairScreen::updateItemName() {
158 std::string itemName;
159 Slot* slot = repairMenu->getSlot(0);
160 if (slot != nullptr && slot->hasItem()) {
161 if (!slot->getItem()->hasCustomHoverName() &&
162 itemName == slot->getItem()->getHoverName()) {
163 itemName = "";
164 }
165 }
166
167 repairMenu->setItemName(itemName);
168
169 ByteArrayOutputStream baos;
170 DataOutputStream dos(&baos);
171 dos.writeUTF(itemName);
172 Minecraft::GetInstance()->player->connection->send(
173 std::make_shared<CustomPayloadPacket>(
174 CustomPayloadPacket::SET_ITEM_NAME_PACKET, baos.toByteArray()));
175}
