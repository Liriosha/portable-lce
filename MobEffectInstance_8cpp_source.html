49void MobEffectInstance::update(MobEffectInstance* takeOver) {
50 if (id != takeOver->id) {
51 Log::info("This method should only be called for matching effects!");
52 }
53 if (takeOver->amplifier > amplifier) {
54 amplifier = takeOver->amplifier;
55 duration = takeOver->duration;
56 } else if (takeOver->amplifier == amplifier &&
57 duration < takeOver->duration) {
