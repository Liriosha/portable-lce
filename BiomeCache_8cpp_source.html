77BiomeCache::Block* BiomeCache::getBlockAt(int x, int z) {
78 std::lock_guard<std::mutex> lock(m_CS);
79 x >>= ZONE_SIZE_BITS;
80 z >>= ZONE_SIZE_BITS;
81 int64_t slot =
82 (((int64_t)x) & 0xffffffffl) | ((((int64_t)z) & 0xffffffffl) << 32l);
83 auto it = cached.find(slot);
84 Block* block = nullptr;
85 if (it == cached.end()) {
86 block = new Block(x, z, this);
87 cached[slot] = block;
88 all.push_back(block);
89 } else {
90 block = it->second;
91 }
92 block->lastUse = gameServices().getAppTime();
93 return block;
94}

108void BiomeCache::update() {
109 std::lock_guard<std::mutex> lock(m_CS);
110 int64_t now = gameServices().getAppTime();
111 int64_t utime = now - lastUpdateTime;
112 if (utime > DECAY_TIME / 4 || utime < 0) {
113 lastUpdateTime = now;
114
115 for (auto it = all.begin(); it != all.end();) {
116 Block* block = *it;
117 int64_t time = now - block->lastUse;
118 if (time > DECAY_TIME || time < 0) {
119 it = all.erase(it);
120 int64_t slot = (((int64_t)block->x) & 0xffffffffl) |
121 ((((int64_t)block->z) & 0xffffffffl) << 32l);
122 cached.erase(slot);
123 delete block;
124 } else {
125 ++it;
126 }
127 }
128 }
